19 TEST_F(eigen3, Diagonal_class)
20 {
21  DiagonalMatrix<native_matrix_t<double, 3, 1>> d1;
22  d1 << 1, 2, 3;
23  EXPECT_TRUE(is_near(d1.nested_matrix(), make_native_matrix<double, 3, 1>(1, 2, 3)));
24  DiagonalMatrix d2 = make_native_matrix(1., 2, 3);
25  EXPECT_TRUE(is_near(d2, d1));
26  DiagonalMatrix d3(d2);
27  EXPECT_TRUE(is_near(d3, d1));
28  DiagonalMatrix d4(DiagonalMatrix<native_matrix_t<double, 3, 1>>{1., 2, 3});
29  EXPECT_TRUE(is_near(d4, d1));
30  DiagonalMatrix d5 = MatrixTraits<DiagonalMatrix<native_matrix_t<double, 3, 1>>>::zero();
31  EXPECT_TRUE(is_near(d5, native_matrix_t<double, 3, 3>::Zero()));
32  DiagonalMatrix d6 = MatrixTraits<DiagonalMatrix<native_matrix_t<double, 3, 1>>>::identity();
33  EXPECT_TRUE(is_near(d6, native_matrix_t<double, 3, 3>::Identity()));
34  DiagonalMatrix d7 = ((0.7 * Mat3::Identity()) * (0.3 * Mat3::Identity() * 0.7 + 0.7 * Mat3::Identity()) - Mat3::Identity() * 0.3);
35  EXPECT_TRUE(is_near(d7, native_matrix_t<double, 3, 3>::Identity() * 0.337));
36  DiagonalMatrix<native_matrix_t<double, 3, 1>> d8 = ((0.7 * Mat3::Identity()) * (0.3 * Mat3::Identity() * 0.7 + 0.7 * Mat3::Identity()) - Mat3::Identity() * 0.3);
37  EXPECT_TRUE(is_near(d8, native_matrix_t<double, 3, 3>::Identity() * 0.337));
38  DiagonalMatrix<native_matrix_t<double, 3, 1>> d9 = ZeroMatrix<native_matrix_t<double, 3, 3>>();
39  EXPECT_TRUE(is_near(d9, native_matrix_t<double, 3, 3>::Zero()));
40  DiagonalMatrix<native_matrix_t<double, 3, 1>> d10 = native_matrix_t<double, 3, 3>::Identity();
41  EXPECT_TRUE(is_near(d10, native_matrix_t<double, 3, 3>::Identity()));
42  EXPECT_TRUE(is_near(DiagonalMatrix<native_matrix_t<double, 3, 1>>(ZeroMatrix<native_matrix_t<double, 3, 1>>()), native_matrix_t<double, 3, 3>::Zero()));
43  d3 = d2;
44  EXPECT_TRUE(is_near(d3, d1));
45  d4 = DiagonalMatrix {1., 2, 3};
46  EXPECT_TRUE(is_near(d4, d1));
47  d4 = {1., 2, 3};
48  EXPECT_TRUE(is_near(d4, d1));
49  d1 += d2;
50  EXPECT_TRUE(is_near(d1, DiagonalMatrix {2., 4, 6}));
51  d1 -= DiagonalMatrix {1., 2, 3};
52  EXPECT_TRUE(is_near(d1, DiagonalMatrix {1., 2, 3}));
53  d1 *= 3;
54  EXPECT_TRUE(is_near(d1, DiagonalMatrix {3., 6, 9}));
55  d1 /= 3;
56  EXPECT_TRUE(is_near(d1, DiagonalMatrix {1., 2, 3}));
57  d1 *= d2;
58  EXPECT_TRUE(is_near(d1, DiagonalMatrix {1., 4, 9}));
59  EXPECT_TRUE(is_near(d1.square_root(), DiagonalMatrix {1., 2, 3}));
60  EXPECT_TRUE(is_near(d1.square(), DiagonalMatrix {1., 16, 81}));
61 }
