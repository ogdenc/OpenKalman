18using M_int2 = Mean<Axes<2>, eigen_matrix_t<int, 2, 1>>;

21template<auto t, auto tn = t>
22struct Scale
23{
24 template<typename Arg, typename ... Noise>
25 auto operator()(const Arg& in, const Noise& ... n) const
26 {
27 return make_self_contained(((t * in) + ... + (tn * n)));
28 }
29};

31TEST(transformations, Scale_no_noise_1D)
32{
33 using M_int1 = Mean<Coefficients<Axis>, eigen_matrix_t<int, 1, 1>>;
34 Transformation<Scale<3>> t;
35 EXPECT_EQ(t(M_int1 {2}), M_int1 {6});
36}

38TEST(transformations, Scale_no_noise_2D)
39{
40 Transformation<Scale<7>> t;

42}

44TEST(transformations, Scale_additive)
45{
46 Transformation<Scale<7>> t;
47 EXPECT_EQ(t(M_int2 {2, 3}), (M_int2 {14, 21}));
48 EXPECT_EQ(t(M_int2(2, 3)) + M_int2(1, 1), M_int2(15, 22));
49 static_assert(std::is_same_v<decltype(t(M_int2(2, 3)) + M_int2(1, 1)), decltype(M_int2(15, 22))>);
50 EXPECT_EQ(t(M_int2(2, 3)) + M_int2(3, 3), M_int2(17, 24));
51}

53TEST(transformations, Scale_augmented)

107TEST(transformations, Identity)
108{
109 IdentityTransformation t;
