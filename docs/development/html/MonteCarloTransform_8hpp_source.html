119  struct MonteCarloBinaryOp
120  {
121  auto operator()(const MonteCarloSum& set1, const MonteCarloSum& set2)
122  {
123  using Scalar = scalar_type_of_t<decltype(set1.y_E)>;
124  if (set2.count == 1) // This is most likely. Take advantage of prior knowledge of set2.
125  {
126  const auto count = set1.count + 1;
127  const Scalar s_count = count;
128  const auto x = (set1.count * set1.x + set2.x) / s_count;
129  const auto y_E = (set1.count * set1.y_E + set2.y_E) / s_count;
130  const auto delta = from_euclidean(set2.y_E) - from_euclidean(set1.y_E);
131  const auto delta_adj_factor = adjoint(delta) * set1.count / s_count;
132  const OutputCovariance yy {set1.yy + delta * delta_adj_factor};
133  if constexpr (return_cross)
134  {
135  const auto xy = set1.xy + (set2.x - set1.x) * delta_adj_factor;
136  return MonteCarloSum {count, x, y_E, yy, xy};
137  }
138  else
139  {
140  return MonteCarloSum {count, x, y_E, yy};
141  }
142  }
143  else
144  {
145  const auto count = set1.count + set2.count;
146  const Scalar s_count = count;
147  const auto x = (set1.count * set1.x + set2.count * set2.x) / s_count;
148  const auto y_E = (set1.count * set1.y_E + set2.count * set2.y_E) / s_count;
149  const auto delta = from_euclidean(set2.y_E) - from_euclidean(set1.y_E);
150  const auto delta_adj_factor = adjoint(delta) * set1.count * set2.count / s_count;
151  const OutputCovariance yy {set1.yy + set2.yy + delta * delta_adj_factor};
152  if constexpr (return_cross)
153  {
154  const auto xy = set1.xy + set2.xy + (set2.x - set1.x) * delta_adj_factor;
155  return MonteCarloSum {count, x, y_E, yy, xy};
156  }
157  else
158  {
159  return MonteCarloSum {count, x, y_E, yy};
160  }
161  }
162  }
