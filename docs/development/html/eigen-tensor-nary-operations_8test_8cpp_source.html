28 TEST(eigen_tensor, nullary_operations)
29 {
30  auto a232 = make_dense_object<M22>(N2{}, N3{}, N2{});
31  using A232 = decltype(a232);
32  EXPECT_TRUE(is_near(a232.nullaryExpr([](auto x){ return 7; }), a232.constant(7)));
33 
34  static_assert(self_contained<decltype(a232.constant(7))>);
35  using T2 = Eigen::TensorFixedSize<double, Eigen::Sizes<2,2>>;
36  using I2 = Eigen::TensorCwiseNullaryOp<Eigen::internal::scalar_identity_op<double>, T2>;
37  static_assert(constant_diagonal_coefficient_v<I2> == 1);
38  using T2d = Eigen::Tensor<double, 2>;
39  using I2d = Eigen::TensorCwiseNullaryOp<Eigen::internal::scalar_identity_op<double>, T2d>;
40  static_assert(constant_diagonal_coefficient_v<I2d> == 1);
41  EXPECT_EQ(constant_coefficient {a232.constant(7)}(), 7);
42  EXPECT_EQ(constant_coefficient {a232.nullaryExpr([]{ return 7.; })}(), 7.);
43 
44  auto b3 = make_dense_object<A232>(3);
45  using B3 = decltype(b3);
46  b3.setValues({0, 1, 2});
47  EXPECT_TRUE(is_near(n_ary_operation<B3>(std::tuple {Dimensions<3>{}}, [](std::size_t i){return 3.5 + i;}), b3.constant(3.5) + b3));
