133  EXPECT_TRUE(is_near(average_reduce<1>(M21 {1, 4}), M21 {1, 4}));
134  EXPECT_TRUE(is_near(average_reduce<0>(make_eigen_matrix<double, 1, 3>(1, 2, 3)), make_eigen_matrix<double, 1, 3>(1, 2, 3)));
135 
136  auto m23 = make_dense_object_from<M23>(1, 2, 3, 4, 5, 6);
137  auto m2x_3 = M2x {m23};
138  auto mx3_2 = Mx3 {m23};
139  auto mxx_23 = Mxx {m23};
140 
141  auto m21_25 = make_eigen_matrix<double, 2, 1>(2, 5);
142 
143  EXPECT_TRUE(is_near(average_reduce<1>(m23), m21_25));
144  EXPECT_TRUE(is_near(average_reduce<1>(Eigen3::make_eigen_wrapper(m23)), m21_25));
145  EXPECT_TRUE(is_near(average_reduce<1>(mx3_2), m21_25));

147  EXPECT_TRUE(is_near(average_reduce<1>(mxx_23), m21_25));

151  auto ix2_2 = Mx2::Identity(2, 2);
152  auto ixx_22 = Mxx::Identity(2, 2);
153 
154  EXPECT_TRUE(is_near(average_reduce<1>(ixx_22), M21::Constant(0.5)));
