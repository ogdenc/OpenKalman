74 auto covariance_op(const F1& f1, const F2& f2) const &
75 {
76 using Derived2 = decltype(MatrixTraits<Derived>::make(f1(nested_matrix())));
77 using R = typename MatrixTraits<Derived2>::SelfContainedFrom;
78 if (synchronization_direction() >= 0)
79 {
80 R r {f1(nested_matrix())};
81 if constexpr (not case1or2<Derived, decltype(f1(nested_matrix()))>)
82 {
83 if (r.synchronization_direction() > 0 and synchronization_direction() <= 0)
84 {
85 r.cholesky_nested_matrix() =
86 to_covariance_nestable<decltype(r.cholesky_nested_matrix())>(f2(cholesky_nested_matrix()));
87 r.mark_synchronized();
88 }
89 }
90 return r;
91 }
92 else
93 {
94 return R {f2(cholesky_nested_matrix())};
95 }
96 }

286 auto determinant() const
287 {
288 if constexpr (square_root_covariance<Derived> and not triangular_matrix<NestedMatrix>)
289 {
290 if (synchronization_direction() > 0)
291 {
292 return std::sqrt(OpenKalman::determinant(nested_matrix()));
293 }
294 else
295 {
296 return OpenKalman::determinant(cholesky_nested_matrix());
297 }
298 }
299 else if constexpr (not square_root_covariance<Derived> and not self_adjoint_matrix<NestedMatrix>)
300 {
301 if (synchronization_direction() > 0)
302 {
303 auto d = OpenKalman::determinant(nested_matrix());
304 return d * d;
305 }
306 else
307 {
308 return OpenKalman::determinant(cholesky_nested_matrix());
309 }
310 }
311 else
312 {
313 if (synchronization_direction() < 0)
314 {
315 if constexpr (square_root_covariance<Derived>)
316 {
317 return std::sqrt(OpenKalman::determinant(cholesky_nested_matrix()));
318 }
319 else
320 {
321 auto d = OpenKalman::determinant(cholesky_nested_matrix());
322 return d * d;
323 }
324 }
325 else
326 {
327 return OpenKalman::determinant(nested_matrix());
