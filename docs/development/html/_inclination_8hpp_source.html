163  from_euclidean_array =
164  {
165  [](const GetCoeff<Scalar>& get_coeff)
166  {
167  constexpr Scalar up = Limits<Scalar>::up;
168  constexpr Scalar down = Limits<Scalar>::down;
169  const auto x = get_coeff(i);
170  const auto y = get_coeff(i + 1);
171  if constexpr (up != -down)
172  {
173  return std::atan2(y, std::abs(x)) / cf<Scalar>;
174  }
175  else
176  {
177  constexpr Scalar range = up - down;
178  constexpr Scalar period = range * 2;
179  auto a = std::atan2(y, x) / cf<Scalar>;
180  if (a < down) return down - a;
181  if (a > range) a = down + period - a;

183  }
