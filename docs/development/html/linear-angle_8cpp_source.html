19using C2 = Coefficients<Polar<>>;
20using M2 = Mean<C2>;
21using Mat2 = Matrix<C2, C2>;
22using Cov2 = Covariance<C2>;
23
24TEST(transform_linear, linear_additive_angle)
25{
26 auto a = Mat2::identity() * 1.1;
27 const GaussianDistribution input {M2(1, pi * 19 / 20), Cov2(pi * pi / 9, 0, 0, 0.01)};
28 Cov2 P_output {1.21 * pi * pi / 9, 0,
29 0, 0.0121};
30 const GaussianDistribution output {M2(1.1, 1.1 * 19 / 20 * pi - 2 * pi), P_output};
31 Mat2 cross_output {1.1 * pi * pi / 9, 0,
32 0, 0.011};
33 LinearTransformation g {a};
34 LinearTransform t;
35 auto [out, cross] = t.transform_with_cross_covariance(input, g);
36 EXPECT_TRUE(is_near(out, output));
37 EXPECT_TRUE(is_near(cross, cross_output));
38}
