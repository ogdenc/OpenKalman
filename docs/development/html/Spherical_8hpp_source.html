94 #endif
95  {
96  if (euclidean_local_index == d2_i)
97  {
98  return g(start + d_i);

101  {
102  using Scalar = std::decay_t<decltype(g(std::declval<std::size_t>()))>;
103  using R = std::decay_t<decltype(internal::constexpr_real(std::declval<Scalar>()))>;
104  const Scalar cf_inc {numbers::pi_v<R> / (InclinationLimits::up - InclinationLimits::down)};
105  const Scalar horiz {R{InclinationLimits::up + InclinationLimits::down} * R{0.5}};
106 
107  Scalar phi = cf_inc * (g(start + i_i) - horiz);
108  if (euclidean_local_index == z_i)
109  {
110  using std::sin;
111  return sin(phi);
112  }
113  else
114  {

116  const Scalar mid {R{CircleLimits::max + CircleLimits::min} * R{0.5}};
117  Scalar theta = cf_cir * (g(start + a_i) - mid);
118 
119  using std::cos, std::sin;
120  if (euclidean_local_index == x_i) return cos(theta) * cos(phi);
121  else return sin(theta) * cos(phi); // euclidean_local_index == y_i

136  static constexpr scalar_type auto
137  from_euclidean_element(const auto& g, std::size_t local_index, std::size_t euclidean_start)
138  requires requires (std::size_t i){ {g(i)} -> scalar_type; }
139 #else
140  template<typename G, std::enable_if_t<scalar_type<typename std::invoke_result<G, std::size_t>::type>, int> = 0>
141  static constexpr auto
142  from_euclidean_element(const G& g, std::size_t local_index, std::size_t euclidean_start)
143 #endif
144  {
145  using Scalar = decltype(g(std::declval<std::size_t>()));
146  Scalar d = g(euclidean_start + d2_i);
147  auto dr = internal::constexpr_real(d);
148 
149  if (local_index == d_i)
150  {
151  using std::abs;
152  return internal::update_real_part(d, abs(dr));
153  }
154  else
155  {
156  using R = std::decay_t<decltype(internal::constexpr_real(std::declval<Scalar>()))>;
157  const Scalar cf_cir {2 * numbers::pi_v<R> / (CircleLimits::max - CircleLimits::min)};
158  const Scalar mid {R{CircleLimits::max + CircleLimits::min} * R{0.5}};
159 
160  Scalar x = g(euclidean_start + x_i);
161  Scalar y = g(euclidean_start + y_i);
162 
163  switch(local_index)
164  {
165  case a_i:
166  {
167  auto xp = internal::constexpr_real(g(euclidean_start + x_i));
168  auto yp = internal::constexpr_real(g(euclidean_start + y_i));
169  // If distance is negative, flip x and y axes 180 degrees:
170  using std::signbit;
171  Scalar x2 = internal::update_real_part(x, signbit(dr) ? -xp : xp);
172  Scalar y2 = internal::update_real_part(y, signbit(dr) ? -yp : yp);
173 
174  if constexpr (complex_number<Scalar>) return internal::constexpr_atan2(y2, x2) / cf_cir + mid;
175  else { using std::atan2; return atan2(y2, x2) / cf_cir + mid; }
176  }
177  default: // case i_i
178  {
179  const Scalar cf_inc {numbers::pi_v<R> / (InclinationLimits::up - InclinationLimits::down)};
180  const Scalar horiz {R{InclinationLimits::up + InclinationLimits::down} * R{0.5}};
181  Scalar z {g(euclidean_start + z_i)};
182  auto zp = internal::constexpr_real(z);
183  using std::signbit;
184  Scalar z2 {internal::update_real_part(z, signbit(dr) ? -zp : zp)};
185  using std::sqrt;
186  Scalar r {sqrt(x*x + y*y + z2*z2)};
187  if constexpr (complex_number<Scalar>)
188  {
189  using std::asin;
190  auto theta = (r == Scalar{0}) ? r : asin(z2/r);
191  return theta / cf_inc + horiz;
192  }
193  else
194  {
195  using std::asin;
196  using R = std::decay_t<decltype(asin(z2/r))>;
197  // This is so that a zero-radius or faulty spherical coordinate has horizontal inclination:
198  auto theta = (r == 0 or r < z2 or z2 < -r) ? R{0} : asin(z2/r);
199  return theta / cf_inc + horiz;
200  }

210  template<typename Scalar>
211  static constexpr auto
212  inclination_wrap_impl(const Scalar& a)
213  {
214  using Ret = std::tuple<std::decay_t<std::decay_t<decltype(internal::constexpr_real(a))>>, bool>;
215  auto ap = internal::constexpr_real(a);
216  using R = std::decay_t<decltype(ap)>;
217  if (ap >= InclinationLimits::down and ap <= InclinationLimits::up) // A shortcut, for the easy case.
218  {
219  return Ret { ap, false };
220  }
221  else
222  {
223  constexpr R period = 2 * (InclinationLimits::up - InclinationLimits::down);
224  using std::fmod;
225  R ar = fmod(ap - R{InclinationLimits::down}, period);
226  R ar2 = ar < 0 ? ar + period : ar;
227  bool b = ar2 > InclinationLimits::up - InclinationLimits::down; // Whether there is a mirror reflection about vertical axis.
228  return Ret { R{InclinationLimits::down} + (b ? period - ar2 : ar2), b };
229  }

233  template<typename Scalar>
234  static constexpr std::decay_t<Scalar>
235  azimuth_wrap_impl(bool reflect_azimuth, Scalar&& a)
236  {
237  using R = std::decay_t<decltype(internal::constexpr_real(std::declval<decltype(a)>()))>;
238  constexpr R period {CircleLimits::max - CircleLimits::min};
239  constexpr R half_period {(CircleLimits::max - CircleLimits::min) / R{2}};
240  R ap = reflect_azimuth ? internal::constexpr_real(a) - half_period : internal::constexpr_real(a);
241 
242  if (ap >= CircleLimits::min and ap < CircleLimits::max) // Check if angle doesn't need wrapping.
243  {
244  return internal::update_real_part(std::forward<decltype(a)>(a), ap);;
245  }
246  else // Wrap the angle.
247  {
248  using std::fmod;
249  auto ar = fmod(ap - R{CircleLimits::min}, period);
250  if (ar < 0) return internal::update_real_part(std::forward<decltype(a)>(a), R{CircleLimits::min} + ar + period);
251  else return internal::update_real_part(std::forward<decltype(a)>(a), R{CircleLimits::min} + ar);
252  }
253  }
