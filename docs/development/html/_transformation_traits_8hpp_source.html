150  {
151  template<typename OutputCoefficients, typename In>
152  inline auto zero_hessian_impl()
153  {
154  using InputCoefficients = typename MatrixTraits<In>::RowCoefficients;
155  constexpr std::size_t input_size = InputCoefficients::size;
156  constexpr std::size_t output_size = OutputCoefficients::size;
157  using HessianMatrixInBase = native_matrix_t<In, input_size, input_size>;
158  using HessianMatrixIn = Matrix<InputCoefficients, InputCoefficients, HessianMatrixInBase>;
159  using HessianArrayIn = std::array<HessianMatrixIn, output_size>;
160 
161  HessianArrayIn a;
162  a.fill(HessianMatrixIn::zero());
163  return a;
164  }
165  }
