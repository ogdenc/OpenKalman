247  inline auto
248  split_item_impl(Arg&& arg)
249  {
250  if constexpr(one_dimensional<Arg> and self_adjoint_covariance<Expr> and cholesky_form<Expr>)
251  {

268  inline auto split_cov_diag_impl(const F& f, Arg&& arg)
269  {
270  if constexpr(one_dimensional<Arg> and self_adjoint_covariance<Expr> and cholesky_form<Expr>)

275  {
