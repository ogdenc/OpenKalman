11 #ifndef OPENKALMAN_TYPEDMATRIXBASE_H
12 #define OPENKALMAN_TYPEDMATRIXBASE_H
13 
14 namespace OpenKalman::internal
15 {
16  /// Base class for means or matrices.
17  template<
18  typename Derived,
19  typename RowCoeffs,
20  typename ColCoeffs,
21  typename NestedType>
22  struct TypedMatrixBase : internal::MatrixBase<Derived, NestedType>
23  {
24  using Base = internal::MatrixBase<Derived, NestedType>;
25  using RowCoefficients = RowCoeffs;
26  using ColumnCoefficients = ColCoeffs;
27  using NestedMatrix = NestedType; ///< The nested class.
28  using Scalar = typename MatrixTraits<NestedMatrix>::Scalar; ///< Scalar type for this variable.
29  static constexpr auto dimension = MatrixTraits<NestedMatrix>::dimension; ///< Dimension of the vector.
30  static constexpr auto columns = MatrixTraits<NestedMatrix>::columns; ///< Number of columns.
31 
32  /***************
33  * Constructors
34  ***************/
35 
36  /// Default constructor.
37  TypedMatrixBase() : Base() {}
38 
39  /// Copy constructor.
40  TypedMatrixBase(const TypedMatrixBase& other) : Base(other.nested_matrix()) {}
41 
42  /// Move constructor.
43  TypedMatrixBase(TypedMatrixBase&& other) noexcept : Base(std::move(other).nested_matrix()) {}
44 
45  /// Construct from a typed_matrix_nestable.
46 #ifdef __cpp_concepts
47  template<typed_matrix_nestable Arg>
48 #else
49  template<typename Arg, std::enable_if_t<typed_matrix_nestable<Arg>, int> = 0>
50 #endif
51  TypedMatrixBase(Arg&& arg) noexcept : Base(std::forward<Arg>(arg))
52  {
53  static_assert(MatrixTraits<Arg>::dimension == dimension);
54  static_assert(MatrixTraits<Arg>::columns == columns);
55  }
56 
57  /// Construct from a list of coefficients.
58  template<typename ... Args, std::enable_if_t<std::conjunction_v<std::is_convertible<Args, const Scalar>...>, int> = 0>
59  TypedMatrixBase(Args ... args) : Base(MatrixTraits<NestedMatrix>::make(args...)) {}
60 
61  /**********************
62  * Assignment Operators
63  **********************/
64 
65  using Base::operator=;
66 
67  /// Multiply by a scale factor.
68  template<typename S, std::enable_if_t<std::is_convertible_v<S, Scalar>, int> = 0>
69  auto& operator*=(const S s)
70  {
71  this->nested_matrix() *= s;
72  return *this;
73  }
74 
75  /// Divide by a scale factor.
76  template<typename S, std::enable_if_t<std::is_convertible_v<S, Scalar>, int> = 0>
77  auto& operator/=(const S s)
78  {
79  this->nested_matrix() /= s;
80  return *this;
81  }

83  auto operator()(std::size_t i, std::size_t j) &
84  {
85  return make_ElementSetter<not element_settable<Derived, 2>>(static_cast<Derived&>(*this), i, j);
86  }
87 
88  auto operator()(std::size_t i, std::size_t j) &&
89  {
90  return make_ElementSetter<true>(static_cast<Derived&&>(*this), i, j);
91  }
92 
93  auto operator()(std::size_t i, std::size_t j) const &
94  {
95  return make_ElementSetter<true>(static_cast<const Derived&>(*this), i, j);
96  }
97 
98  auto operator()(std::size_t i, std::size_t j) const &&
99  {
100  return make_ElementSetter<true>(static_cast<const Derived&&>(*this), i, j);
101  }
102 
103  auto operator[](std::size_t i) &
104  {
105  return make_ElementSetter<not element_settable<Derived, 1>>(static_cast<Derived&>(*this), i);
106  }
107 
108  auto operator[](std::size_t i) &&
109  {
110  return make_ElementSetter<true>(static_cast<Derived&&>(*this), i);
111  }
112 
113  auto operator[](std::size_t i) const &
114  {
115  return make_ElementSetter<true>(static_cast<const Derived&>(*this), i);
116  }
117 
118  auto operator[](std::size_t i) const &&
119  {
120  return make_ElementSetter<true>(static_cast<const Derived&&>(*this), i);
121  }
122 
123  auto operator()(std::size_t i) { return operator[](i); }
124 
125  auto operator()(std::size_t i) const { return operator[](i); }
126  };
