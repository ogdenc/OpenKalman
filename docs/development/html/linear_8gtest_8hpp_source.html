23  template<typename Jacobians, typename Covs, std::size_t...ints>
24  constexpr auto sumprod(const Jacobians& j, const Covs& c, std::index_sequence<ints...>)
25  {
26  return make_self_contained(((std::get<ints>(j) * std::get<ints>(c) * adjoint(std::get<ints>(j))) + ...));
27  }

31  template<typename Transformation, typename Transform, typename InputDist, typename ... Noise>
32  ::testing::AssertionResult run_linear_test(
33  double err,
34  const Transformation& g,
35  const Transform& t,
36  InputDist&& in,
37  Noise&& ... noise)
38  {
39  auto x = mean_of(in);
40  auto p = covariance_of(in);
41  auto y = g(x, mean_of(noise)...);
42  auto [a] = g.jacobian(x);
43  auto cross_cov = p*adjoint(a);
44  auto jacobians = g.jacobian(x, mean_of(noise)...);
45  auto covariances = std::forward_as_tuple(std::move(p), covariance_of(noise)...);
46  auto cov = detail::sumprod(jacobians, covariances, std::make_index_sequence<sizeof...(Noise) + 1> {});
47  auto out_true = make_tuple(GaussianDistribution {y, cov}, std::move(cross_cov));
48 
49  auto out = t.transform_with_cross_covariance(std::forward<InputDist>(in), g, std::forward<Noise>(noise)...);
50  auto res = is_near(out, out_true, err);
51  if (res)
52  return ::testing::AssertionSuccess();
53  else
54  return ::testing::AssertionFailure() << res.message();
55  }

85  template<std::size_t IN, std::size_t OUT, typename Cov, typename T>
86  void run_multiple_linear_tests(Cov cov, const T& t, double err = 1e-6, int N = 5)
87  {
88  using MatIn = Matrix<Dimensions<OUT>, Dimensions<IN>, eigen_matrix_t<double, OUT, IN>>;
89  using MatNoise = Matrix<Dimensions<OUT>, Dimensions<OUT>, eigen_matrix_t<double, OUT, OUT>>;
90  using MIn = Mean<Dimensions<IN>, eigen_matrix_t<double, IN, 1>>;
91  using MNoise = Mean<Dimensions<OUT>, eigen_matrix_t<double, OUT, 1>>;
92  for (int i=1; i<=N; i++)
93  {
94  auto a = randomize<MatIn>(std::uniform_real_distribution<double> {-i, i});
95  auto n = randomize<MatNoise>(std::uniform_real_distribution<double> {-0.1 * i, 0.1 * i});
96  auto g = LinearTransformation(a, n);
97  auto in = GaussianDistribution {make_zero<MIn>(), cov};
98  auto b = randomize<MNoise>(std::normal_distribution<double> {0., i*2.});
99  auto noise_cov = Covariance {0.5 * eigen_matrix_t<double, OUT, OUT>::Identity()};
100  auto noise = GaussianDistribution {b, noise_cov};
101  EXPECT_TRUE(run_linear_test(err, g, t, in, noise));
102  }
103  }

106  template<std::size_t DIM, typename Cov>
107  void run_multiple_identity_tests(Cov&& cov, double err = 1e-6, int N = 5)
108  {
109  using MatIn = Matrix<Dimensions<DIM>, Dimensions<DIM>, eigen_matrix_t<double, DIM, DIM>>;
110  using MatNoise = Matrix<Dimensions<DIM>, Dimensions<DIM>, eigen_matrix_t<double, DIM, DIM>>;
111  using MIn = Mean<Dimensions<DIM>, eigen_matrix_t<double, DIM, 1>>;
112  using MNoise = Mean<Dimensions<DIM>, eigen_matrix_t<double, DIM, 1>>;
113  for (int i=1; i<=N; i++)
114  {
115  auto a = make_identity_matrix_like<MatIn>();
116  auto n = make_identity_matrix_like<MatNoise>();
117  auto g = LinearTransformation(a, n);
118  auto t = IdentityTransform();
119  auto in = GaussianDistribution {make_zero<MIn>(), make_self_contained(i * std::forward<Cov>(cov))};
120  auto b = randomize<MNoise>(std::normal_distribution {0., i*2.});
121  auto noise_cov = Covariance {i / 5. * eigen_matrix_t<double, DIM, DIM>::Identity()};
122  auto noise = GaussianDistribution {b, noise_cov};
123  EXPECT_TRUE(run_identity_test(err, g, t, in, noise));
124  }
125  }
