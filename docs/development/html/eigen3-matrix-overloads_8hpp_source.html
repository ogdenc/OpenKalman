315  template<eigen_native A>
316 #else
317  template<typename A, std::enable_if_t<eigen_native<A>, int> = 0>
318 #endif
319  constexpr auto
320  LQ_decomposition(A&& a)
321  {
322  if constexpr (diagonal_matrix<A> or lower_triangular_matrix<A>)
323  {
324  return std::forward<A>(a);
325  }
326  else
327  {
328  using Scalar = typename MatrixTraits<A>::Scalar;
329  constexpr auto dim = MatrixTraits<A>::dimension, col = MatrixTraits<A>::columns;
330  using MatrixType = Eigen::Matrix<Scalar, col, dim>;
331  using ResultType = Eigen::Matrix<Scalar, dim, dim>;
332  Eigen::HouseholderQR<MatrixType> QR(std::forward<A>(a).adjoint());
333  ResultType ret;
334  if constexpr(col < dim)
335  {
336  ret << QR.matrixQR().template topRows<col>().adjoint(), Eigen::Matrix<Scalar, dim, dim - col>::Zero();
337  }
338  else
339  {
340  ret = QR.matrixQR().template topRows<dim>().adjoint();
341  }
342  using TType = typename MatrixTraits<ResultType>::template TriangularBaseType<TriangleType::lower>;
343  return MatrixTraits<TType>::make(std::move(ret));
344  }
345  }

454  // Concatenate one or more Eigen::MatrixBase objects diagonally.
455  template<typename M, typename ... Vs, std::size_t ... ints>
456  void
457  concatenate_diagonal_impl(M& m, const std::tuple<Vs...>& vs, std::index_sequence<ints...>)
458  {
459  ((m << std::get<0>(vs)), ..., [&]
460  {
461  constexpr auto row = (ints + 1) / sizeof...(Vs);
462  constexpr auto col = (ints + 1) % sizeof...(Vs);
463  constexpr auto row_size = MatrixTraits<decltype(std::get<row>(vs))>::dimension;
464  constexpr auto col_size = MatrixTraits<decltype(std::get<col>(vs))>::columns;
465  if constexpr (row == col) return std::get<row>(vs);
466  else return Eigen::Matrix<typename M::Scalar, row_size, col_size>::Zero();
467  }());
468  }
469  }

549  using CC = Axes<MatrixTraits<Arg>::columns>;
550  constexpr Eigen::Index dim1 = RC_size, dim2 = std::decay_t<Arg>::RowsAtCompileTime - dim1;

552  if constexpr (std::is_lvalue_reference_v<Arg&&>)
553  return std::forward<decltype(m)>(m);
554  else
555  return make_native_matrix(std::forward<decltype(m)>(m));
556  };

560  auto split2 = g(arg.template bottomRows<dim2>());

562  internal::make_split_tuple<F, RC, CC>(std::move(split1)),
563  split_vertical<F, euclidean, RCs...>(std::move(split2)));
564  }
565  else if constexpr (dim1 < std::decay_t<Arg>::RowsAtCompileTime)
566  {
567  return internal::make_split_tuple<F, RC, CC>(g(arg.template topRows<dim1>()));
568  }
569  else
