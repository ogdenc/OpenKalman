20  * Simon J. Julier. The spherical simplex unscented transformation.
21  * In Proceedings of American Control Conference, Denver, Colorado, pages 2430â€“2434, 2003.

27  struct SphericalSimplexParameters
28  {
29  /// Scaling factor (typically 0.001 but may be from 0.0001 to 1).
30  static constexpr double alpha = 0.001;
31  /// Factor to compensate for the distribution (beta==2 is optimal for Gaussian distributions).
32  static constexpr double beta = 2.0;
33  /// The first weight, 0 <= W0 <= 1. It is a free parameter that affects the fourth and higher moments of the sigma point set.
34  /// Here, for simplicity, it is equal to the other weights.
35  template<int dim> static constexpr double W0 = 1 / (dim + 2);
36  };
37 
38 
39  template<typename Parameters = SphericalSimplexParameters>
40  struct SphericalSimplex : internal::ScaledSigmaPointsBase<SphericalSimplex<Parameters>, Parameters>
41  {
42  /**
43  * Number of sigma points.
44  * \tparam dim Number of dimensions of the input variable.
45  * \return Number of sigma points.
46  */
47  template<std::size_t dim>
48  static constexpr size_t
49  sigma_point_count() { return dim + 2; };
50 
51  private:
52  /// Compile time sqrt.
53  template<typename Scalar>
54  static constexpr Scalar constexpr_sqrt(Scalar x, Scalar guess)
55  {
56  return (0.25 * (guess + x / guess) * (guess + x / guess)/x - 1 <= std::numeric_limits<Scalar>::epsilon()) ?
57  (0.5 * (guess + x / guess)) :
58  constexpr_sqrt(x, 0.5 * (guess + x / guess));
59  }
60 
61  template<typename Scalar>
62  static constexpr Scalar constexpr_sqrt(Scalar x) { return constexpr_sqrt(x, 1.); }
63 
64  template<std::size_t j, std::size_t i, std::size_t dim, typename Scalar>
65  static constexpr auto
66  sigma_point_coeff()
67  {
68  constexpr auto denom = 1 / constexpr_sqrt((j + 1) * (j + 2) * unscaled_W<dim, Scalar>());
69  if constexpr(i == 0)
70  return Scalar(0);
71  else if constexpr(i < j + 2)
72  return -denom;
73  else if constexpr (i == j + 2)
74  return (j + 1) * denom;
75  else
76  return Scalar(0);
77  }
78 
79  template<typename Dist, std::size_t dim, std::size_t pos, std::size_t...ns>
80  static constexpr auto
81  unscaled_sigma_points(std::index_sequence<ns...>)
82  {
83  using Coefficients = typename DistributionTraits<Dist>::Coefficients;
84  using Scalar = typename DistributionTraits<Dist>::Scalar;
85  constexpr auto rows = DistributionTraits<Dist>::dimension;
86  constexpr auto count = sigma_point_count<dim>();
87  using Xnative = native_matrix_t<typename DistributionTraits<Dist>::Mean, rows, count>;
88  Matrix<Coefficients, Axes<count>, Xnative> X {sigma_point_coeff<ns / count + pos, ns % count, dim, Scalar>()...};
89  return X;
90  }
91 
92  template<std::size_t dim, std::size_t pos = 0, typename D, typename...Ds>
93  static auto
94  sigma_points_impl(const D& d, const Ds&...ds)
95  {
96  constexpr auto rows = DistributionTraits<D>::dimension;
97  constexpr auto count = sigma_point_count<dim>();
98  auto X = unscaled_sigma_points<D, dim, pos>(std::make_index_sequence<count * rows>());
99  // Scale based on covariance:
100  auto ret = make_self_contained(square_root(covariance_of(d)) * Parameters::alpha * X);
101  //
102  if constexpr(sizeof...(ds) > 0)
103  return std::tuple_cat(std::tuple {std::move(ret)}, sigma_points_impl<dim, pos + rows>(ds...));
104  else
105  return std::tuple {std::move(ret)};
106  }
107 
108  public:
109  template<typename...Dist>
110  static constexpr auto
111  sigma_points(const Dist&...ds)
112  {
113  static_assert((gaussian_distribution<Dist> and ...));
114  constexpr auto dim = (DistributionTraits<Dist>::dimension + ...);
115  return sigma_points_impl<dim>(ds...);
116  }
117 
118  protected:
119  SphericalSimplex() {}; // Prevent instantiation.

121  friend struct internal::ScaledSigmaPointsBase<SphericalSimplex<Parameters>, Parameters>;

123  template<std::size_t dim, typename Scalar = double>
124  static constexpr Scalar
125  unscaled_W0()
126  {
127  return Parameters::template W0<dim>;
128  }
129 
130  template<std::size_t dim, typename Scalar = double>
131  static constexpr Scalar
132  unscaled_W()
133  {
134  return (1 - unscaled_W0<dim, Scalar>()) / (sigma_point_count<dim>() - 1);
135  }
