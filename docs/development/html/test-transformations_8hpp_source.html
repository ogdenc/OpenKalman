22 template<int n>
23 inline auto sum_of_squares = Transformation
24 {
25 [](const auto& x, const auto& ...ps) // function
26 {
27 static_assert(equivalent_to<typename MatrixTraits<decltype(x)>::RowCoefficients, Axes<n>>);
28 static_assert(equivalent_to<typename MatrixTraits<decltype(x)>::ColumnCoefficients, Axis > );
29 static_assert((equivalent_to<typename MatrixTraits<decltype(ps)>::RowCoefficients, Axis > and ...));
30 static_assert((equivalent_to<typename MatrixTraits<decltype(ps)>::ColumnCoefficients, Axis > and ...));
31
32 return make_self_contained(((transpose(x) * x) + ... + ps));
33 },
34 [](const auto& x, const auto& ...ps) // Jacobians
35 {
36 static_assert(equivalent_to<typename MatrixTraits<decltype(x)>::RowCoefficients, Axes<n>>);
37 static_assert(equivalent_to<typename MatrixTraits<decltype(x)>::ColumnCoefficients, Axis > );
38 static_assert((equivalent_to<typename MatrixTraits<decltype(ps)>::RowCoefficients, Axis > and ...));
39 static_assert((equivalent_to<typename MatrixTraits<decltype(ps)>::ColumnCoefficients, Axis > and ...));
40
41 return std::tuple_cat(std::tuple {
42 2 * transpose(x)}, OpenKalman::internal::tuple_replicate<sizeof...(ps)>(Mean {1.}));
43 },
44 [](const auto& x, const auto& ...ps) // Hessians
45 {
46 static_assert(equivalent_to<typename MatrixTraits<decltype(x)>::RowCoefficients, Axes<n>>);
47 static_assert(equivalent_to<typename MatrixTraits<decltype(x)>::ColumnCoefficients, Axis > );
48 static_assert((equivalent_to<typename MatrixTraits<decltype(ps)>::RowCoefficients, Axis > and ...));
49 static_assert((equivalent_to<typename MatrixTraits<decltype(ps)>::ColumnCoefficients, Axis > and ...));
50
51 std::array<Matrix<Axes<n>, Axes<n>, eigen_matrix_t<double, n, n>>, 1> I;
52 I[0] = 2 * eigen_matrix_t<double, n, n>::Identity();
53 return std::tuple {std::move(I), std::get<0>(zero_hessian<Axis>(ps))...};
54 }

57 template<int n>
58 inline auto time_of_arrival = Transformation
59 {
60 [](const auto& x, const auto& ...ps) // function
61 {
62 static_assert(equivalent_to<typename MatrixTraits<decltype(x)>::RowCoefficients, Axes<n>>);
63 static_assert(equivalent_to<typename MatrixTraits<decltype(x)>::ColumnCoefficients, Axis > );
64 static_assert((equivalent_to<typename MatrixTraits<decltype(ps)>::RowCoefficients, Axis > and ...));
65 static_assert((equivalent_to<typename MatrixTraits<decltype(ps)>::ColumnCoefficients, Axis > and ...));
66
67 return make_self_contained((apply_coefficientwise(adjoint(x) * x,
68 [](const auto& c) { return std::sqrt(c); }) + ... + ps));
69 },
70 [](const auto& x, const auto& ...ps) // Jacobians
71 {
72 static_assert(equivalent_to<typename MatrixTraits<decltype(x)>::RowCoefficients, Axes<n>>);
73 static_assert(equivalent_to<typename MatrixTraits<decltype(x)>::ColumnCoefficients, Axis > );
74 static_assert((equivalent_to<typename MatrixTraits<decltype(ps)>::RowCoefficients, Axis > and ...));
75 static_assert((equivalent_to<typename MatrixTraits<decltype(ps)>::ColumnCoefficients, Axis > and ...));
76
77 return std::make_tuple(make_self_contained(adjoint(x) / std::sqrt(trace(adjoint(x) * x))),
78 Matrix<Axis, Axis, native_matrix_t<decltype(ps), 1, 1>> {1.}...);
79 },
80 [](const auto& x, const auto& ...ps) // Hessians
81 {
82 static_assert(equivalent_to<typename MatrixTraits<decltype(x)>::RowCoefficients, Axes<n>>);
83 static_assert(equivalent_to<typename MatrixTraits<decltype(x)>::ColumnCoefficients, Axis > );
84 static_assert((equivalent_to<typename MatrixTraits<decltype(ps)>::RowCoefficients, Axis > and ...));
85 static_assert((equivalent_to<typename MatrixTraits<decltype(ps)>::ColumnCoefficients, Axis > and ...));
86
87 std::array<Matrix<Axes<n>, Axes<n>, eigen_matrix_t<double, n, n>>, 1> ret;
88 double sq = trace(adjoint(x) * x);
89 ret[0] = pow(sq, -1.5) * (-x * adjoint(x) + sq * MatrixTraits<decltype(ret[0])>::identity());
90 return std::make_tuple(std::move(ret), std::get<0>(zero_hessian<Axis>(ps))...);
91 }

101 inline auto radar = Transformation
102 {
103 [](const M2t& x, const auto& ...ps) // function
104 {
105 return (M2t {x(0) * cos(x(1)), x(0) * sin(x(1))} + ... + ps);
106 },
107 [](const M2t& x, const auto& ...ps) // Jacobians
108 {
109 M22t ret = {
110 std::cos(x(1)), -x(0) * std::sin(x(1)),
111 std::sin(x(1)), x(0) * std::cos(x(1))};
112 return std::tuple_cat(std::tuple {std::move(ret)},
113 OpenKalman::internal::tuple_replicate<sizeof...(ps)>(M22t::identity()));
114 },
115 [](const M2t& x, const auto& ...ps) // Hessians
116 {
117 std::array<M22t, 2> ret;
118 ret[0] = {0, -sin(x(1)),
119 -sin(x(1)), -x(0) * cos(x(1))};
120 ret[1] = {0, cos(x(1)),
121 cos(x(1)), -x(0) * sin(x(1))};
122 return std::tuple {std::move(ret), std::get<0>(zero_hessian<C2t>(ps))...};
123 }

157 inline auto Cartesian2polar = Transformation
158 {
159 [](const M2t& x, const auto& ...ps)
160 {
161 return (MP1t {std::hypot(x(0), x(1)), std::atan2(x(1), x(0))} + ... + Matrix {ps});
162 },
163 [](const M2t& x, const auto& ...ps) // Jacobians
164 {
165 const auto h = 1 / std::hypot(x(0), x(1));
166 const auto h2 = h * h;
167 MP2t ret = {
168 x(0) * h, x(1) * h,
169 -x(1) * h2, x(0) * h2};
170 return std::tuple_cat(std::tuple {
171 std::move(ret)}, OpenKalman::internal::tuple_replicate<sizeof...(ps)>(MPPt::identity()));
172 },
173 [](const M2t& x, const auto& ...ps) // Hessians
174 {
175 const auto h = 1 / std::hypot(x(0), x(1));
176 const auto h2 = h * h;
177 const auto h3 = h2 * h;
178 const auto h42 = 2 * h2 * h2;
179 const auto x00 = x(0) * x(0);
180 const auto x01 = x(0) * x(1);
181 const auto x11 = x(1) * x(1);
182 std::array<M22t, 2> ret;
183 ret[0] = {h - x00 * h3, -x01 * h3,
184 -x01 * h3, h - x11 * h3};
185 ret[1] = {x01 * h42, h2 - x00 * h42,
186 x11 * h42 - h2, -x01 * h42};
187 return std::tuple {std::move(ret), std::get<0>(zero_hessian<Polar<>>(ps))...};
188 }

201 inline auto Cylindrical2spherical = Transformation
202 {
203 [](const MC1t& x, const auto& ...ps)
204 {
205 return (MS1t {std::hypot(x(0), x(2)), x(1), std::atan2(x(2), x(0))} + ... + Matrix {ps});
206 },
207 [](const MC1t& x, const auto& ...ps) // Jacobians
208 {
209 const auto h = 1 / std::hypot(x(0), x(2));
210 const auto h2 = h * h;
211 MSCt ret = {
212 x(0) * h, 0, x(2) * h,
213 0, 1, 0,
214 -x(2) * h2, 0, x(0) * h2};
215 return std::tuple_cat(std::tuple {
216 std::move(ret)}, OpenKalman::internal::tuple_replicate<sizeof...(ps)>(MSSt::identity()));
217 },
218 [](const MC1t& x, const auto& ...ps) // Hessians
219 {
220 const auto h = 1 / std::hypot(x(0), x(2));
221 const auto h2 = h * h;
222 const auto h3 = h2 * h;
223 const auto h42 = 2 * h2 * h2;
224 const auto x00 = x(0) * x(0);
225 const auto x02 = x(0) * x(2);
226 const auto x22 = x(2) * x(2);
227 std::array<MCCt, 3> ret;
228 ret[0] = {h - x00 * h3, 0, -x02 * h3,
229 0, 0, 0,
230 -x02 * h3, 0, h - x22 * h3};
231 ret[1] = MCCt::zero();
232 ret[2] = {x02 * h42, 0, h2 - x00 * h42,
233 0, 0, 0,
234 x22 * h42 - h2, 0, -x02 * h42};
235 return std::tuple {std::move(ret), std::get<0>(zero_hessian<Spherical<>>(ps))...};
236 }
