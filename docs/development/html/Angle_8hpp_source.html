154  /*
155  * \details The angle corresponds to x and y coordinates on a unit circle.
156  * By convention, the minimum angle limit Limits<Scalar::min corresponds to the point (-1,0) in Euclidean space,
157  * and the angle is scaled so that the difference between Limits<Scalar>::min and Limits<<Scalar>::max is 2&pi;,
158  * so Limits<Scalar>::max wraps back to the point (-1, 0).
159  * \param euclidean_local_index A local index accessing either the x (if 0) or y (if 1) coordinate in Euclidean space
160  */
161 #ifdef __cpp_concepts
162  static constexpr scalar_type auto
163  to_euclidean_element(const auto& g, std::size_t euclidean_local_index, std::size_t start)
164  requires requires (std::size_t i){ {g(i)} -> scalar_type; }
165 #else
166  template<typename G, std::enable_if_t<scalar_type<typename std::invoke_result<G, std::size_t>::type>, int> = 0>
167  static constexpr auto
168  to_euclidean_element(const G& g, std::size_t euclidean_local_index, std::size_t start)
169 #endif
170  {
171  using Scalar = std::decay_t<decltype(g(std::declval<std::size_t>()))>;
172  using R = std::decay_t<decltype(internal::constexpr_real(std::declval<Scalar>()))>;
173  const Scalar cf {2 * numbers::pi_v<R> / (Limits::max - Limits::min)};
174  const Scalar mid { R{Limits::max + Limits::min} * R{0.5}};
175 
176  Scalar theta = cf * (g(start) - mid); // Convert to radians
177 
178  using std::cos, std::sin;
179  if (euclidean_local_index == 0) return cos(theta);
180  else return sin(theta);
181  }

193 #else

196  from_euclidean_element(const G& g, std::size_t local_index, std::size_t euclidean_start)
197 #endif
198  {
199  using Scalar = std::decay_t<decltype(g(std::declval<std::size_t>()))>;
200  using R = std::decay_t<decltype(internal::constexpr_real(std::declval<Scalar>()))>;
201  const Scalar cf {2 * numbers::pi_v<R> / (Limits::max - Limits::min)};
202  const Scalar mid { R{Limits::max + Limits::min} * R{0.5}};
203 
204  Scalar x = g(euclidean_start);
205  Scalar y = g(euclidean_start + 1);
206 
207  if constexpr (complex_number<Scalar>) return internal::constexpr_atan2(y, x) / cf + mid;
208  else { using std::atan2; return atan2(y, x) / cf + mid; }
209  }

220  {
221  auto ap = internal::constexpr_real(a);
222  if (not (ap < Limits::min) and ap < Limits::max)
223  {
224  return std::forward<decltype(a)>(a);
225  }
226  else

230  using std::fmod;
231  R ar {fmod(ap - R{Limits::min}, period)};
232  if (ar < 0) return internal::update_real_part(std::forward<decltype(a)>(a), R{Limits::min} + ar + period);
233  else return internal::update_real_part(std::forward<decltype(a)>(a), R{Limits::min} + ar);
234  }
