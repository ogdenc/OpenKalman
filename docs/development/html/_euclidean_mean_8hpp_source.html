343  static constexpr std::size_t columns = MatrixTraits<NestedMatrix>::columns;
344  using RowCoefficients = Coeffs;
345  using ColumnCoefficients = Axes<columns>;
346  using Scalar = typename MatrixTraits<NestedMatrix>::Scalar;
347  static_assert(RowCoefficients::dimension == dimension);

350  using NativeMatrix = typename MatrixTraits<NestedMatrix>::template NativeMatrix<rows, cols, S>;
351 
352  using SelfContained = EuclideanMean<RowCoefficients, self_contained_t<NestedMatrix>>;

355 #ifdef __cpp_concepts
356  template<coefficients RC = RowCoefficients, typename CC = void, typed_matrix_nestable Arg> requires
357  coefficients<CC> or std::same_as<CC, void>
358 #else
359  template<typename RC = RowCoefficients, typename CC = void, typename Arg,
360  std::enable_if_t<typed_matrix_nestable<Arg>, int> = 0>
361 #endif
362  static auto make(Arg&& arg) noexcept
363  {
364  static_assert(MatrixTraits<Arg>::dimension == RC::dimension);
365  if constexpr(not std::is_void_v<CC>) static_assert(equivalent_to<CC, Axes<MatrixTraits<Arg>::columns>>);
366  return EuclideanMean<RC, std::decay_t<Arg>>(std::forward<Arg>(arg));
