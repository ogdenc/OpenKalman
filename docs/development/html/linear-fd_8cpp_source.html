31TEST(transformations, finite_diff_linear_2by2)
32{
33 A22 a {1, 2, 3, 4};
34 auto f = [&] (const auto& x, const auto&...n) { return make_self_contained(((a * x) + ... + (A22::identity() * n))); };
35 auto t = FiniteDifferenceLinearization {f, M2 {1e-4, 1e-4}, M2 {1e-4, 1e-4}, M2 {1e-4, 1e-4}};
36 EXPECT_TRUE(is_near(t(M2(1, 2)), M2(5, 11)));
37 EXPECT_TRUE(is_near(std::get<0>(t.jacobian(M2(1, 2))), a));
38 EXPECT_TRUE(is_near(std::get<0>(t.jacobian(M2(3, -4))), a));
39 EXPECT_TRUE(is_near(std::get<0>(t.jacobian(M2(1, 2), M2(0.1, 0.2))), a));
40 EXPECT_TRUE(is_near(std::get<1>(t.jacobian(M2(1, 2), M2(0.1, 0.2))), A22::identity()));
41 EXPECT_TRUE(is_near(std::get<2>(t.jacobian(M2(1, 2), M2(0.1, 0.2), M2(0.3, 0.4))), A22::identity()));
42 EXPECT_TRUE(is_near(t.hessian(M2(1, 2), M2(0.1, 0.2), M2(0.3, 0.4)), zero_hessian<Axes<2>, M2, M2, M2>()));
43}
