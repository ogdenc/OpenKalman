14 namespace OpenKalman

16  /**
17  * \brief Propagates a recursive least squares error distribution of parameters, with a forgetting factor Î».
18  * Useful for parameter estimation, where the parameter is expected to possibly drift over time
19  * \tparam Scalar The scalar type.
20  */
21  template<typename Scalar = double>
22  struct RecursiveLeastSquaresTransform : TransformBase<RecursiveLeastSquaresTransform<Scalar>>
23  {
24  explicit RecursiveLeastSquaresTransform(const Scalar lambda = 0.9995)
25  : inv_lambda(1/lambda) {}
26 
27  /**
28  * Apply the RLS transform on an input distribution. Any noise distributions are treated as additive.
29  * \tparam InputDist Input distribution.
30  * \tparam NoiseDists Noise distribution.
31  **/
32 #ifdef __cpp_concepts
33  template<distribution InputDist, distribution ... NoiseDists> requires
34  (equivalent_to<typename DistributionTraits<InputDist>::Coefficients,
35  typename DistributionTraits<NoiseDists>::Coefficients>and ...)
36 #else
37  template<typename InputDist, typename ... NoiseDists,
38  std::enable_if_t<(distribution<InputDist> and ... and distribution<NoiseDists>) and
39  (equivalent_to<typename DistributionTraits<InputDist>::Coefficients,
40  typename DistributionTraits<NoiseDists>::Coefficients>and ...), int> = 0>
41 #endif
42  auto operator()(const InputDist& x, const NoiseDists& ...ns) const
43  {
44  const auto scaled_x = GaussianDistribution {mean_of(x), covariance_of(x) * inv_lambda};
45  return make_self_contained((scaled_x + ... + ns));
46  }
47 
48  /**
49  * Perform RLS transform, also returning the cross-covariance.
50  * \tparam InputDist Input distribution.
51  * \tparam NoiseDist Noise distributions.
52  **/
53 #ifdef __cpp_concepts
54  template<distribution InputDist, distribution ... NoiseDists> requires
55  (equivalent_to<typename DistributionTraits<InputDist>::Coefficients,
56  typename DistributionTraits<NoiseDists>::Coefficients>and ...)
57 #else
58  template<typename InputDist, typename ... NoiseDists,
59  std::enable_if_t<(distribution<InputDist> and ... and distribution<NoiseDists>) and
60  (equivalent_to<typename DistributionTraits<InputDist>::Coefficients,
61  typename DistributionTraits<NoiseDists>::Coefficients>and ...), int> = 0>
62 #endif
63  auto transform_with_cross_covariance(const InputDist& x, const NoiseDists& ...ns) const
64  {
65  auto scaled_cov = make_self_contained(covariance_of(x) * inv_lambda);
66  const auto scaled_x = GaussianDistribution {mean_of(x), scaled_cov};
67  auto y = make_self_contained((scaled_x + ... + ns));
68  auto cross = Matrix {scaled_cov};
69  return std::tuple {std::move(y), std::move(cross)};
70  }
71 
72  protected:
73  const Scalar inv_lambda;
74 
75  };

77 }
