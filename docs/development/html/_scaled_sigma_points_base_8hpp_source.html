16  /*************ScaledSigmaPointsBase*********
17  * \brief Base class that embodies a scaled set of sample (e.g., sigma) points.
18  *
19  * Given random variable X:Ω->ℝⁿ, sample points S⊂ℝⁿ are a finite set of
20  * samples within ℝⁿ that are specifically arranged so that the
21  * weighted distribution of S encodes approximate statistical information
22  * (e.g., mean, standard deviation, etc.) about the distribution of X.
23  * As implemented in
24  * S. Julier. The scaled unscented transformation. In Proceedings of the American
25  * Control Conference, Evanston, IL, pages 1108–1114, 2002.
26  */
27  template<typename Derived, typename Parameters>
28  struct ScaledSigmaPointsBase
29  {
37  template<std::size_t dim, typename Scalar = double>
38  static constexpr Scalar
39  W_m0()
40  {
41  constexpr Scalar alpha = Parameters::alpha;
42  constexpr Scalar W0 = Derived::template unscaled_W0<dim, Scalar>();
43  return (W0 - 1) / (alpha * alpha) + 1;
44  };
45 
53  template<std::size_t dim, typename Scalar = double>
54  static constexpr Scalar
55  W_c0()
56  {
57  constexpr Scalar alpha = Parameters::alpha;
58  constexpr Scalar beta = Parameters::beta;
59  return W_m0<dim, Scalar>() + 1 - alpha * alpha + beta;
60  };
61 
69  template<std::size_t dim, typename Scalar = double>
70  static constexpr Scalar
71  W()
72  {
73  constexpr Scalar alpha = Parameters::alpha;
74  constexpr Scalar W0 = Derived::template unscaled_W<dim, Scalar>();
75  return W0 / (alpha * alpha);
76  };

78  };
