21  /**
22  * Internal base class for linear or linearized transformations from one statistical distribution to another.
23  **/
24  template<typename Derived>
25  struct LinearTransformBase : TransformBase<Derived>
26  {
27  private:
28  template<std::size_t return_cross, typename J, typename Dist, std::size_t...ints>
29  static auto sum_noise_terms(const J& j, const Dist& dist, std::index_sequence<ints...>)
30  {
31  using InputDist = std::tuple_element<0, Dist>;
32  if constexpr(cholesky_form<InputDist>)
33  {
34  if constexpr (return_cross)
35  {
36  auto sqrt_c0 =square_root(covariance_of(std::get<0>(dist)));
37  auto term0 = std::get<0>(j) * sqrt_c0;
38  return std::tuple {
39  LQ_decomposition(concatenate_horizontal(term0, Matrix(
40  (std::get<ints+1>(j) * (square_root(covariance_of(std::get<ints+1>(dist))))))...)),
41  make_self_contained(sqrt_c0 * adjoint(term0))};
42  }
43  else
44  {
45  auto term0 = std::get<0>(j) * square_root(covariance_of(std::get<0>(dist)));
46  return LQ_decomposition(concatenate_horizontal(term0, Matrix(
47  (std::get<ints+1>(j) * (square_root(covariance_of(std::get<ints+1>(dist))))))...));
48  }
49  }
50  else
51  {
52  if constexpr (return_cross)
53  {
54  auto cross = make_self_contained(covariance_of(std::get<0>(dist)) * adjoint(std::get<0>(j)));
55  auto cov = make_covariance((
56  (std::get<0>(j) * cross) +
57  ... + (std::get<ints+1>(j) * (covariance_of(std::get<ints+1>(dist)) * adjoint(std::get<ints+1>(j))))));
58  return std::tuple {std::move(cov), std::move(cross)};
59  }
60  else
61  {
62  return make_covariance((
63  (std::get<0>(j) * covariance_of(std::get<0>(dist)) * adjoint(std::get<0>(j))) +
64  ... + (std::get<ints+1>(j) * (covariance_of(std::get<ints+1>(dist)) * adjoint(std::get<ints+1>(j))))));
65  }
66  }
67  }
68 
69  protected:
70  using Base = TransformBase<Derived>;
71 
72  /**
73  * Linearly transform one statistical distribution to another.
74  * \tparam TransformFunction Underlying transform function that takes an input distribution and an optional set of
75  * noise distributions and returns the following information used in constructing the output distribution
76  * and cross-covariance:
77  * -# the output mean, and
78  * -# a tuple of Jacobians corresponding to each input and noise term,
79  * \tparam InputDist Input distribution.
80  * \tparam NoiseDist Noise distribution.
81  **/
82  template<std::size_t return_cross, typename TransformFunction, typename InputDist, typename ... NoiseDists,
83  std::enable_if_t<(distribution<InputDist> and ... and distribution<NoiseDists>), int> = 0>
84  auto transform(const TransformFunction& f, const InputDist& in, const NoiseDists& ...n) const
85  {
86  auto[mean_output, jacobians] = f(mean_of(in), mean_of(n)...);
87  if constexpr (return_cross)
88  {
89  auto [cov_out, cross_covariance] = sum_noise_terms<true>(jacobians, std::tuple {in, n...},
90  std::make_index_sequence<std::min(sizeof...(NoiseDists), std::tuple_size_v<decltype(jacobians)> - 1)>{});
91  auto out = make_GaussianDistribution(mean_output, cov_out);
92  if constexpr(TransformFunction::correction)
93  return std::tuple {make_self_contained(out + f.add_correction(in, n...)), cross_covariance};
94  else
95  return std::tuple {out, cross_covariance};
96  }
97  else
98  {
99  auto cov_out = sum_noise_terms<false>(jacobians, std::tuple {in, n...},
100  std::make_index_sequence<std::min(sizeof...(NoiseDists), std::tuple_size_v<decltype(jacobians)> - 1)>{});
101  auto out = make_GaussianDistribution(mean_output, cov_out);
102  if constexpr(TransformFunction::correction)
103  return make_self_contained(out + f.add_correction(in, n...));
104  else
105  return out;
106  }
107  }
108 
109  public:
110  using Base::operator();
111  using Base::transform_with_cross_covariance;
112 
113  /**
114  * Perform a linear(ized) transform from one statistical distribution to another.
115  * \tparam InputDist Input distribution.
116  * \tparam Transformation The transformation on which the transform is based (e.g., LinearTransformation).
117  * \tparam NoiseDists Noise distributions.
118  **/
119  template<typename InputDist, typename Trans, typename ... NoiseDists, std::enable_if_t<
120  (distribution<InputDist> and ... and distribution<NoiseDists>) and is_linearized_function_v<Trans, 1>, int> = 0>
121  auto operator()(const InputDist& x, const Trans& g, const NoiseDists&...ns) const
122  {
123  return transform<false>(typename Derived::template TransformFunction<Trans> {g}, x, ns...);
124  }
125 
126  /**
127  * Perform a linear(ized) transform, also returning the cross-covariance.
128  * \tparam InputDist Input distribution.
129  * \tparam Transformation The transformation on which the transform is based (e.g., LinearTransformation).
130  * \tparam NoiseDists Noise distributions.
131  **/
132  template<typename InputDist, typename Trans, typename ... NoiseDists, std::enable_if_t<
133  (distribution<InputDist> and ... and distribution<NoiseDists>) and is_linearized_function_v<Trans, 1>, int> = 0>
134  auto transform_with_cross_covariance(const InputDist& x, const Trans& g, const NoiseDists&...ns) const
135  {
136  return transform<true>(typename Derived::template TransformFunction<Trans> {g}, x, ns...);
137  }
