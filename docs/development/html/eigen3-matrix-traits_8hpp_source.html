20  /**
21  * \brief Specifies a native Eigen3 matrix.
22  * \details This includes any class in the Eigen library descending from Eigen::MatrixBase.
23  * \note This is a concept when compiled with c++20, and a constexpr bool in c++17.
24  */
25  template<typename T>
26 #ifdef __cpp_concepts
27  concept eigen_native = std::derived_from<std::decay_t<T>, Eigen::MatrixBase<std::decay_t<T>>> and
28  (not std::derived_from<std::decay_t<T>, internal::Eigen3Base<std::decay_t<T>>>);
29 #else
30  inline constexpr bool eigen_native = std::is_base_of_v<Eigen::MatrixBase<std::decay_t<T>>, std::decay_t<T>> and
31  (not std::is_base_of_v<internal::Eigen3Base<std::decay_t<T>>, std::decay_t<T>>);
32 #endif
33  } // namespace Eigen3

36  /*
37  * \brief Default matrix traits for any matrix derived from Eigen::MatrixBase.
38  * \tparam M The matrix.
39  */
40 #ifdef __cpp_concepts
41  template<typename M> requires std::same_as<M, std::decay_t<M>> and Eigen3::eigen_native<M>
42  struct MatrixTraits<M>
43 #else
44  template<typename M>
45  struct MatrixTraits<M, std::enable_if_t<std::is_same_v<M, std::decay_t<M>> and Eigen3::eigen_native<M>>>
46 #endif
47  {
48  using Scalar = typename M::Scalar;
49 
50  static constexpr std::size_t dimension = M::RowsAtCompileTime;
51  static constexpr std::size_t columns = M::ColsAtCompileTime; //<\TODO: make columns potentially dynamic (0 = dynamic?)
52  //Note: rows or columns at compile time are -1 if the matrix is dynamic:
53  static_assert(dimension > 0);
54  static_assert(columns > 0);
55 
56  template<std::size_t rows = dimension, std::size_t cols = columns, typename S = Scalar>
57  using NativeMatrix = Eigen::Matrix<S, (Eigen::Index) rows, (Eigen::Index) cols>;
58 
59  using SelfContained = typename MatrixTraits<M>::template NativeMatrix<dimension, columns>;
60 
61  template<TriangleType storage_triangle = TriangleType::lower, std::size_t dim = dimension, typename S = Scalar>
62  using SelfAdjointBaseType = Eigen3::SelfAdjointMatrix<NativeMatrix<dim, dim, S>, storage_triangle>;
63 
64  template<TriangleType triangle_type = TriangleType::lower, std::size_t dim = dimension, typename S = Scalar>
65  using TriangularBaseType = Eigen3::TriangularMatrix<NativeMatrix<dim, dim, S>, triangle_type>;
66 
67  template<std::size_t dim = dimension, typename S = Scalar>
68  using DiagonalBaseType = Eigen3::DiagonalMatrix<NativeMatrix<dim, 1, S>>;
69 
70  template<typename Derived>
71  using MatrixBaseType = Eigen3::internal::Eigen3MatrixBase<Derived, M>;
72 
73 #ifdef __cpp_concepts
74  template<typename Arg> requires (not std::convertible_to<Arg, const Scalar>)
75 #else
76  template<typename Arg, std::enable_if_t<not std::is_convertible_v<Arg, const Scalar>, int> = 0>
77 #endif
78  static decltype(auto)
79  make(Arg&& arg) noexcept
80  {
81  return std::forward<Arg>(arg);
82  }
83 
84  // Make matrix from a list of coefficients in row-major order.
85 #ifdef __cpp_concepts
86  template<std::convertible_to<const Scalar> Arg, std::convertible_to<const Scalar> ... Args>
87  requires (1 + sizeof...(Args) == dimension * columns)
88 #else
89  template<typename Arg, typename ... Args,
90  std::enable_if_t<std::conjunction_v<std::is_convertible<Arg, Scalar>, std::is_convertible<Args, Scalar>...> and
91  1 + sizeof...(Args) == dimension * columns, int> = 0>
92 #endif
93  static auto
94  make(const Arg arg, const Args ... args)
95  {
96  return ((NativeMatrix<dimension, columns>() << arg), ... , args).finished();
97  }
98 
99  static auto zero() { return Eigen3::ZeroMatrix<NativeMatrix<dimension, columns>>(); }
100 
101  static auto identity() { return NativeMatrix<dimension, dimension>::Identity(); }
102 
103  };
