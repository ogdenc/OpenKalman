60  template<typename T, std::size_t max_dim, std::size_t position = 0>
61  auto get_vector_space_descriptor_slice_impl(std::size_t offset, std::size_t extent, DynamicDescriptor<>&& ds = {})
62  {
63  if constexpr (vector_space_component_count_v<T> == 0 and position == max_dim)
64  {
65  return std::move(ds);
66  }
67  else if (position == offset + extent)
68  {
69  return std::move(ds);
70  }
71  else if ((position <= offset or get_dimension_size_of(ds) > 0) and
72  (position < offset + extent))
73  {
74  using Head = std::tuple_element_t<0, internal::split_head_tail_fixed_t<T>>;
75  using Tail = std::tuple_element_t<1, internal::split_head_tail_fixed_t<T>>;
76  constexpr std::size_t next_position = position + dimension_size_of_v<Head>;
77  if (position < offset)
78  return get_vector_space_descriptor_slice_impl<Tail, max_dim, next_position>(offset, extent);
79  else
80  return get_vector_space_descriptor_slice_impl<Tail, max_dim, next_position>(offset, extent, std::move(ds += Head{}));
81  }
82  else

84  throw std::invalid_argument {"get_vector_space_descriptor_slice: invalid arguments"};
