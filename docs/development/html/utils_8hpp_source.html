98  /**
99  * \internal
100  * \brief Joins two arrays.
101  * \tparam T The type of the array elements.
102  * \tparam size_L The size of the left array.
103  * \tparam size_R The size of the right array.
104  * \param left The left array.
105  * \param right The right array.
106  * \return The left array concatenated with the right array.
107  */
108  template<typename T, std::size_t size_L, std::size_t size_R>
109  constexpr std::array<T, size_L + size_R>
110  join(const std::array<T, size_L>& left, const std::array<T, size_R>& right)
111  {
112  return join_impl(left, right, typename detail::Counter<size_L, 0>::type(),
113  typename detail::Counter<size_R, 0>::type());

135  template<typename T>
136  constexpr T sqrt_impl(T x, T lo, T hi)
137  {
138  if (lo == hi) return lo;
139  const T mid = (lo + hi + 1) / 2;
140  if (x / mid < mid) return sqrt_impl<T>(x, lo, mid - 1);
141  else return sqrt_impl(x, mid, hi);
142  }
143  }

168  /**
169  * \internal
170  * \brief Takes a slice of a tuple, given an index range.
171  * \tparam index1 The index of the beginning of the slice.
172  * \tparam index2 The first index just beyond the end of the slice.
173  * \tparam T The tuple type.
174  * \param t The tuple.
175  * \return The tuple slice.
176  */
177  template<std::size_t index1, std::size_t index2, typename T>
178  constexpr auto tuple_slice(T&& t)
179  {
180  static_assert(index1 <= index2, "Index range is invalid");
181  static_assert(index2 <= std::tuple_size_v<std::decay_t<T>>, "Index is out of bounds");
182  return detail::tuple_slice_impl<index1>(std::forward<T>(t), std::make_index_sequence<index2 - index1>());
183  }

186  /**
187  * \internal
188  * \brief Creates a tuple that replicates a value N number of times.
189  * \tparam N The number of times to replicate.
190  * \tparam T The type of the tuple element to replicate.
191  * \param t The tuple element to replicate.
192  * \return A tuple containing N copies of t.
193  */
194  template<std::size_t N, typename T>
195  constexpr auto tuple_replicate(T&& t)
196  {
197  if constexpr (N == 0)
198  {
199  return std::tuple {};
200  }
201  else if constexpr (N == 1)
202  {
203  return std::make_tuple(std::forward<T>(t));
204  }
205  else
206  {
207  return std::tuple_cat(std::make_tuple(t), tuple_replicate<N - 1>(std::forward<T>(t)));
208  }
209  }
