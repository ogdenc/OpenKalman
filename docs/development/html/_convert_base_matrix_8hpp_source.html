14 namespace OpenKalman::internal
15 {
16 #ifdef __cpp_concepts
17  template<covariance_nestable T, typename Arg>
18  requires covariance<Arg> or (typed_matrix<Arg> and equivalent_to<typename MatrixTraits<Arg>::RowCoefficients,
19  typename MatrixTraits<Arg>::ColumnCoefficients>)
20 #else
21  template<typename T, typename Arg, typename>
22 #endif
23  constexpr decltype(auto)
24  convert_nested_matrix(Arg&& arg) noexcept
25  {
26  static_assert(covariance<Arg> or typed_matrix<Arg>);
27  using ArgBase = nested_matrix_t<Arg>;
28 
29  // Typed matrices:
30  if constexpr(typed_matrix<Arg>)
31  {
32  static_assert(equivalent_to<typename MatrixTraits<Arg>::RowCoefficients, typename MatrixTraits<Arg>::ColumnCoefficients>);
33  using SA = typename MatrixTraits<ArgBase>::template SelfAdjointBaseType<>;
34  return MatrixTraits<T>::make(nested_matrix(MatrixTraits<SA>::make(std::forward<Arg>(arg).nested_matrix())));
35  }
36 
37  // strictly triangular or diagonal square root --> self-adjoint
38  else if constexpr(((triangular_matrix<ArgBase> and not diagonal_matrix<ArgBase>) or
39  (square_root_covariance<Arg> and diagonal_matrix<ArgBase>)) and self_adjoint_matrix<T> and not diagonal_matrix<T>)
40  {
41  if constexpr((self_adjoint_matrix<ArgBase> and not square_root_covariance<Arg>) or
42  (triangular_matrix<ArgBase> and square_root_covariance<Arg>))
43  return Cholesky_square(std::forward<Arg>(arg).nested_matrix());
44  else
45  return std::forward<Arg>(arg).get_apparent_nested_matrix();
46  }
47 
48  // strictly self-adjoint or diagonal non-square root --> strictly triangular
49  else if constexpr(((not diagonal_matrix<ArgBase> and not triangular_matrix<ArgBase>) or
50  (diagonal_matrix<ArgBase> and not square_root_covariance<Arg> )) and triangular_matrix<T> and not diagonal_matrix<T>)
51  {
52  if constexpr(diagonal_matrix<ArgBase>) // diagonal non-square root
53  {
54  return Cholesky_factor(std::forward<Arg>(arg).nested_matrix());
55  }
56  else // ArgBase is strictly self-adjoint.
57  {
58  using B = decltype(Cholesky_factor(std::forward<Arg>(arg).nested_matrix()));
59  if constexpr(internal::same_triangle_type_as<B, T>) // Converted triangle types match.
60  {
61  if constexpr((self_adjoint_matrix<ArgBase> and not square_root_covariance<Arg>) or
62  (triangular_matrix<ArgBase> and square_root_covariance<Arg>))
63  return Cholesky_factor(std::forward<Arg>(arg).nested_matrix());
64  else
65  {
66  return std::forward<Arg>(arg).get_apparent_nested_matrix();
67  }
68  }
69  else // Converted triangle types don't match.
70  {
71  return Cholesky_factor(adjoint(std::forward<Arg>(arg).nested_matrix()));
72  }
73  }
74  }
75 
76  // upper triangular <--> lower triangular
77  else if constexpr(cholesky_form<Arg> and triangular_matrix<T> and not (internal::same_triangle_type_as<ArgBase, T>))
78  {
79  return adjoint(std::forward<Arg>(arg).nested_matrix());
80  }
81 
82  // pass through
83  else
84  {
85  return std::forward<Arg>(arg).nested_matrix();
86  }
87  }

104  }
105  else
106  {
107  return std::forward<Arg>(arg).get_apparent_nested_matrix();
108  }
