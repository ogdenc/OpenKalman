16using M = Mean<Axes<2>>;
17using A = Matrix<Axes<2>, Axes<2>>;
18
19TEST(transformations, stochastic_additive)
20{
21 auto f = [](const auto& x) -> M { return A {1, 2, 3, 4} * x; };
22 auto dist = GaussianDistribution {M::zero(), A::identity()};
23 auto t = Transformation<decltype(f)> {f};
24 M x {2, 3};
25 M true_y {f(x)};
26 M mean_y {M::zero()};
27 for (int i = 0; i < 100; i++)
28 {
29 const M y {t(x) + dist()};
30 mean_y = (mean_y * i + y) / (i + 1);
31 }
32 EXPECT_NE(mean_y, true_y);
33 EXPECT_TRUE(is_near(mean_y, true_y, nested_matrix_t<M>::Constant(1.0)));
34}

37TEST(transformations, stochastic_augmented)
38{
39 auto f = [](const auto& x, const auto&...n) { return make_self_contained(((A {1, 2, 4, 3} * x) + ... + (A {3, 4, 2, 1} * n))); };
40 auto dist = GaussianDistribution {M::zero(), A::identity()};
41 auto t = Transformation<decltype(f)> {f};
42 M x {2, 3}, n {0, 0};
43 M true_y {f(x, n)};
44 M mean_y {M::zero()};
45 for (int i = 0; i < 100; i++)
46 {
47 const M y {t(x, dist)};
48 mean_y = (mean_y * i + y) / (i + 1);
49 }
50 EXPECT_NE(mean_y, true_y);
51 EXPECT_TRUE(is_near(mean_y, true_y, nested_matrix_t<M>::Constant(1.0)));
52}
