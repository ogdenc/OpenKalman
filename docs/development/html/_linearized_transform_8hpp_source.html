32  template<typename MeanType, typename Hessian, typename Cov, std::size_t...ints>
33  static auto construct_mean(const Hessian& hessian, const Cov& P, std::index_sequence<ints...>)
34  {
35  return MeanType {0.5 * trace(P * hessian[ints])...};

40  {
41  return std::tuple {0.5 * trace(P * hessian[i] * P * hessian[js])...};

46  {
47  auto mat = std::tuple_cat(construct_cov_row<is, js...>(hessian, P)...);
48  return std::make_from_tuple<CovType>(std::move(mat));

61  template<typename OutputCoeffs, typename Hessian, typename Dist>
62  static auto second_order_term(const Hessian& hessian, const Dist& x)
63  {
64  constexpr auto output_dim = std::tuple_size_v<Hessian>;
65  static_assert(OutputCoeffs::size == output_dim);
66  //
67  // Convert input distribution type to output distribution types, and initialize mean and covariance:
68  using CovIn = nested_matrix_t<typename DistributionTraits<Dist>::Covariance>;
69  using MeanOut = native_matrix_t<CovIn, output_dim, 1>;
70  constexpr TriangleType tri = triangle_type_of<typename MatrixTraits<CovIn>::template TriangularBaseType<>>;
71  using CovOut = typename MatrixTraits<CovIn>::template SelfAdjointBaseType<tri, output_dim>;
72 
73  const auto P = covariance_of(x);
74  const std::make_index_sequence<output_dim> ints;
75  auto mean_terms = construct_mean<Mean<OutputCoeffs, MeanOut>>(hessian, P, ints);
76  auto cov_terms = construct_cov<Covariance<OutputCoeffs, CovOut>>(hessian, P, ints, ints);
77  return GaussianDistribution(mean_terms, cov_terms);
78  }

80  template<typename OutputCoeffs, typename T1, typename T2, std::size_t...I>
81  static auto zip_tuples_impl(const T1& t1, const T2& t2, std::index_sequence<I...>)
82  {
83  return make_self_contained((second_order_term<OutputCoeffs>(std::get<I>(t1), std::get<I>(t2)) + ...));
84  }
