17  /**
18  * \brief A linearized transform, using a 1st or 2nd order Taylor approximation of a linear transformation.
19  */
20  template<unsigned int order = 1> ///< Order of the Taylor approximation (1 or 2).
21  struct LinearizedTransform : internal::LinearTransformBase<LinearizedTransform<order>>
22  {
23  protected:
24  using Base = internal::LinearTransformBase<LinearizedTransform>;
25  friend Base;
26 
27  //-----------------------------------------------
28  template<typename Transformation>
29  struct TransformFunction
30  {
31  private:
32  template<typename MeanType, typename Hessian, typename Cov, std::size_t...ints>
33  static auto construct_mean(const Hessian& hessian, const Cov& P, std::index_sequence<ints...>)
34  {
35  return MeanType {0.5 * trace(P * hessian[ints])...};
36  }
37 
38  template<std::size_t i, std::size_t...js, typename Hessian, typename Cov>
39  static constexpr auto construct_cov_row(const Hessian& hessian, const Cov& P)
40  {
41  return std::tuple {0.5 * trace(P * hessian[i] * P * hessian[js])...};
42  }
43 
44  template<typename CovType, typename Hessian, typename Cov, std::size_t...is, std::size_t...js>
45  static auto construct_cov(const Hessian& hessian, const Cov& P, std::index_sequence<is...>, std::index_sequence<js...>)
46  {
47  auto mat = std::tuple_cat(construct_cov_row<is, js...>(hessian, P)...);
48  return std::make_from_tuple<CovType>(std::move(mat));
49  }
50 
51  protected:
52  const Transformation& transformation;
53 
54  /**
55  * Add second-order moment terms, based on Hessian matrices.
56  * \tparam Hessian An array of Hessian matrices. Must be accessible by bracket index, as in hessian[i].
57  * Each matrix is a regular matrix type.
58  * \tparam Dist Input or noise distribution.
59  * \return
60  */
61  template<typename OutputCoeffs, typename Hessian, typename Dist>
62  static auto second_order_term(const Hessian& hessian, const Dist& x)
63  {
64  constexpr auto output_dim = std::tuple_size_v<Hessian>;
65  static_assert(OutputCoeffs::size == output_dim);
66  //
67  // Convert input distribution type to output distribution types, and initialize mean and covariance:
68  using CovIn = nested_matrix_t<typename DistributionTraits<Dist>::Covariance>;
69  using MeanOut = native_matrix_t<CovIn, output_dim, 1>;
70  constexpr TriangleType tri = triangle_type_of<typename MatrixTraits<CovIn>::template TriangularBaseType<>>;
71  using CovOut = typename MatrixTraits<CovIn>::template SelfAdjointBaseType<tri, output_dim>;
72 
73  const auto P = covariance_of(x);
74  const std::make_index_sequence<output_dim> ints;
75  auto mean_terms = construct_mean<Mean<OutputCoeffs, MeanOut>>(hessian, P, ints);
76  auto cov_terms = construct_cov<Covariance<OutputCoeffs, CovOut>>(hessian, P, ints, ints);
77  return GaussianDistribution(mean_terms, cov_terms);
78  }
79 
80  template<typename OutputCoeffs, typename T1, typename T2, std::size_t...I>
81  static auto zip_tuples_impl(const T1& t1, const T2& t2, std::index_sequence<I...>)
82  {
83  return make_self_contained((second_order_term<OutputCoeffs>(std::get<I>(t1), std::get<I>(t2)) + ...));
84  }
85 
86  template<typename OutputCoeffs, typename T1, typename T2>
87  static constexpr auto zip_tuples(const T1& t1, const T2& t2)
88  {
89  static_assert(std::tuple_size_v<T1> == std::tuple_size_v<T2>);
90  return zip_tuples_impl<OutputCoeffs>(t1, t2, std::make_index_sequence<std::tuple_size_v<T1>>());
91  }
92 
93  public:
94  TransformFunction(const Transformation& t) : transformation(t) {}
95 
96  template<typename InputMean, typename ... NoiseMean>
97  auto operator()(const InputMean& x, const NoiseMean& ... n) const
98  {
99  return std::tuple {transformation(x, n...), transformation.jacobian(x, n...)};
100  }
101 
102  static constexpr bool correction = order > 1;
103 
104  template<typename InputDist, typename ... Noise>
105  auto add_correction(const InputDist& x, const Noise& ... n) const
106  {
107  using In_Mean = typename DistributionTraits<InputDist>::Mean;
108  using Out_Mean = std::invoke_result_t<Transformation, In_Mean>;
109  using OutputCoeffs = typename MatrixTraits<Out_Mean>::RowCoefficients;
110  const auto hessians = transformation.hessian(mean_of(x), mean_of(n)...);
111  return zip_tuples<OutputCoeffs>(hessians, std::tuple {x, n...});
112  }
113  };
114  //-----------------------------------------------
115 
116  };
