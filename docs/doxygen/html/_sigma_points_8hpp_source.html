18  /*************GeneralSigmaPoints*********
19  * \brief This class embodies a set of sample (e.g., sigma) points.
20  *
21  * Given random variable X:Ω->ℝⁿ, sample points S⊂ℝⁿ are a finite set of
22  * samples within ℝⁿ that are specifically arranged so that the
23  * weighted distribution of S encodes approximate statistical information
24  * (e.g., mean, standard deviation, etc.) about the distribution of X.
25  * As implemented in
26  * S. Julier. The scaled unscented transformation. In Proceedings of the American
27  * Control Conference, Evanston, IL, pages 1108–1114, 2002.
28  * This class has only static members, and is not to be instantiated.
29  */
30  template<typename SigmaPointsType>
31  struct SigmaPoints;

57  template<std::size_t dim, typename Weights>
58  static auto mean_weights()
59  {
60  using Scalar = typename MatrixTraits<Weights>::Scalar;
61  constexpr auto count = MatrixTraits<Weights>::dimension;
62  constexpr auto w0 = SigmaPointsType::template W_m0<dim, Scalar>();
63  return cat_weights<dim, Weights>(w0, std::make_index_sequence<count - 1>());
64  };

67  static auto covariance_weights()
68  {
69  using Scalar = typename MatrixTraits<Weights>::Scalar;
70  constexpr auto count = MatrixTraits<Weights>::dimension;
71  constexpr auto w0 = SigmaPointsType::template W_c0<dim, Scalar>();
72  return cat_weights<dim, Weights>(w0, std::make_index_sequence<count - 1>());
73  };

88 #ifdef __cpp_concepts
89  template<std::size_t dim, euclidean_mean YMeans>
90 #else
91  template<std::size_t dim, typename YMeans, std::enable_if_t<euclidean_mean<YMeans>, int> = 0>
92 #endif
93  static auto
94  weighted_means(const YMeans& y_means)
95  {
96  static_assert(column_vector<YMeans>);
97  constexpr auto count = MatrixTraits<YMeans>::columns;
98  static_assert(count == SigmaPointsType::template sigma_point_count<dim>(), "Wrong number of sigma points.");
99  using Weights = Matrix<Axes<count>, Axis, native_matrix_t<YMeans, count, 1>>;
100  return make_self_contained(y_means * mean_weights<dim, Weights>());
101  }

103  template<std::size_t dim, typename InputDist, bool return_cross = false, typename X, typename Y>
104  static auto
105  covariance(const X& x_deviations, const Y& y_deviations)
106  {
107  static_assert(typed_matrix<X> and typed_matrix<Y>);
108  static_assert(equivalent_to<typename MatrixTraits<X>::RowCoefficients,
109  typename DistributionTraits<InputDist>::Coefficients>);
110  constexpr auto count = MatrixTraits<X>::columns;
111  static_assert(count == MatrixTraits<Y>::columns);
112  static_assert(count == SigmaPointsType::template sigma_point_count<dim>(), "Wrong number of sigma points.");
113  using Weights = Matrix<Axes<count>, Axis, native_matrix_t<X, count, 1>>;
114  auto weights = covariance_weights<dim, Weights>();
115  if constexpr(cholesky_form<InputDist>)
116  {
117  using Scalar = typename MatrixTraits<X>::Scalar;
118  //
119  constexpr auto W_c0 = SigmaPointsType::template W_c0<dim, Scalar>();
120  if constexpr (W_c0 < 0)
121  {
122  // Discard first weight and first y-deviation column for now, since square root of weight would be negative.
123  const auto [y_deviations_head, y_deviations_tail] = split_horizontal<1, count - 1>(y_deviations);
124  const auto [weights_head, weights_tail] = split_vertical<Axis, Axes<count - 1>>(weights);
125  const auto sqrt_weights_tail = apply_coefficientwise(weights_tail, [](const auto x){ return std::sqrt(x); });
126  auto out_covariance = LQ_decomposition(y_deviations_tail * to_diagonal(sqrt_weights_tail));
127  rank_update(out_covariance, y_deviations_head, W_c0);
128  if constexpr (return_cross)
129  {
130  auto cross_covariance = make_self_contained(x_deviations * to_diagonal(weights) * adjoint(y_deviations));
131  return std::tuple{std::move(out_covariance), std::move(cross_covariance)};
132  }
133  else
134  {
135  return out_covariance;
136  }
137  }
138  else
139  {
140  const auto sqrt_weights = apply_coefficientwise(weights, [](const auto x){ return std::sqrt(x); });
141  auto out_covariance = Covariance {LQ_decomposition(y_deviations * to_diagonal(sqrt_weights))};
142  if constexpr (return_cross)
143  {
144  auto cross_covariance = make_self_contained(x_deviations * to_diagonal(weights) * adjoint(y_deviations));
145  return std::tuple{std::move(out_covariance), std::move(cross_covariance)};
146  }
147  else
148  {
149  return out_covariance;
150  }
151  }
152  }
153  else
154  {
155  const auto w_yT = to_diagonal(weights) * adjoint(y_deviations);
156  auto out_covariance = make_self_contained(make_covariance(y_deviations * w_yT));
157  if constexpr (return_cross)
158  {
159  auto cross_covariance = make_self_contained(x_deviations * w_yT);
160  return std::tuple{std::move(out_covariance), std::move(cross_covariance)};
161  }
162  else
163  {
164  return out_covariance;
165  }
166  }
167  }
168 
169  };
