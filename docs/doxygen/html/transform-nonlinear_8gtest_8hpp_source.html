20  template<typename Transform,
21  typename Transformation,
22  typename ReverseTransformation,
23  typename AngleDist,
24  typename AngleRotDist,
25  typename ... Noise>
26  ::testing::AssertionResult rotational_invariance_test(
27  const Transform& t,
28  const Transformation& f1,
29  const ReverseTransformation& f2,
30  const AngleDist& angle_input,
31  const AngleRotDist& angle_input_rot,
32  const Noise& ... noise)
33  {
34  const auto[output, cross] = t.transform_with_cross_covariance(angle_input, f1, noise...);
35  const auto[output_rot, cross_rot] = t.transform_with_cross_covariance(angle_input_rot, f1, noise...);
36  const auto res1 = is_near(
37  f2(mean_of(output)),
38  Mean {f2(mean_of(output_rot)) +
39  make_mean<typename DistributionTraits<AngleRotDist>::FixedDescriptor>(0, numbers::pi)},
40  1e-4);
41  const auto res2 = is_near(covariance_of(output), covariance_of(output_rot), 1e-3);
42  const auto res3 = is_near(cross, -cross_rot, 1e-4);
43  if (res1 and res2 and res3) return ::testing::AssertionSuccess();
44  else
45  return ::testing::AssertionFailure() <<
46  (res1 ? "" : "mean") << (res1 ? "" : res1.message()) <<
47  (res2 ? "" : "covariance") << (res2 ? "" : res2.message()) <<
48  (res3 ? "" : "cross covariance") << (res3 ? "" : res3.message());
49  }
