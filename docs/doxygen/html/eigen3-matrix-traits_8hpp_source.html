29 #ifdef __cpp_concepts
30  template<Eigen3::eigen_native M> requires std::same_as<M, std::decay_t<M>>
31  struct MatrixTraits<M>
32 #else
33  template<typename M>
34  struct MatrixTraits<M, std::enable_if_t<Eigen3::eigen_native<M> and std::is_same_v<M, std::decay_t<M>>>>
35 #endif
36  {
37  using Scalar = typename M::Scalar;
38 
39  static constexpr std::size_t dimension = M::RowsAtCompileTime;
40  static constexpr std::size_t columns = M::ColsAtCompileTime; //<\todo: make columns potentially dynamic (0 = dynamic?)
41  //Note: rows or columns at compile time are -1 if the matrix is dynamic:
42  static_assert(dimension > 0, "Cannot currently use dynamically sized matrices with OpenKalman.");
43  static_assert(columns > 0, "Cannot currently use dynamically sized matrices with OpenKalman.");
44  //static_assert(one_by_one_matrix<M> or identity_matrix<M> or not (zero_matrix<M> or upper_triangular_matrix<M> or
45  // lower_triangular_matrix<M> or self_adjoint_matrix<M>), "Native Eigen3 matrices other than 1x1 or identity must"
46  // "have their own MatrixTraits if they are zero, triangular, and self-adjoint.");
47 
48  // Cannot use dimension and columns constant expressions here because of bug in GCC 10.1.0 (but not clang 10.0.0):
49  template<std::size_t rows = std::size_t(M::RowsAtCompileTime),
50  std::size_t cols = std::size_t(M::ColsAtCompileTime), typename S = Scalar>
51  using NativeMatrixFrom = Eigen::Matrix<S, (Eigen::Index) rows, (Eigen::Index) cols>;
52 
53  using SelfContainedFrom = NativeMatrixFrom<>;
54 
55  template<TriangleType storage_triangle = TriangleType::lower, std::size_t dim = dimension, typename S = Scalar>
56  using SelfAdjointMatrixFrom = Eigen3::SelfAdjointMatrix<NativeMatrixFrom<dim, dim, S>, storage_triangle>;
57 
58  template<TriangleType triangle_type = TriangleType::lower, std::size_t dim = dimension, typename S = Scalar>
59  using TriangularMatrixFrom = Eigen3::TriangularMatrix<NativeMatrixFrom<dim, dim, S>, triangle_type>;
60 
61  template<std::size_t dim = dimension, typename S = Scalar>
62  using DiagonalMatrixFrom = Eigen3::DiagonalMatrix<NativeMatrixFrom<dim, 1, S>>;
63 
64  template<typename Derived>
65  using MatrixBaseFrom = Eigen3::internal::Eigen3MatrixBase<Derived, M>;
66 
67 #ifdef __cpp_concepts
68  template<Eigen3::eigen_native Arg>
69 #else
70  template<typename Arg, std::enable_if_t<Eigen3::eigen_native<Arg>, int> = 0>
71 #endif
72  static decltype(auto) make(Arg&& arg) noexcept
73  {
74  return std::forward<Arg>(arg);
75  }
76 
77  // Make matrix from a list of coefficients in row-major order.
78 #ifdef __cpp_concepts
79  template<std::convertible_to<Scalar> Arg, std::convertible_to<Scalar> ... Args>
80  requires (1 + sizeof...(Args) == dimension * columns)
81 #else
82  template<typename Arg, typename ... Args,
83  std::enable_if_t<std::conjunction_v<std::is_convertible<Arg, Scalar>, std::is_convertible<Args, Scalar>...> and
84  1 + sizeof...(Args) == dimension * columns, int> = 0>
85 #endif
86  static auto make(const Arg arg, const Args ... args)
87  {
88  return ((NativeMatrixFrom<dimension, columns>() << arg), ... , args).finished();
89  }
90 
91  static auto zero() { return Eigen3::ZeroMatrix<Scalar, dimension, columns>(); }
92 
93  static auto identity() { return NativeMatrixFrom<dimension, dimension>::Identity(); }
94
