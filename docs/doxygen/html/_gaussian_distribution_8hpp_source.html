364  auto operator()() const
365  {
366  auto norm = randomize<
367  Matrix<Coefficients, Axis, MeanNestedMatrix>, std::normal_distribution, random_number_engine>(0.0, 1.0);
368  auto s = square_root(sigma);
369  if constexpr(not lower_triangular_matrix<CovarianceNestedMatrix>)
370  return make_self_contained(make_matrix(mu) + transpose(s) * norm);
371  else
372  return make_self_contained(make_matrix(mu) + s * norm);
373  }

378  template<typename...Z>
379  auto log_likelihood(const Z&...z) const
380  {
381  static_assert((typed_matrix<Z> and ...));
382  static_assert((column_vector<Z> and ...));
383  static_assert(((MatrixTraits<Z>::dimension == dimension) and ...));
384  static constexpr auto n = sizeof...(Z);
385  static_assert(n >= 1);
386  auto sum = (trace(transpose(z - mu) * solve(sigma, z - mu)) + ...);
387  return -0.5 * (n * (dimension * std::log(2 * std::numbers::pi_v<Scalar>) + std::log(determinant(sigma))) + sum);
388  }
389 
390 
392  auto entropy() const
393  {
394  return 0.5 * (dimension * (1 + std::log2(std::numbers::pi_v<Scalar>) + std::numbers::log2e_v<Scalar>)
395  + std::log2(determinant(sigma)));
396  }
