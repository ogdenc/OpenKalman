256  auto operator()() const
257  {
258  auto norm = randomize<
259  Matrix<Coefficients, Axis, MatrixBase>, std::normal_distribution, random_number_engine>(0.0, 1.0);
260  auto s = square_root(sigma);
261  if constexpr(not lower_triangular_matrix<CovarianceBase>)
262  return make_self_contained(make_matrix(mu) + transpose(s) * norm);
263  else
264  return make_self_contained(make_matrix(mu) + s * norm);

269  template<typename...Z>
270  auto log_likelihood(const Z&...z) const
271  {
272  static_assert((column_vector<Z> and ...));
273  static_assert(((MatrixTraits<Z>::columns == 1) and ...));
274  static_assert(((MatrixTraits<Z>::dimension == dimension) and ...));
275  static constexpr auto n = sizeof...(Z);
276  static_assert(n >= 1);
277  auto sum = (trace(transpose(z - mu) * solve(sigma, z - mu)) + ...);
278  return -0.5 * (n * (dimension * std::log(2 * std::numbers::pi_v<Scalar>) + std::log(determinant(sigma))) + sum);
279  }
280 
282  auto entropy() const
283  {
284  return 0.5 * (dimension * (1 + std::log2(std::numbers::pi_v<Scalar>) + std::numbers::log2e_v<Scalar>)
285  + std::log2(determinant(sigma)));
286  }
