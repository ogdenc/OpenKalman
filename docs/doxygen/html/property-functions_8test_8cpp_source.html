156 TEST(eigen3, raw_data)
157 {
158  Eigen::Matrix<double, 2, 2, Eigen::ColMajor> m22_1234c;
159  m22_1234c << 1, 2, 3, 4;
160  auto* datac = internal::raw_data(m22_1234c);
161  static_assert(layout_of_v<decltype(m22_1234c)> == Layout::left);
162 
163  EXPECT_EQ(std::get<0>(internal::strides(m22_1234c)), 1);
164  EXPECT_EQ(std::get<1>(internal::strides(m22_1234c)), 2);
165  EXPECT_EQ(datac[0], 1);
166  EXPECT_EQ(datac[1], 3);
167  EXPECT_EQ(datac[2], 2);
168  EXPECT_EQ(datac[3], 4);
169 
170  Eigen::Matrix<double, 2, 2, Eigen::RowMajor> m22_1234r;
171  m22_1234r << 1, 2, 3, 4;
172  auto* datar = internal::raw_data(m22_1234r);
173  static_assert(layout_of_v<decltype(m22_1234r)> == Layout::right);
174  EXPECT_EQ(std::get<0>(internal::strides(m22_1234r)), 2);
175  EXPECT_EQ(std::get<1>(internal::strides(m22_1234r)), 1);
176 
177  EXPECT_EQ(datar[0], 1);
178  EXPECT_EQ(datar[1], 2);
179  EXPECT_EQ(datar[2], 3);
180  EXPECT_EQ(datar[3], 4);
181 }
