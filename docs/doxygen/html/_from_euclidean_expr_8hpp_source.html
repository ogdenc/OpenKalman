123  /**
124  * Assign from a compatible from-Euclidean expression.
125  */
126 #ifdef __cpp_concepts
127  template<from_euclidean_expr Arg>
128 #else
129  template<typename Arg, std::enable_if_t<from_euclidean_expr < Arg>, int> = 0>
130 #endif
131  auto& operator=(Arg&& other) noexcept
132  {
133  static_assert(equivalent_to < typename MatrixTraits<Arg>::RowCoefficients, Coefficients > );
134  static_assert(MatrixTraits<Arg>::columns == columns);
135  if constexpr (zero_matrix < NestedMatrix >)
136  {
137  static_assert(zero_matrix < Arg > );
138  }
139  else if constexpr (identity_matrix<NestedMatrix>)
140  {
141  static_assert(identity_matrix<Arg>);
142  }
143  else
144  {
145  this->nested_matrix() = std::forward<Arg>(other).nested_matrix();
146  }
147  return *this;
148  }

271  /**
272  * Access the coefficient at row i and column j
273  * \param i The row.
274  * \param j The column.
275  * \return If <code>element_settable<FromEuclideanExpr, 2></code>, the element is settable. Therefore,
276  * this function returns an object that can be assigned the coefficient to be set.
277  * Otherwise, it will return the (non-settable) coefficient as a value.
278  */
279  auto operator()(std::size_t i, std::size_t j)
280  {
281  if constexpr (element_settable < FromEuclideanExpr, 2 >)
282  return OpenKalman::internal::ElementSetter(*this, i, j);
283  else
284  return const_cast<const FromEuclideanExpr&>(*this)(i, j);
285  }

295  /**

297  * \param i The row.
298  * \return If <code>element_settable<FromEuclideanExpr, 1></code>, the element is settable. Therefore,

342  template<typename Coeffs, typename ArgType>
343  struct MatrixTraits<Eigen3::FromEuclideanExpr<Coeffs, ArgType>>
344  {
345  using NestedMatrix = ArgType;
346  using Scalar = typename MatrixTraits<NestedMatrix>::Scalar;
347  static constexpr auto dimension = Coeffs::size;
348  static constexpr auto columns = MatrixTraits<NestedMatrix>::columns;
349  using RowCoefficients = Coeffs;
350  using ColumnCoefficients = Axes<columns>;
351  static_assert(Coeffs::dimension == MatrixTraits<NestedMatrix>::dimension);
352 
353  template<typename Derived>
354  using MatrixBaseType = Eigen3::internal::Eigen3MatrixBase<Derived, Eigen3::FromEuclideanExpr<Coeffs, ArgType>>;
355 
356  template<std::size_t rows = dimension, std::size_t cols = columns, typename S = Scalar>
357  using NativeMatrix = typename MatrixTraits<NestedMatrix>::template NativeMatrix<rows, cols, S>;
358 
359  using SelfContained = Eigen3::FromEuclideanExpr<Coeffs, self_contained_t<NestedMatrix>>;
360 
361  template<TriangleType storage_triangle = TriangleType::lower, std::size_t dim = dimension, typename S = Scalar>
362  using SelfAdjointBaseType = Eigen3::SelfAdjointMatrix<NativeMatrix<dim, dim, S>, storage_triangle>;
363 
364  template<TriangleType triangle_type = TriangleType::lower, std::size_t dim = dimension, typename S = Scalar>
365  using TriangularBaseType = Eigen3::TriangularMatrix<NativeMatrix<dim, dim, S>, triangle_type>;
366 
367  template<std::size_t dim = dimension, typename S = Scalar>
368  using DiagonalBaseType = Eigen3::DiagonalMatrix<NativeMatrix<dim, 1, S>>;
369 
370  // Make from a regular matrix.
371 #ifdef __cpp_concepts
372  template<typename C = Coeffs, typename Arg> requires
373  Eigen3::eigen_matrix<Arg> or Eigen3::to_euclidean_expr<Arg>
374 #else
375  template<typename C = Coeffs, typename Arg,
376  std::enable_if_t<Eigen3::eigen_matrix<Arg> or Eigen3::to_euclidean_expr<Arg>, int> = 0>
377 #endif
378  static auto make(Arg&& arg) noexcept
379  {
380  static_assert(MatrixTraits<Arg>::dimension == C::dimension);
381  using namespace Eigen3;
382  return from_Euclidean<C>(std::forward<Arg>(arg));
383  }
384 
385 #ifdef __cpp_concepts
386  template<std::convertible_to<const Scalar> ... Args>
387 #else
388  template<typename ... Args, std::enable_if_t<
389  std::conjunction_v<std::is_convertible<Args, const Scalar>...>, int> = 0>
390 #endif
391  static auto
392  make(Args ... args)
393  {
394  static_assert(sizeof...(Args) == Coeffs::dimension * columns);
395  return make(MatrixTraits<NestedMatrix>::make(args...));
