55#ifdef __cpp_concepts
56 TypedMatrixBase() requires std::default_initializable<NestedMatrix> : Base {} {}
57#else
58 template<typename T = NestedMatrix, std::enable_if_t<std::is_default_constructible_v<T>, int> = 0>
59 TypedMatrixBase() : Base {} {}
60#endif

62
64#ifdef __cpp_concepts
65 template<typed_matrix_nestable Arg> requires (MatrixTraits<Arg>::rows == MatrixTraits<NestedMatrix>::rows) and
66 (MatrixTraits<Arg>::columns == MatrixTraits<NestedMatrix>::columns) and
67 std::constructible_from<NestedMatrix, Arg&&>
68#else
69 template<typename Arg, std::enable_if_t<typed_matrix_nestable<Arg> and
70 (MatrixTraits<Arg>::rows == MatrixTraits<NestedMatrix>::rows) and
71 (MatrixTraits<Arg>::columns == MatrixTraits<NestedMatrix>::columns) and
72 std::is_constructible_v<NestedMatrix, Arg&&>, int> = 0>
73#endif
74 TypedMatrixBase(Arg&& arg) noexcept : Base {std::forward<Arg>(arg)} {}
75
76
78#ifdef __cpp_concepts
79 template<std::convertible_to<const Scalar> ... Args> requires (sizeof...(Args) > 0) and
80 requires(Args ... args) { NestedMatrix {MatrixTraits<NestedMatrix>::make(static_cast<const Scalar>(args)...)}; }
81#else
82 template<typename ... Args, std::enable_if_t<
83 (std::is_convertible_v<Args, const Scalar> and ...) and (sizeof...(Args) > 0) and
84 ((diagonal_matrix<NestedMatrix> and std::is_constructible_v<NestedMatrix,
85 typename MatrixTraits<NestedMatrix>::template NativeMatrixFrom<sizeof...(Args), 1>>) or
86 (sizeof...(Args) == MatrixTraits<NestedMatrix>::rows * MatrixTraits<NestedMatrix>::columns and
87 std::is_constructible_v<NestedMatrix,
88 typename MatrixTraits<NestedMatrix>::template NativeMatrixFrom<MatrixTraits<NestedMatrix>::rows,
89 MatrixTraits<NestedMatrix>::columns>>)), int> = 0>
90#endif
91 TypedMatrixBase(Args ... args)
92 : Base {MatrixTraits<NestedMatrix>::make(static_cast<const Scalar>(args)...)} {}

100#ifdef __cpp_concepts
101 template<typed_matrix_nestable Arg>
102#else
103 template<typename Arg, std::enable_if_t<typed_matrix_nestable<Arg>, int> = 0>
104#endif
105 auto& operator=(Arg&& arg)
106 {
107 if constexpr (not zero_matrix<NestedMatrix> and not identity_matrix<NestedMatrix>)
108 {
109 Base::operator=(std::forward<Arg>(arg));
110 return *this;
111 }
112 }

116#ifdef __cpp_concepts
117 template<std::convertible_to<Scalar> S>
118#else
119 template<typename S, std::enable_if_t<std::is_convertible_v<S, Scalar>, int> = 0>
120#endif
121 auto& operator*=(const S s)
122 {
123 this->nested_matrix() *= s;
124 return *this;
125 }

129#ifdef __cpp_concepts

133#endif
134 auto& operator/=(const S s)

136 this->nested_matrix() /= s;
137 return *this;
