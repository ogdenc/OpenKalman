110  requires requires (std::size_t i){ {g(i)} -> scalar_type; }
111 #else
112  template<typename G, std::enable_if_t<scalar_type<typename std::invoke_result<G, std::size_t>::type>, int> = 0>
113  static constexpr auto
114  from_euclidean_element(const G& g, std::size_t local_index, std::size_t euclidean_start)
115 #endif
116  {
117  using Scalar = std::decay_t<decltype(g(std::declval<std::size_t>()))>;
118  Scalar d = g(euclidean_start + d2_i);
119  auto dr = internal::constexpr_real(d);
120  if (local_index == d_i)
121  {
122  // A negative distance is reflected to the positive axis.
123  using std::abs;
124  return internal::update_real_part(d, abs(dr));
125  }
126  else
127  {
128  using R = std::decay_t<decltype(internal::constexpr_real(std::declval<Scalar>()))>;
129  const Scalar cf {2 * numbers::pi_v<R> / (Limits::max - Limits::min)};
130  const Scalar mid {R{Limits::max + Limits::min} * R{0.5}};
131 
132  // If distance is negative, flip 180 degrees:
133  using std::signbit;
134  Scalar x = signbit(dr) ? -g(euclidean_start + x_i) : g(euclidean_start + x_i);
135  Scalar y = signbit(dr) ? -g(euclidean_start + y_i) : g(euclidean_start + y_i);
136 
137  if constexpr (complex_number<Scalar>) return internal::constexpr_atan2(y, x) / cf + mid;
138  else { using std::atan2; return atan2(y, x) / cf + mid; }
139  }
140  }

149 #endif

152  constexpr R period {Limits::max - Limits::min};
153  R ap {distance_is_negative ? internal::constexpr_real(a) + period * R{0.5} : internal::constexpr_real(a)};
154 
155  if (ap >= Limits::min and ap < Limits::max) // Check if the angle doesn't need wrapping.
156  {
157  return internal::update_real_part(std::forward<decltype(a)>(a), ap);;
158  }
159  else // Wrap the angle.
160  {
161  using std::fmod;
162  auto ar = fmod(ap - R{Limits::min}, period);
163  if (ar < 0) return internal::update_real_part(std::forward<decltype(a)>(a), R{Limits::min} + ar + period);
164  else return internal::update_real_part(std::forward<decltype(a)>(a), R{Limits::min} + ar);
