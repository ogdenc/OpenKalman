22  inline auto sum_of_squares = Transformation
23  {
24  [](const auto& x, const auto& ...ps) // function
25  {
26  static_assert(equivalent_to<vector_space_descriptor_of_t<decltype(x), 0>, Dimensions<n>>);
27  static_assert(equivalent_to<vector_space_descriptor_of_t<decltype(x), 1>, Axis > );
28  static_assert((equivalent_to<vector_space_descriptor_of_t<decltype(ps), 0>, Axis > and ...));
29  static_assert((equivalent_to<vector_space_descriptor_of_t<decltype(ps), 1>, Axis > and ...));
30 
31  return sum(((transpose(x) * x), ... , ps));
32  },
33  [](const auto& x, const auto& ...ps) // Jacobians
34  {
35  static_assert(equivalent_to<vector_space_descriptor_of_t<decltype(x), 0>, Dimensions<n>>);
36  static_assert(equivalent_to<vector_space_descriptor_of_t<decltype(x), 1>, Axis > );
37  static_assert((equivalent_to<vector_space_descriptor_of_t<decltype(ps), 0>, Axis > and ...));
38  static_assert((equivalent_to<vector_space_descriptor_of_t<decltype(ps), 1>, Axis > and ...));
39 
40  return std::tuple_cat(std::tuple {
41  2 * transpose(x)}, OpenKalman::internal::tuple_replicate<sizeof...(ps)>(Mean {1.}));
42  },
43  [](const auto& x, const auto& ...ps) // Hessians
44  {
45  static_assert(equivalent_to<vector_space_descriptor_of_t<decltype(x), 0>, Dimensions<n>>);
46  static_assert(equivalent_to<vector_space_descriptor_of_t<decltype(x), 1>, Axis > );
47  static_assert((equivalent_to<vector_space_descriptor_of_t<decltype(ps), 0>, Axis > and ...));
48  static_assert((equivalent_to<vector_space_descriptor_of_t<decltype(ps), 1>, Axis > and ...));
49 
50  std::array<Matrix<Dimensions<n>, Dimensions<n>, eigen_matrix_t<double, n, n>>, 1> I;
51  I[0] = 2 * eigen_matrix_t<double, n, n>::Identity();
52  return std::tuple {std::move(I), std::get<0>(zero_hessian<Axis>(ps))...};
53  }

57  inline auto time_of_arrival = Transformation
58  {
59  [](const auto& x, const auto& ...ps) // function
60  {
61  static_assert(equivalent_to<vector_space_descriptor_of_t<decltype(x), 0>, Dimensions<n>>);
62  static_assert(equivalent_to<vector_space_descriptor_of_t<decltype(x), 1>, Axis > );
63  static_assert((equivalent_to<vector_space_descriptor_of_t<decltype(ps), 0>, Axis > and ...));
64  static_assert((equivalent_to<vector_space_descriptor_of_t<decltype(ps), 1>, Axis > and ...));
65 
66  return sum(apply_coefficientwise([](const auto& c) { return std::sqrt(c); }, adjoint(x) * x), ps...);
67  },
68  [](const auto& x, const auto& ...ps) // Jacobians
69  {
70  static_assert(equivalent_to<vector_space_descriptor_of_t<decltype(x), 0>, Dimensions<n>>);
71  static_assert(equivalent_to<vector_space_descriptor_of_t<decltype(x), 1>, Axis > );
72  static_assert((equivalent_to<vector_space_descriptor_of_t<decltype(ps), 0>, Axis > and ...));
73  static_assert((equivalent_to<vector_space_descriptor_of_t<decltype(ps), 1>, Axis > and ...));
74 
75  return std::make_tuple(scalar_quotient(adjoint(x), std::sqrt(trace(adjoint(x) * x))),
76  Matrix<Axis, Axis, untyped_dense_writable_matrix_t<decltype(ps), Layout::none, scalar_type_of_t<decltype(ps)>, 1, 1>> {1.}...);
77  },
78  [](const auto& x, const auto& ...ps) // Hessians
79  {
80  static_assert(equivalent_to<vector_space_descriptor_of_t<decltype(x), 0>, Dimensions<n>>);
81  static_assert(equivalent_to<vector_space_descriptor_of_t<decltype(x), 1>, Axis > );
82  static_assert((equivalent_to<vector_space_descriptor_of_t<decltype(ps), 0>, Axis > and ...));
83  static_assert((equivalent_to<vector_space_descriptor_of_t<decltype(ps), 1>, Axis > and ...));
84 
85  std::array<Matrix<Dimensions<n>, Dimensions<n>, eigen_matrix_t<double, n, n>>, 1> ret;
86  double sq = trace(adjoint(x) * x);
87  ret[0] = pow(sq, -1.5) * (-x * adjoint(x) + sq * make_identity_matrix_like<decltype(ret[0])>());
88  return std::make_tuple(std::move(ret), std::get<0>(zero_hessian<Axis>(ps))...);
89  }

99  inline auto radar = Transformation
100  {
101  [](const M2t& x, const auto& ...ps) // function
102  {
103  return (M2t {x(0) * cos(x(1)), x(0) * sin(x(1))} + ... + ps);
104  },
105  [](const M2t& x, const auto& ...ps) // Jacobians
106  {
107  M22t ret = {
108  std::cos(x(1)), -x(0) * std::sin(x(1)),
109  std::sin(x(1)), x(0) * std::cos(x(1))};
110  return std::tuple_cat(std::tuple {std::move(ret)},
111  OpenKalman::internal::tuple_replicate<sizeof...(ps)>(make_identity_matrix_like(ret)));
112  },

155  inline auto Cartesian2polar = Transformation
156  {
157  [](const M2t& x, const auto& ...ps)
158  {
159  return (MP1t {std::hypot(x(0), x(1)), std::atan2(x(1), x(0))} + ... + Matrix {ps});
160  },
161  [](const M2t& x, const auto& ...ps) // Jacobians
162  {
163  const auto h = 1 / std::hypot(x(0), x(1));
164  const auto h2 = h * h;
165  MP2t ret = {
166  x(0) * h, x(1) * h,
167  -x(1) * h2, x(0) * h2};
168  return std::tuple_cat(std::tuple {
169  std::move(ret)}, OpenKalman::internal::tuple_replicate<sizeof...(ps)>(make_identity_matrix_like<MPPt>()));
170  },
171  [](const M2t& x, const auto& ...ps) // Hessians
172  {
173  const auto h = 1 / std::hypot(x(0), x(1));
174  const auto h2 = h * h;
175  const auto h3 = h2 * h;
176  const auto h42 = 2 * h2 * h2;
177  const auto x00 = x(0) * x(0);
178  const auto x01 = x(0) * x(1);
179  const auto x11 = x(1) * x(1);
180  std::array<M22t, 2> ret;
181  ret[0] = {h - x00 * h3, -x01 * h3,
182  -x01 * h3, h - x11 * h3};
183  ret[1] = {x01 * h42, h2 - x00 * h42,
184  x11 * h42 - h2, -x01 * h42};
185  return std::tuple {std::move(ret), std::get<0>(zero_hessian<Polar<>>(ps))...};
186  }
187  };

199  inline auto Cylindrical2spherical = Transformation
200  {
201  [](const MC1t& x, const auto& ...ps)
202  {
203  return (MS1t {std::hypot(x(0), x(2)), x(1), std::atan2(x(2), x(0))} + ... + Matrix {ps});
204  },
205  [](const MC1t& x, const auto& ...ps) // Jacobians
206  {
207  const auto h = 1 / std::hypot(x(0), x(2));
208  const auto h2 = h * h;
209  MSCt ret = {
210  x(0) * h, 0, x(2) * h,
211  0, 1, 0,
212  -x(2) * h2, 0, x(0) * h2};
213  return std::tuple_cat(std::tuple {
214  std::move(ret)}, OpenKalman::internal::tuple_replicate<sizeof...(ps)>(make_identity_matrix_like<MSSt>()));
215  },
216  [](const MC1t& x, const auto& ...ps) // Hessians
217  {
218  const auto h = 1 / std::hypot(x(0), x(2));
219  const auto h2 = h * h;
220  const auto h3 = h2 * h;
221  const auto h42 = 2 * h2 * h2;
222  const auto x00 = x(0) * x(0);
223  const auto x02 = x(0) * x(2);
224  const auto x22 = x(2) * x(2);
225  std::array<MCCt, 3> ret;
226  ret[0] = {h - x00 * h3, 0, -x02 * h3,
227  0, 0, 0,
228  -x02 * h3, 0, h - x22 * h3};
229  ret[1] = make_zero<MCCt>();
230  ret[2] = {x02 * h42, 0, h2 - x00 * h42,
231  0, 0, 0,
232  x22 * h42 - h2, 0, -x02 * h42};
233  return std::tuple {std::move(ret), std::get<0>(zero_hessian<Spherical<>>(ps))...};
234  }
