53  // Reconstruct the flattened augmented sample points into a tuple of tuples.
54  template<std::size_t pos = 0, typename FlattenedPs, typename D, typename...Ds>
55  static auto construct_ps(const FlattenedPs& flattened_ps, const D&, const Ds&...ds)
56  {

58  auto ps_group = internal::tuple_slice<pos, pos + group_size>(flattened_ps);

71  template<typename Transformation, typename...Points, typename...Dists, std::size_t...ints>
72  static constexpr auto y_means_impl(
73  const Transformation& g,
74  const std::tuple<Points...>& points,
75  const std::tuple<Dists...>& dists,
76  std::index_sequence<ints...>)
77  {
78  constexpr auto count = MatrixTraits<decltype(std::get<0>(points))>::columns;
79  return apply_columnwise<count>([&](size_t i) {
80  return g((column(std::get<ints>(points), i) + mean_of(std::get<ints>(dists)))...);
81  });
82  }

85  /*
86  * \brief
87  * \tparam dim The number of augmented sample points.
88  * \tparam InputDist The initial input distribution.
89  * \tparam return_cross Whether the transform must return a cross-covariance.
90  * \tparam i The index of the current transformation in Gs.
91  * \tparam Gs A sequence of transformations.
92  * \tparam D The input distribution for the current transformation in <var>Gs</var>.
93  * \tparam Ds A set comprising, for each transformation in <var>Gs</var>, a tuple of zero or more noise
94  * distributions that correspond to that transformation.
95  * \tparam P The augmented sample points corresponding to the current transformation in <var>Gs</var>.
96  * \tparam Ps The augmented sample points, as a tuple of tuples following the same structure as Ds.
97  * \return The posterior covariance or, if <code>return_cross</cross> is <code>true</code>, a tuple containing
98  * the posterior covariance and the cross-covariance.
99  */
100  template<std::size_t dim, typename InputDist, bool return_cross, std::size_t i = 0,
101  typename...Gs, typename D, typename...Ds, typename P, typename...Ps>
102  static auto transform_impl(
103  const std::tuple<Gs...>& gs,
104  const D& x,
105  const std::tuple<Ds...>& ds,
106  const P& xpoints,
107  const std::tuple<Ps...>& ps)
108  {
109  static_assert((sizeof...(Gs) == sizeof...(Ds)) and (sizeof...(Gs) == sizeof...(Ps)));
110  auto g = std::get<i>(gs);
111  auto xpoints_tup = std::tuple_cat(std::tuple {xpoints}, std::get<i>(ps));
112  auto xdists_tup = std::tuple_cat(std::tuple {x}, std::get<i>(ds));
113 
114  constexpr std::size_t N = std::tuple_size_v<decltype(xpoints_tup)>;
115  static_assert(N == std::tuple_size_v<decltype(xdists_tup)>);
116 
117  auto y_means = Mean {y_means_impl(g, xpoints_tup, xdists_tup, std::make_index_sequence<N>())};
118  auto y_mean = from_euclidean(SamplePointsType::template weighted_means<dim>(to_euclidean(y_means)));
119 
120  // Each column is a deviation from y mean for each transformed sigma point:
121  auto ypoints = apply_columnwise(y_means, [&](const auto& col) { return make_self_contained(col - y_mean); });
122 
123  if constexpr (i + 1 < sizeof...(Gs))
124  {
125  auto y_covariance = SamplePointsType::template covariance<dim, InputDist, false>(xpoints, ypoints);
126  auto y = GaussianDistribution {make_self_contained(std::move(y_mean)), std::move(y_covariance)};
127  return transform_impl<dim, InputDist, return_cross, i + 1>(gs, y, ds, ypoints, ps);
128  }
129  else // processing for the last transformation in Gs:
130  {
131  auto y_covariance = SamplePointsType::template covariance<dim, InputDist, return_cross>(xpoints, ypoints);
132  if constexpr (return_cross)
133  {
134  auto [y_cov, cross] = y_covariance;
135  return std::tuple {GaussianDistribution {make_self_contained(std::move(y_mean)), std::move(y_cov)},
136  std::move(cross)};
137  }
138  else
139  {
140  return GaussianDistribution {std::move(y_mean), std::move(y_covariance)};
141  }
142  }
143  }

157  template<bool return_cross = false, typename InputDist, typename...Ts>
158  auto transform(const InputDist& x, const Ts&...ts) const
159  {
160  auto gs = std::tuple {std::get<0>(ts)...}; //< Extract the transformations.
161  auto ds = std::make_tuple(internal::tuple_slice<1, std::tuple_size_v<Ts>>(ts)...); //< Extract the noise terms.
162 
163  // Flatten ds to a 1D tuple of noise terms:
164  auto flattened_ds = std::apply([](const auto&...args) {return std::tuple_cat(args...); }, ds);
165 
166  // Create a tuple comprising the augmented sample points (based on input x and each noise term in flattened_ds):
167  auto points_tuple = std::apply([&x](const auto&...args) {
168  return SamplePointsType::sample_points(x, args...);
169  }, flattened_ds);
170 
171  constexpr auto dim = count_dim<InputDist>(flattened_ds); //< Number of augmented sample points.
172  auto xpoints = std::get<0>(points_tuple); //< The augmented sample points corresponding to input x.
173 
174  // The augmented sample points corresponding to each noise term in flattened_ds.
175  auto flattened_ps = internal::tuple_slice<1, std::tuple_size_v<decltype(points_tuple)>>(points_tuple);
176 
177  // The augmented sample points, reconstructed into the same tuple-of-tuples structure as ds.
178  auto ps = std::apply([&](const auto&...args) { return construct_ps(flattened_ps, args...); }, ds);
179 
180  return transform_impl<dim, InputDist, return_cross>(gs, x, ds, xpoints, ps);
181  }
