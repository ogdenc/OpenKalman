443  {
444  using M = Eigen::Matrix<Scalar, dimension, MatrixTraits<B>::columns>;
445  auto llt = this->nested_view().llt();
446  M ret = llt.solve(b);
447  if (llt.info() != Eigen::Success) [[unlikely]]

450  auto ldlt = this->nested_view().ldlt();
451  M ret2 = ldlt.solve(b);
452  if ((not ldlt.isPositive() and not ldlt.isNegative()) or ldlt.info() != Eigen::Success)
453  {
454  throw (std::runtime_error("SelfAdjointMatrix solve: A is indefinite"));
455  }
456  return ret2;
457  }
458  return ret;
