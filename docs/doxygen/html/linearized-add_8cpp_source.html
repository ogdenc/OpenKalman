21TEST(transform_linear, linearized_additive1)
22{
23 Mat2 a {1, 2,
24 4, 3};
25 LinearTransformation g {a};
26 LinearizedTransform t;
27 //
28 const GaussianDistribution input0 {M2(1, 2), Mat2::identity()};
29 const GaussianDistribution noise0 {M2::zero(), Mat2::identity()};
30 const GaussianDistribution output0 {M2(5, 10), Mat2 {6, 10, 10, 26}};
31 auto [out0, cross0] = t.transform_with_cross_covariance(input0, g);
32 EXPECT_TRUE(is_near(out0 + noise0, output0));
33 EXPECT_TRUE(is_near(cross0, Mat2 {1, 4, 2, 3}));
34 //
35 const GaussianDistribution input1 {M2(1, 2), make_covariance<TriangleType::lower>(Mat2::identity())};
36 const GaussianDistribution noise1 {M2::zero(), make_covariance<TriangleType::lower>(Mat2::identity())};
37 const GaussianDistribution output1 {M2(5, 10), make_covariance<TriangleType::lower>( Mat2 {6, 10, 10, 26})};
38 auto [out1, cross1] = t.transform_with_cross_covariance(input1, g);
39 EXPECT_TRUE(is_near(out1 + noise1, output1));
40 EXPECT_TRUE(is_near(cross1, Mat2 {1, 4, 2, 3}));
41 //
42 const GaussianDistribution input2 {M2(1, 2), make_covariance<TriangleType::lower>(Mat2 {2, 1, 1, 2})};
43 const GaussianDistribution noise2 {M2::zero(), make_covariance<TriangleType::lower>(Mat2 {1, 0, 0, 1})};
44 auto [out2, cross2] = t.transform_with_cross_covariance(input2, g);
45 EXPECT_TRUE(is_near(out2 + noise2, GaussianDistribution{M2(5, 10), Mat2 {15, 31, 31, 75}}));
46 EXPECT_TRUE(is_near(cross2, Mat2 {4, 11, 5, 10}));
47
48 const GaussianDistribution input3 {M2(1, 2), make_covariance<TriangleType::upper>(Mat2 {2, 1, 1, 2})};
49 const GaussianDistribution noise3 {M2::zero(), make_covariance<TriangleType::upper>(Mat2 {1, 0, 0, 1})};
50 auto [out3, cross3] = t.transform_with_cross_covariance(input3, g);
51 EXPECT_TRUE(is_near(out3 + noise3, GaussianDistribution{M2(5, 10), Mat2 {15, 31, 31, 75}}));
52 EXPECT_TRUE(is_near(cross3, Mat2 {4, 11, 5, 10}));
53
54 const GaussianDistribution input4 {M2(1, 2), DiagonalMatrix(1., 1)};
55 const GaussianDistribution noise4 {M2::zero(), DiagonalMatrix(1., 1)};
56 auto [out4, cross4] = t.transform_with_cross_covariance(input4, g);

62TEST(transform_linear, linearized_dual)
63{
64 Mat2 a {1, 2,
65 3, 4};
66 LinearTransformation g {a};

68 GaussianDistribution input {M2(1, 2), Covariance(Mat2::identity())};

72 EXPECT_TRUE(is_near(covariance_of(out1), Mat2 {149, 325, 325, 709}));
