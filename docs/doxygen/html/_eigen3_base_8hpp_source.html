26 template<typename Derived>
27 struct Eigen3Base : Eigen::MatrixBase<Derived>
28 {
29 using Base = Eigen::MatrixBase<Derived>;
30
32 using Nested = Eigen3Base;

38#ifdef __cpp_concepts
39 static constexpr Eigen::Index rows() requires (not dynamic_rows<Derived>)
40#else
41 template<typename D = Derived, std::enable_if_t<(not dynamic_rows<D>), int> = 0>
42 static constexpr Eigen::Index rows()
43#endif
44 {
45 return MatrixTraits<Derived>::rows;
46 }

48
53#ifdef __cpp_concepts
54 constexpr Eigen::Index rows() const requires dynamic_rows<Derived>
55#else
56 template<typename D = Derived, std::enable_if_t<dynamic_rows<D>, int> = 0>
57 constexpr Eigen::Index rows() const
58#endif
59 {

63
68#ifdef __cpp_concepts
69 static constexpr Eigen::Index cols() requires (not dynamic_columns<Derived>)
70#else

72 static constexpr Eigen::Index cols()

78
83#ifdef __cpp_concepts
84 constexpr Eigen::Index cols() const requires dynamic_columns<Derived>
85#else
86 template<typename D = Derived, std::enable_if_t<dynamic_columns<D>, int> = 0>
87 constexpr Eigen::Index cols() const
88#endif
89 {

97#ifdef __cpp_concepts
98 template<std::convertible_to<std::size_t> ... Args> requires
99 (sizeof...(Args) == (dynamic_rows<Derived> ? 1 : 0) + (dynamic_columns<Derived> ? 1 : 0))
100#else
101 template<typename D = Derived, typename...Args, std::enable_if_t<
102 (std::is_convertible_v<Args, std::size_t> and ...) and
103 (sizeof...(Args) == (dynamic_rows<D> ? 1 : 0) + (dynamic_columns<D> ? 1 : 0)), int> = 0>
104#endif
105 static decltype(auto) zero(const Args...args)
106 {
107 return MatrixTraits<Derived>::zero(static_cast<std::size_t>(args)...);
108 }
109
110
116#ifdef __cpp_concepts
117 template<std::convertible_to<Eigen::Index> ... Args>
118#else
119 template<typename...Args, std::enable_if_t<(std::is_convertible_v<Args, Eigen::Index> and ...), int> = 0>
120#endif
121 [[deprecated("Use zero() instead.")]]
122 static decltype(auto) Zero(const Args...args)
123 {
124 if constexpr (sizeof...(Args) == (dynamic_rows<Derived> ? 1 : 0) + (dynamic_columns<Derived> ? 1 : 0))
125 return MatrixTraits<Derived>::zero(static_cast<std::size_t>(args)...);
126 else
127 return Base::Zero(static_cast<Eigen::Index>(args)...);
128 }
129
130
134#ifdef __cpp_concepts
135 template<std::convertible_to<Eigen::Index> ... Args> requires
136 (sizeof...(Args) == (dynamic_shape<Derived> ? 1 : 0))
137#else
138 template<typename D = Derived, typename...Args, std::enable_if_t<
139 (std::is_convertible_v<Args, Eigen::Index> and ...) and (sizeof...(Args) == (dynamic_shape<D> ? 1 : 0)), int> = 0>
140#endif
141 static decltype(auto) identity(const Args...args)
142 {
143 return MatrixTraits<Derived>::identity(static_cast<std::size_t>(args)...);
144 }
145
146
152#ifdef __cpp_concepts
153 template<std::convertible_to<Eigen::Index> Arg, std::convertible_to<Eigen::Index> ... Args> requires
154 (1 + sizeof...(Args) == (dynamic_rows<Derived> ? 1 : 0) + (dynamic_columns<Derived> ? 1 : 0))
155#else
156 template<typename D = Derived, typename Arg, typename...Args, std::enable_if_t<
157 (std::is_convertible_v<Arg, Eigen::Index> and ... and std::is_convertible_v<Args, Eigen::Index>) and
158 (1 + sizeof...(Args) == (dynamic_rows<D> ? 1 : 0) + (dynamic_columns<D> ? 1 : 0)), int> = 0>
159#endif
160 [[deprecated("Use identity() instead.")]]
161 static decltype(auto) Identity(const Arg arg, const Args...args)
162 {
163 if (((arg == args) and ...)) return identity(arg, args...);
164 else return Base::Identity(static_cast<Eigen::Index>(arg), static_cast<Eigen::Index>(args)...);
165 }
166
167
173#ifdef __cpp_concepts
174 [[deprecated("Use identity() instead.")]]
175 static decltype(auto) Identity() requires (not dynamic_shape<Derived>)
176#else
177 template<typename D = Derived, std::enable_if_t<not dynamic_shape<D>, int> = 0>
178 [[deprecated("Use identity() instead.")]]
179 static decltype(auto) Identity()
180#endif
181 {
182 return Base::Identity();
183 }
