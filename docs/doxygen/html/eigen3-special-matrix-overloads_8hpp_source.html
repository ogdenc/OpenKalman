137 #ifdef __cpp_concepts
138  template<typename V, typename ... Vs> requires
139  (eigen_self_adjoint_expr<V> and ... and eigen_self_adjoint_expr<Vs>) or
140  (eigen_triangular_expr<V> and ... and eigen_triangular_expr<Vs>)
141 #else
142  template<typename V, typename ... Vs, std::enable_if_t<
143  (eigen_self_adjoint_expr<V> and ... and eigen_self_adjoint_expr<Vs>) or
144  (eigen_triangular_expr<V> and ... and eigen_triangular_expr<Vs>), int> = 0>
145 #endif
146  constexpr decltype(auto)
147  concatenate_diagonal(V&& v, Vs&& ... vs)
148  {
149  if constexpr (sizeof...(Vs) > 0)
150  {
152  if constexpr (
153  (eigen_self_adjoint_expr<V> and
154  ((upper_triangular_storage<V> == upper_triangular_storage<Vs>) and ...)) or
155  (eigen_triangular_expr<V> and ((upper_triangular_matrix<V> == upper_triangular_matrix<Vs>) and ...)))
156  {
157  return MatrixTraits<V>::make(
158  concatenate_diagonal(nested_matrix(std::forward<V>(v)), nested_matrix(std::forward<Vs>(vs))...));
159  }
160  else if constexpr (eigen_self_adjoint_expr<V>)
161  {
162  constexpr auto t = MatrixTraits<V>::storage_triangle;
163  return concatenate_diagonal(std::forward<V>(v), make_EigenSelfAdjointMatrix<t>(std::forward<Vs>(vs))...);
164  }
165  else // eigen_triangular_expr<V> and there is a mixture of upper and lower triangles.
166  {
167  return concatenate_diagonal(make_native_matrix(std::forward<V>(v)), make_native_matrix(std::forward<Vs>(vs))...);
168  }
169  }
170  else
171  {
172  return std::forward<V>(v);
173  }
174  };
