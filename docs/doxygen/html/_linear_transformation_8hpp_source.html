42  template<
43  typename InputCoefficients_,
44  typename OutputCoefficients_,
45  typename TransformationMatrix,
46  typename ... PerturbationTransformationMatrices>
47  struct LinearTransformation
48  {
49  using InputCoefficients = InputCoefficients_;
50  using OutputCoefficients = OutputCoefficients_;
51  static_assert(typed_matrix_nestable<TransformationMatrix>);
52  static_assert((typed_matrix_nestable<PerturbationTransformationMatrices> and ...));
53  static_assert(MatrixTraits<TransformationMatrix>::dimension == OutputCoefficients::size);
54  static_assert(MatrixTraits<TransformationMatrix>::columns == InputCoefficients::size);
55  static_assert(((MatrixTraits<PerturbationTransformationMatrices>::dimension == OutputCoefficients::size) and ...));
56  static_assert(((MatrixTraits<PerturbationTransformationMatrices>::columns == OutputCoefficients::size) and ...));
57 
58  protected:
59  template<typename T, typename ColumnCoefficients = OutputCoefficients>
60  static constexpr bool is_valid_input_matrix_v()
61  {
62  static_assert(typed_matrix<T> or typed_matrix_nestable<T>);
63  if constexpr(typed_matrix<T>)
64  return
65  equivalent_to<typename MatrixTraits<T>::RowCoefficients, OutputCoefficients> and
66  equivalent_to<typename MatrixTraits<T>::ColumnCoefficients, ColumnCoefficients>;
67  else
68  return
69  MatrixTraits<T>::dimension == OutputCoefficients::size and
70  MatrixTraits<T>::columns == ColumnCoefficients::size;
71  }

73  template<typename Jacobians, typename InputTuple, std::size_t...ints>
74  constexpr auto sumprod(Jacobians&& js, InputTuple&& inputs, std::index_sequence<ints...>) const
75  {
76  return make_self_contained(
77  ((std::get<ints>(std::forward<Jacobians>(js)) * std::get<ints>(std::forward<InputTuple>(inputs))) + ...));
78  }

80  using TransformationMatricesTuple = std::tuple<
81  const Matrix<OutputCoefficients, InputCoefficients, self_contained_t<TransformationMatrix>>,
82  const Matrix<OutputCoefficients, OutputCoefficients, self_contained_t<PerturbationTransformationMatrices>>...>;
83  const TransformationMatricesTuple transformation_matrices;

85  public:

87  : transformation_matrices(mat, p_mats...) {}

89 #ifdef __cpp_concepts
90  template<typename T, typename ... Ps> requires
91  (typed_matrix<T> or typed_matrix_nestable<T>) and ((typed_matrix<Ps> or typed_matrix_nestable<Ps>) and ...)
92 #else
93  template<typename T, typename ... Ps, std::enable_if_t<
94  (typed_matrix<T> or typed_matrix_nestable<T>) and
95  ((typed_matrix<Ps> or typed_matrix_nestable<Ps>) and ...), int> = 0>
96 #endif
97  LinearTransformation(T&& mat, Ps&& ... p_mats) noexcept
98  : transformation_matrices(std::forward<T>(mat), std::forward<Ps>(p_mats)...)
99  {
100  static_assert(is_valid_input_matrix_v<T, InputCoefficients>());
101  static_assert((is_valid_input_matrix_v<Ps, OutputCoefficients>() and ...));
102  }

105 #ifdef __cpp_concepts
106  template<typed_matrix In, perturbation ... Perturbations> requires untyped_columns<In> and
107  internal::transformation_args<InputCoefficients, OutputCoefficients, In, Perturbations...>
108 #else
109  template<typename In, typename ... Perturbations, std::enable_if_t<
110  typed_matrix<In> and untyped_columns<In> and (perturbation<Perturbations> and ...) and
111  internal::transformation_args<InputCoefficients, OutputCoefficients, In, Perturbations...>, int> = 0>
112 #endif
113  auto operator()(const In& in, const Perturbations& ... ps) const
114  {
115  return sumprod(
116  jacobian(in, ps...),
117  std::forward_as_tuple(in, internal::get_perturbation(ps)...),
118  std::make_index_sequence<sizeof...(Perturbations) + 1>{});
119  }

122 #ifdef __cpp_concepts
123  template<typed_matrix In, perturbation ... Perturbations> requires untyped_columns<In> and
124  internal::transformation_args<InputCoefficients, OutputCoefficients, In, Perturbations...>
125 #else
126  template<typename In, typename ... Perturbations, std::enable_if_t<
127  typed_matrix<In> and untyped_columns<In> and (perturbation<Perturbations> and ...) and
128  internal::transformation_args<InputCoefficients, OutputCoefficients, In, Perturbations...>, int> = 0>
129 #endif
130  auto jacobian(const In&, const Perturbations&...) const
131  {
132  constexpr auto mat_count = std::tuple_size_v<TransformationMatricesTuple>;
133 
134  // If there are more inputs than transformation matrices, pad the list with extra identity matrices.
135  if constexpr(sizeof...(Perturbations) + 1 > mat_count)
136  {
137  constexpr auto pad_size = sizeof...(Perturbations) + 1 - mat_count;
138  auto id = make_matrix<OutputCoefficients, OutputCoefficients>(MatrixTraits<TransformationMatrix>::identity());
139  return std::tuple_cat(transformation_matrices, internal::tuple_replicate<pad_size>(id));
140  }
141  else
142  {
143  return internal::tuple_slice<0, sizeof...(Perturbations) + 1>(transformation_matrices);
144  }
145  }
146 
148 #ifdef __cpp_concepts
149  template<typed_matrix In, perturbation ... Perturbations> requires untyped_columns<In> and
150  internal::transformation_args<InputCoefficients, OutputCoefficients, In, Perturbations...>
151 #else
152  template<typename In, typename ... Perturbations, std::enable_if_t<
153  typed_matrix<In> and untyped_columns<In> and (perturbation<Perturbations> and ...) and
154  internal::transformation_args<InputCoefficients, OutputCoefficients, In, Perturbations...>, int> = 0>
155 #endif
156  auto hessian(const In&, const Perturbations&...) const
157  {
158  return zero_hessian<OutputCoefficients, In, Perturbations...>();
159  }
