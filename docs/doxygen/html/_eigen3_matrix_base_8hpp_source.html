42  template<typename Arg>
43  auto& get_ultimate_nested_matrix_impl(Arg& arg)
44  {
45  auto& b = nested_matrix(arg);
46  using B = decltype(b);
47  static_assert(not std::is_const_v<std::remove_reference_t<B>>);
48  if constexpr(Eigen3::eigen_self_adjoint_expr<B> or Eigen3::eigen_triangular_expr<B> or
49  Eigen3::eigen_diagonal_expr<B> or Eigen3::euclidean_expr<B>)
50  {
51  return get_ultimate_nested_matrix(b);
52  }
53  else
54  {
55  return b;
56  }
57  }

60  template<typename Arg>
61  auto& get_ultimate_nested_matrix(Arg& arg)
62  {
63  if constexpr(Eigen3::eigen_self_adjoint_expr<Arg>)
64  {
65  if constexpr (MatrixTraits<Arg>::storage_triangle == TriangleType::diagonal) return arg;
66  else return get_ultimate_nested_matrix_impl(arg);
67  }
68  else if constexpr(Eigen3::eigen_triangular_expr<Arg>)
69  {
70  if constexpr(MatrixTraits<Arg>::triangle_type == TriangleType::diagonal) return arg;
71  else return get_ultimate_nested_matrix_impl(arg);
72  }
73  else
74  {
75  return get_ultimate_nested_matrix_impl(arg);
76  }
77  }

329  if constexpr (diagonal_matrix<CovNest>)

343  cov = OpenKalman::internal::to_covariance_nestable<CovNest>(std::move(b));
344  }
