21 inline auto get_t2()
22 {
23 auto theta = trace(randomize<Mean<Axis, eigen_matrix_t<double, 1, 1>>, std::uniform_real_distribution>(-pi, pi));
24 using M22 = eigen_matrix_t<double, 2, 2>;
25 auto a = Matrix < Axes<2>, Axes<2>, M22> { std::cos(theta), -std::sin(theta), std::sin(theta), std::cos(theta) };
26 return LinearTransformation(a);
27 }

30 template<typename Cov, typename Trans>
31 void rotation_2D(const Trans& transform)
32 {
33 using M2 = eigen_matrix_t<double, 2, 1>;
34 using Mean2 = Mean<Axes<2>, M2>;
35 for (int i = 0; i < 5; i++)
36 {
37 auto true_state = randomize<Mean2, std::uniform_real_distribution>(5.0, 10.0);
38 auto x = GaussianDistribution < Axes<2>, M2, Cov> { Mean2 {7.5, 7.5}, Cov::identity() };
39 auto meas_cov = Cov {0.01, 0, 0, 0.01};
40 auto r = GaussianDistribution < Axes<2>, M2, Cov> { Mean2::zero(), meas_cov };
41 parameter_test(transform, get_t2(), x, true_state, r, 0.2, 100);
42 }
43 }
