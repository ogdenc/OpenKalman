18 TEST(eigen3, to_diagonal)
19 {
20  auto m11 = M11 {3};
21  auto m1x_1 = M1x {m11};
22  auto mx1_1 = Mx1 {m11};
23  Mxx mxx_11(1,1); mxx_11 << 3;
24 
25  // diagonal_matrix input:
26 
27  EXPECT_TRUE(is_near(to_diagonal(Eigen::DiagonalWrapper {m11}), m11));
28  EXPECT_TRUE(is_near(to_diagonal(Eigen::DiagonalMatrix<double, 1> {m11}), m11));
29  EXPECT_TRUE(is_near(to_diagonal(m11.selfadjointView<Eigen::Lower>()), m11));
30  EXPECT_TRUE(is_near(to_diagonal(m11.triangularView<Eigen::Upper>()), m11));
31  EXPECT_TRUE(is_near(to_diagonal(Eigen3::make_eigen_wrapper(m11.triangularView<Eigen::Upper>())), m11));
32 
33  // constant one-by-one input:
34 
35  static_assert(constant_coefficient_v<decltype(to_diagonal(M11::Identity() + M11::Identity()))> == 2);
36 
37  // Handled by dense Eigen interface:
38 
39  EXPECT_TRUE(is_near(to_diagonal(m11), m11)); static_assert(std::is_same_v<decltype(to_diagonal(m11)), M11&>);
40  EXPECT_TRUE(is_near(to_diagonal(m1x_1), m11)); static_assert(not has_dynamic_dimensions<decltype(to_diagonal(m1x_1))>);
41  EXPECT_TRUE(is_near(to_diagonal(mx1_1), m11)); static_assert(has_dynamic_dimensions<decltype(to_diagonal(mx1_1))>); static_assert(eigen_DiagonalWrapper<decltype(to_diagonal(mx1_1))>);
42  EXPECT_TRUE(is_near(to_diagonal(mxx_11), m11)); static_assert(has_dynamic_dimensions<decltype(to_diagonal(mxx_11))>); static_assert(eigen_DiagonalWrapper<decltype(to_diagonal(mxx_11))>);
43 
44  EXPECT_TRUE(is_near(to_diagonal(M11 {m11}), m11)); static_assert(std::is_same_v<decltype(to_diagonal(M11 {m11})), M11&&>);
45  EXPECT_TRUE(is_near(to_diagonal(M1x {m11}), m11)); static_assert(not has_dynamic_dimensions<decltype(to_diagonal(M1x {m11}))>);
46  // to_diagonal(M1x {m11}) must create OpenKalman::DiagonalMatrix. See adapters/tests/DiagonalMatrix.test.cpp
47  // to_diagonal(Mxx {m11}) must create OpenKalman::DiagonalMatrix. See adapters/tests/DiagonalMatrix.test.cpp
48 }

51 TEST(eigen3, diagonal_of_1x1)

61 
62  // Note: dynamic one-by-one, known at compile time requires creation of ConstantAdapter and is tested elsewhere.

65

68  Mxx mxx_11(1,1); mxx_11 = m11;

71  static_assert(index_count_v<decltype(diagonal_of(m11))> == 0);

73  static_assert(index_count_v<decltype(diagonal_of(mx1_1))> == 1);

75

80 
81  static_assert(dynamic_dimension<decltype(diagonal_of(mxx_11)), 0>);

92  static_assert(not constant_matrix<decltype(diagonal_of(mxx_11))>);
