60  template<typename _Distribution, typename _Noise>
61  _Distribution &
62  predict(_Distribution& x,
63  const _Noise& process_noise) const
64  {
65  x = predict(x);
66 
67  std::random_device rd;
68  std::mt19937 gen = std::mt19937(rd());
69 
70  _Noise::covariance_t L = process_noise.get_sqrt_covariance();
71 
72  const auto add_noise = [&](const State& v) -> const State
73  {
74  // Create uncorrelated, normalized random variable (mean 0, variance 1)
75  State Xnorm;
76  for (int i=0; i<Samples_t::dim; i++) {
77  Xnorm(i) = std::normal_distribution<>(0,1)(gen);
78  }
79  return v + L * Xnorm;
80  };
81 
82  std::generate(x.begin(), x.end(), add_noise);
83  return x;
84  }
