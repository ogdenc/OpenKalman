55  return values::scalar_constant_operation{op, constant_coefficient{std::forward<Arg>(arg)}, n};
56  }
57  else if constexpr (constant_diagonal_matrix<Arg>)

59  std::multiplies<scalar_type_of_t<Arg>> op;
60  auto n = internal::index_to_scalar_constant<scalar_type_of_t<Arg>>(
61  internal::smallest_vector_space_descriptor(get_index_dimension_of<0>(arg), get_index_dimension_of<1>(arg)));;
62  return values::scalar_constant_operation{op, constant_diagonal_coefficient{std::forward<Arg>(arg)}, n};
63  }
64  else if constexpr (triangular_matrix<Arg>) // Includes the diagonal case.
65  {
66  return reduce(std::plus<scalar_type_of_t<Arg>>{}, diagonal_of(std::forward<Arg>(arg)));
67  }
68  else // General case in which we have to add up the diagonal elements.
69  {

71  decltype(auto) diag {diagonal_of(std::forward<Arg>(arg))};
72  if constexpr(dynamic_dimension<decltype(diag), 0>)

75  if (dim >= 2) return static_cast<Scalar>(reduce(std::plus<Scalar>{}, std::forward<decltype(diag)>(diag)));
