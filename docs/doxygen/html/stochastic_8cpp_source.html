19 TEST(transformations, stochastic_additive)
20 {
21  auto f = [](const auto& x) -> M { return A {1, 2, 3, 4} * x; };
22  auto dist = GaussianDistribution {make_zero<M>(), make_identity_matrix_like(A{})};
23  auto t = Transformation<decltype(f)> {f};
24  M x {2, 3};
25  M true_y {f(x)};
26  M mean_y {make_zero<M>()};
27  for (int i = 0; i < 100; i++)
28  {
29  const M y {t(x) + dist()};
30  mean_y = (mean_y * i + y) / (i + 1);
31  }
32  EXPECT_NE(mean_y, true_y);
33  EXPECT_TRUE(is_near(mean_y, true_y, nested_object_of_t<M>::Constant(1.0)));
34 }

37 TEST(transformations, stochastic_augmented)
