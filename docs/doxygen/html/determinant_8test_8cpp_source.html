18 TEST(eigen3, determinant)
19 {
20  auto m22 = make_dense_object_from<M22>(1, 2, 3, 4);
21 
22  EXPECT_NEAR(determinant(m22), -2, 1e-6);
23  EXPECT_NEAR(determinant(m22.array()), -2, 1e-6);
24  EXPECT_NEAR(determinant(Eigen3::make_eigen_wrapper(m22)), -2, 1e-6);
25  EXPECT_NEAR(determinant(M2x {m22}), -2, 1e-6);
26  EXPECT_NEAR(determinant(Mx2 {m22}), -2, 1e-6);
27  EXPECT_NEAR(determinant(Mxx {m22}), -2, 1e-6);
28 
29  const M00 m00 {};
30 
31  EXPECT_NEAR(determinant(m00), 1, 1e-6);
32  EXPECT_NEAR(determinant(M0x {m00}), 1, 1e-6);
33  EXPECT_NEAR(determinant(Mx0 {m00}), 1, 1e-6);
34  EXPECT_NEAR(determinant(Mxx {m00}), 1, 1e-6);
35 
36  EXPECT_NEAR(determinant(M11{2}), 2, 1e-6);
37  EXPECT_NEAR(determinant((M1x(1,1) << 2).finished()), 2, 1e-6);
38  EXPECT_NEAR(determinant((Mx1(1,1) << 2).finished()), 2, 1e-6);
39  EXPECT_NEAR(determinant((Mxx(1,1) << 2).finished()), 2, 1e-6);
40 
41  auto cm22 = make_dense_object_from<CM22>(cdouble {1,4}, cdouble {2,3}, cdouble {3,2}, cdouble {4,1});
42 
43  EXPECT_NEAR(std::real(determinant(cm22)), 0, 1e-6);
44  EXPECT_NEAR(std::imag(determinant(cm22)), 4, 1e-6);
45 
46  EXPECT_NEAR(std::real(determinant(cm22.array())), 0, 1e-6);

48 
49  EXPECT_NEAR(std::real(determinant(Eigen3::make_eigen_wrapper(cm22))), 0, 1e-6);
50  EXPECT_NEAR(std::imag(determinant(Eigen3::make_eigen_wrapper(cm22))), 4, 1e-6);

53
