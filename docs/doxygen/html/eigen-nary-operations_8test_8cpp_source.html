25  EXPECT_TRUE(is_near(n_ary_operation<Mxx>(std::tuple {Dimensions{2}, Dimensions<3>{}}, [](std::size_t r, std::size_t c){return 5.5 + r + c;}), m23 + m23p));

136 TEST(eigen3, unary_operation_in_place)
137 {
138  const auto m23c = make_dense_object_from<M23>(1, 2, 3, 4, 5, 6);
139  M23 m23 {m23c};
140  M2x m2x_3 {m23};
141  Mx3 mx3_2 {m23};
142  Mxx mxx_23 {m23};
143  auto ewm23 = Eigen3::make_eigen_wrapper(m23);
144 
145  EXPECT_TRUE(is_near(unary_operation_in_place([](auto& arg){return arg *= 2;}, m23), m23c * 2));
146  EXPECT_TRUE(is_near(unary_operation_in_place([](auto& arg){return arg *= 3;}, ewm23), m23c * 6));
147  EXPECT_TRUE(is_near(unary_operation_in_place([](auto& arg){return arg *= 4;}, Eigen3::make_eigen_wrapper(m23)), m23c * 24));
148 
149  EXPECT_TRUE(is_near(unary_operation_in_place([](auto& arg){return arg *= 3;}, m2x_3), m23c * 3));
150  EXPECT_TRUE(is_near(unary_operation_in_place([](auto& arg){return arg *= 4;}, mx3_2), m23c * 4));
151  EXPECT_TRUE(is_near(unary_operation_in_place([](auto& arg){return arg *= 5;}, mxx_23), m23c * 5));
152 
153  const auto m33_123 = make_dense_object_from<M33>(
154  1, 0, 0,
155  0, 2, 0,
156  0, 0, 3);
157 
158  const auto m33_234 = make_dense_object_from<M33>(
159  2, 1, 1,
160  1, 3, 1,
161  1, 1, 4);
162 
163  auto m33 = M33 {m33_123}; unary_operation_in_place([](auto& x){ return x += 1; }, m33); EXPECT_TRUE(is_near(m33, m33_234));
164  auto m30 = M3x {m33_123}; unary_operation_in_place([](auto& x){ return x += 1; }, m30); EXPECT_TRUE(is_near(m30, m33_234));
165  auto m03 = Mx3 {m33_123}; unary_operation_in_place([](auto& x){ return x += 1; }, m03); EXPECT_TRUE(is_near(m03, m33_234));
166  auto mxx = Mxx {m33_123}; unary_operation_in_place([](auto& x){ return x += 1; }, mxx); EXPECT_TRUE(is_near(mxx, m33_234));
167  auto ewm33 = Eigen3::make_eigen_wrapper(M33 {m33_123}); unary_operation_in_place([](auto& x){ return x += 1; }, ewm33); EXPECT_TRUE(is_near(ewm33, m33_234));
168 
169  EXPECT_TRUE(is_near(unary_operation_in_place([](double& x){ ++x; }, M33 {m33_123}), m33_234));
170  EXPECT_TRUE(is_near(unary_operation_in_place([](double& x){ ++x; }, M3x {m33_123}), m33_234));
171  EXPECT_TRUE(is_near(unary_operation_in_place([](double& x){ ++x; }, Mx3 {m33_123}), m33_234));

173

175  EXPECT_TRUE(is_near(unary_operation_in_place([](double& x){ ++x; }, m33), m33_234));
176  EXPECT_TRUE(is_near(m33, m33_234));
