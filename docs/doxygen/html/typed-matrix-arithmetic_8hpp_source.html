17  /// Add two typed matrices. If the operands are of different types, the result will be a regular typed matrix.
18 #ifdef __cpp_concepts
19  template<typed_matrix V1, typed_matrix V2>
20 #else
21  template<typename V1, typename V2, std::enable_if_t<typed_matrix<V1> and typed_matrix<V2>, int> = 0>
22 #endif
23  inline auto operator+(V1&& v1, V2&& v2)
24  {
25  using RC1 = typename MatrixTraits<V1>::RowCoefficients;
26  using CC1 = typename MatrixTraits<V1>::ColumnCoefficients;
27  static_assert(equivalent_to<typename MatrixTraits<V2>::RowCoefficients, RC1>);
28  static_assert(equivalent_to<typename MatrixTraits<V2>::ColumnCoefficients, CC1>);
29  static_assert(euclidean_transformed<V1> == euclidean_transformed<V2>);
30  using CommonV = std::decay_t<std::conditional_t<
31  (euclidean_mean<V1> and euclidean_mean<V2>) or (mean<V1> and mean<V2>),
32  V1, decltype(Matrix {v1})>>;
33  auto ret = MatrixTraits<CommonV>::make(nested_matrix(std::forward<V1>(v1)) + nested_matrix(std::forward<V2>(v2)));
34  return make_self_contained<V1, V2>(std::move(ret));
35  }

38  /// Subtract two typed matrices. The result is a regular typed matrix unless both operands are EuclideanMean.
39 #ifdef __cpp_concepts
40  template<typed_matrix V1, typed_matrix V2>
41 #else
42  template<typename V1, typename V2, std::enable_if_t<typed_matrix<V1> and typed_matrix<V2>, int> = 0>
43 #endif
44  inline auto operator-(V1&& v1, V2&& v2)
45  {
46  using RC1 = typename MatrixTraits<V1>::RowCoefficients;
47  using CC1 = typename MatrixTraits<V1>::ColumnCoefficients;
48  static_assert(equivalent_to<typename MatrixTraits<V2>::RowCoefficients, RC1>);
49  static_assert(equivalent_to<typename MatrixTraits<V2>::ColumnCoefficients, CC1>);
50  static_assert(euclidean_transformed<V1> == euclidean_transformed<V2>);
51  using CommonV = std::decay_t<std::conditional_t<
52  (euclidean_mean<V1> and euclidean_mean<V2>), V1, decltype(Matrix {v1})>>;
53  auto b = nested_matrix(std::forward<V1>(v1)) - nested_matrix(std::forward<V2>(v2));
54  if constexpr (mean<V1> and mean<V2>)
55  {
56  // WC is the difference type for the coefficients. However, the result should retain coefficient types RC1.
57  using WC = typename RC1::difference_type;
58  auto ret = MatrixTraits<CommonV>::make(wrap_angles<WC>(std::move(b)));
59  return make_self_contained<V1, V2>(std::move(ret));
60  }
61  else
62  {
63  auto ret = MatrixTraits<CommonV>::make(std::move(b));
64  return make_self_contained<V1, V2>(std::move(ret));
65  }
66  }

69  /// Multiply a typed matrix by a scalar. The result type is the same as the operand type, so angles in the result may be wrapped.
70 #ifdef __cpp_concepts
71  template<typed_matrix V, std::convertible_to<const typename MatrixTraits<V>::Scalar> S>
72 #else
73  template<typename V, typename S, std::enable_if_t<
74  typed_matrix<V> and std::is_convertible_v<S, const typename MatrixTraits<V>::Scalar>, int> = 0>
75 #endif
76  inline auto operator*(V&& v, S scale)
77  {
78  using Sc = typename MatrixTraits<V>::Scalar;
79  auto ret = MatrixTraits<V>::make(nested_matrix(std::forward<V>(v)) * static_cast<Sc>(scale));
80  return make_self_contained<V>(std::move(ret));
81  }

84  /// Multiply a scalar by a typed matrix. The result type is the same as the operand type, so angles in the result may be wrapped.
85 #ifdef __cpp_concepts
86  template<typed_matrix V, std::convertible_to<const typename MatrixTraits<V>::Scalar> S>
87 #else
88  template<typename V, typename S, std::enable_if_t<typed_matrix<V> and
89  std::is_convertible_v<S, const typename MatrixTraits<V>::Scalar>, int> = 0>
90 #endif
91  inline auto operator*(S scale, V&& v)
92  {
93  using Sc = const typename MatrixTraits<V>::Scalar;
94  auto ret = MatrixTraits<V>::make(static_cast<Sc>(scale) * nested_matrix(std::forward<V>(v)));
95  return make_self_contained<V>(std::move(ret));

114  /// Multiply a typed matrix by another typed matrix. The result is a regular typed matrix unless the first operand is EuclideanMean.
115 #ifdef __cpp_concepts
116  template<typed_matrix V1, typed_matrix V2>
117 #else
118  template<typename V1, typename V2, std::enable_if_t<typed_matrix<V1> and typed_matrix<V2>, int> = 0>
119 #endif
120  inline auto operator*(V1&& v1, V2&& v2)
121  {
122  static_assert(equivalent_to<typename MatrixTraits<V1>::ColumnCoefficients, typename MatrixTraits<V2>::RowCoefficients>);
123  static_assert(MatrixTraits<V1>::columns == MatrixTraits<V2>::dimension);
124  using RC = typename MatrixTraits<V1>::RowCoefficients;
125  using CC = typename MatrixTraits<V2>::ColumnCoefficients;
126  auto b = nested_matrix(std::forward<V1>(v1)) * nested_matrix(std::forward<V2>(v2));
127  using CommonV = std::decay_t<std::conditional_t<euclidean_mean<V1>, V1, decltype(Matrix {v1})>>;
128  auto ret = MatrixTraits<CommonV>::template make<RC, CC>(std::move(b));
129  return make_self_contained<V1, V2>(std::move(ret));
130  }

133  /// Negate a vector object. The result is a regular typed matrix unless the operand is EuclideanMean.
134 #ifdef __cpp_concepts
135  template<typed_matrix V>
136 #else
137  template<typename V, std::enable_if_t<typed_matrix<V>, int> = 0>
138 #endif
139  inline auto operator-(V&& v)
140  {
141  using Res = std::decay_t<std::conditional_t<euclidean_mean<V>, V, decltype(Matrix {v})>>;
142  auto ret = MatrixTraits<Res>::make(-nested_matrix(std::forward<V>(v)));
143  return make_self_contained<V>(std::move(ret));
144  }

147  /// Equality operator.
148 #ifdef __cpp_concepts
149  template<typed_matrix V1, typed_matrix V2>
150 #else
151  template<typename V1, typename V2, std::enable_if_t<typed_matrix<V1> and typed_matrix<V2>, int> = 0>
152 #endif
153  constexpr auto operator==(V1&& v1, V2&& v2)
154  {
155  if constexpr(
156  equivalent_to<typename MatrixTraits<V1>::RowCoefficients, typename MatrixTraits<V2>::RowCoefficients> and
157  equivalent_to<typename MatrixTraits<V1>::ColumnCoefficients, typename MatrixTraits<V2>::ColumnCoefficients>)
158  {
159  return make_native_matrix(std::forward<V1>(v1)) == make_native_matrix(std::forward<V2>(v2));
160  }
161  else
162  {
163  return false;
164  }
165  }
