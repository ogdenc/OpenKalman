28  template<template<typename Scalar> typename Limits, typename Scalar>
29  static inline Scalar polar_angle_wrap_impl(const bool distance_is_negative, const Scalar s)
30  {
31  constexpr Scalar max = Limits<Scalar>::max;
32  constexpr Scalar min = Limits<Scalar>::min;
33  constexpr Scalar period = max - min;
34 
35  Scalar a = distance_is_negative ? s + period * 0.5 : s;
36 
37  if (a >= min and a < max) // Check if the angle doesn't need wrapping.
38  {
39  return a;
40  }
41  else // Wrap the angle.
42  {
43  Scalar ar = std::fmod(a - min, period);
44  if (ar < 0)
45  {
46  ar += period;
47  }
48  return ar + min;
49  }
50  }

102  static constexpr Scalar cf = 2 * std::numbers::pi_v<Scalar> / (Limits<Scalar>::max - Limits<Scalar>::min);
103 
104  template<std::size_t i, std::size_t d_i, std::size_t a_i>
105  static constexpr std::array<Scalar (*const)(const GetCoeff&), 2>
106  to_euclidean_array =
107  {
108  [](const GetCoeff& get_coeff) { return std::cos(get_coeff(i + a_i) * cf); },
109  [](const GetCoeff& get_coeff) { return std::sin(get_coeff(i + a_i) * cf); }
110  };
111 
112  template<std::size_t i, std::size_t d_i, std::size_t x_i, std::size_t y_i>
113  static constexpr std::array<Scalar (*const)(const GetCoeff&), 1>
114  from_euclidean_array = {[](const GetCoeff& get_coeff)
115  {
116  const auto x = std::signbit(get_coeff(i + d_i)) ? -get_coeff(i + x_i) : get_coeff(i + x_i);
117  const auto y = std::signbit(get_coeff(i + d_i)) ? -get_coeff(i + y_i) : get_coeff(i + y_i);
118  return std::atan2(y, x) / cf;
