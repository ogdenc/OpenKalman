325  auto operator()() const
326  {
327  auto norm = randomize<Matrix<FixedDescriptor, Axis, MeanNestedMatrix>, random_number_engine>(
328  std::normal_distribution {0.0, 1.0});
329  auto s = square_root(sigma);
330  if constexpr(triangular_matrix<decltype(s), TriangleType::upper>)
331  return sum(Matrix {mu}, contract(transpose(std::move(s)), std::move(norm)));
332  else
333  return sum(Matrix {mu}, contract(std::move(s), std::move(norm)));
334  }

341 #ifdef __cpp_concepts
342  template<typed_matrix...Z> requires (sizeof...(Z) > 0) and
343  ((vector<Z> and equivalent_to<vector_space_descriptor_of_t<Z, 0>, FixedDescriptor>) and ...)
344 #else
345  template<typename...Z, std::enable_if_t<(sizeof...(Z) > 0) and ((typed_matrix<Z> and vector<Z> and
346  equivalent_to<vector_space_descriptor_of_t<Z, 0>, FixedDescriptor>) and ...), int> = 0>
347 #endif
348  Scalar log_likelihood(const Z&...z) const
349  {
350  static constexpr auto n = sizeof...(Z);
351  auto sum = (trace(transpose(z - mu) * solve(sigma, z - mu)) + ...);
352  return -0.5 * (n * (dim * interface::scalar_traits<Scalar>::log(2 * numbers::pi_v<long double>) +
353  interface::scalar_traits<Scalar>::log(determinant(sigma))) + sum);
354  }
355 
356 
358  Scalar entropy() const
359  {
360  return 0.5 * (dim * (1 + interface::scalar_traits<Scalar>::log2(numbers::pi_v<long double>) +
361  Scalar {numbers::log2e_v<long double>}) + interface::scalar_traits<Scalar>::log2(determinant(sigma)));
362  }
