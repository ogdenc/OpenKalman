63  constexpr auto get_slice_impl(Arg&& arg, const std::tuple<Begin...>& begin, const std::tuple<Size...>& size)
64  {
65  if constexpr (zero<Arg>)
66  {
67  return std::apply([](auto&&...ds){
68  return make_zero<Arg>(std::forward<decltype(ds)>(ds)...);
69  }, get_slice_descriptors(std::forward<Arg>(arg), begin, size));
70  }
71  else if constexpr (constant_matrix<Arg>)

77  }
