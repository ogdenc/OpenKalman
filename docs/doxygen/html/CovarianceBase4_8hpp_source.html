399  auto& operator=(const CovarianceBase& other)
400  {
401  if constexpr (not zero<NestedMatrix> and not identity_matrix<NestedMatrix>) if (this != &other)
402  {
403  if (other.synchronization_direction() > 0)
404  {
405  Base::nested_object() = other.nested_object();
406  mark_nested_matrix_changed();
407  }
408  else if (other.synchronization_direction() < 0)
409  {
410  cholesky_nested_link = other.cholesky_nested_link;
411  mark_cholesky_nested_matrix_changed();
412  if (synchronization_direction() > 0) synchronize_reverse(); //< if synchronization_direction() is always 1
413  }
414  else // other.synchronization_direction() == 0
415  {
416  Base::nested_object() = other.nested_object();
417  mark_synchronized();
418  // Don't bother updating cholesky_nested() if synchronization_direction() is always 1:
419  if (synchronization_direction() <= 0) cholesky_nested_link = other.cholesky_nested_link;
420  }
421  }
422  return *this;
423  }

465  {
466  if constexpr(not (zero<nested_object_of_t<Arg>> and zero<NestedMatrix>) and
467  not (identity_matrix<nested_object_of_t<Arg>> and identity_matrix<NestedMatrix>))
468  {
469  if constexpr (triangular_matrix<nested_object_of_t<Arg>> == triangular_matrix<NestedMatrix> and
470  not diagonal_matrix<Arg>)
471  {
472  if (arg.synchronization_direction() > 0)
473  {
474  Base::nested_object() = to_covariance_nestable<NestedMatrix>(OpenKalman::nested_object(std::forward<Arg>(arg)));

476  }
477  else if (arg.synchronization_direction() < 0)
478  {
479  cholesky_nested_link = to_covariance_nestable<CholeskyNestedMatrix>(
480  std::forward<Arg>(arg).cholesky_nested_matrix());
481  mark_cholesky_nested_matrix_changed();
482  if (synchronization_direction() > 0) synchronize_reverse();
483  }
484  else // arg.synchronization_direction() == 0
485  {
486  Base::nested_object() = to_covariance_nestable<NestedMatrix>(OpenKalman::nested_object(std::forward<Arg>(arg)));
487  mark_synchronized();

489  {
490  cholesky_nested_link = to_covariance_nestable<CholeskyNestedMatrix>(
491  std::forward<Arg>(arg).cholesky_nested_matrix());
492  }
493  }
494  }
495  else
496  {
497  if (arg.synchronization_direction() < 0)
498  {
499  Base::nested_object() = to_covariance_nestable<NestedMatrix>(

506  mark_cholesky_nested_matrix_changed();
