33 TEST_F(eigen3, ToEuclideanExpr_class)
34 {
35  M4 m;
36  m << 1, 2, std::sqrt(3)/2, 0.5, 0.5, std::sqrt(3)/2, 3, 4;
37  To3 d1;
38  d1 << 1, 2, pi/6, pi/3, 3, 4;
39  EXPECT_TRUE(is_near(d1.nested_matrix(), mat3(1, 2, pi/6, pi/3, 3, 4)));
40  EXPECT_TRUE(is_near(d1, m));
41  ToFrom4 d1b;
42  d1b << 1, 2, std::sqrt(3)/2, 0.5, 0.5, std::sqrt(3)/2, 3, 4;
43  EXPECT_TRUE(is_near(d1b.nested_matrix(), mat3(1, 2, pi/6, pi/3, 3, 4)));
44  EXPECT_TRUE(is_near(d1b, mat4(1, 2, std::sqrt(3)/2, 0.5, 0.5, std::sqrt(3)/2, 3, 4)));
45  //
46  To3 d2 = (M3() << 1, 2, pi/6, pi/3, 3, 4).finished();
47  EXPECT_TRUE(is_near(d2, m));
48  To3 d3 = d2;
49  EXPECT_TRUE(is_near(d3, m));
50  To3 d4 = To3{1, 2, pi/6, pi/3, 3, 4};
51  EXPECT_TRUE(is_near(d4, m));
52  To3 d5 = MatrixTraits<M3>::zero();
53  EXPECT_TRUE(is_near(d5, mat4(0, 0, 1, 1, 0, 0, 0, 0)));
54  To3 d6 = ZeroMatrix<M3>();
55  EXPECT_TRUE(is_near(d6, mat4(0, 0, 1, 1, 0, 0, 0, 0)));
56  To3 d7 = To3(ZeroMatrix<M3>());
57  EXPECT_TRUE(is_near(d7, mat4(0, 0, 1, 1, 0, 0, 0, 0)));
58  To3 d8 {1, 2, pi/6, pi/3, 3, 4};
59  EXPECT_TRUE(is_near(d8, m));
60  EXPECT_TRUE(is_near(To3(ZeroMatrix<M3>()), mat4(0, 0, 1, 1, 0, 0, 0, 0)));
61  ToFrom4 d9 {1, 2, std::sqrt(3)/2, 0.5, 0.5, std::sqrt(3)/2, 3, 4};
62  EXPECT_TRUE(is_near(d9.nested_matrix(), mat3(1, 2, pi/6, pi/3, 3, 4)));
63  EXPECT_TRUE(is_near(d9, mat4(1, 2, std::sqrt(3)/2, 0.5, 0.5, std::sqrt(3)/2, 3, 4)));
64  //
65  d5 = d1;
66  EXPECT_TRUE(is_near(d5, m));
67  d6 = To3 {1, 2, pi/6, pi/3, 3, 4};
68  EXPECT_TRUE(is_near(d5, m));
69  d7 = m;
70  EXPECT_TRUE(is_near(d7, m));
71  d7 = M4::Zero();
72  d7 = {1, 2, pi/6, pi/3, 3, 4};
73  EXPECT_TRUE(is_near(d7, m));
74  d9 = M4::Zero();
75  d9 = {1, 2, std::sqrt(3)/2, 0.5, 0.5, std::sqrt(3)/2, 3, 4};
76  EXPECT_TRUE(is_near(d9, m));
77  //
78  d1 += d2;
79  EXPECT_TRUE(is_near(d1, mat4(2, 4, std::sqrt(3)/2, 0.5, 0.5, std::sqrt(3)/2, 6, 8)));
80  d1 -= To3 {1, 2, pi/6, pi/3, 3, 4};
81  EXPECT_TRUE(is_near(d1, mat4(1, 2, 1, 1, 0, 0, 3, 4)));
82  d1 *= 3;
83  EXPECT_TRUE(is_near(d1, mat4(3, 6, 1, 1, 0, 0, 9, 12)));
84  d1 /= 3;
85  EXPECT_TRUE(is_near(d1, mat4(1, 2, 1, 1, 0, 0, 3, 4)));
86  EXPECT_TRUE(is_near(d1.zero(), M4::Zero()));
87  EXPECT_TRUE(is_near(ToEuclideanExpr<Axes<2>, native_matrix_t<double, 2, 2>>::identity(), native_matrix_t<double, 2, 2>::Identity()));
88 }

115  EXPECT_NEAR((To3 {1, 2, pi/6, pi/3, 3, 4})(2, 0), 0.5, 1e-6);
116 }

132  mat4(1, 2, std::sqrt(3)/2, 0.5, 0.5, std::sqrt(3)/2, 3, 4)));

156  EXPECT_TRUE(is_near(LQ_decomposition(To2 {1, 2, 3, pi/6, pi/3, pi/4}),
157  LQ_decomposition(make_native_matrix<double, 3, 3>(

160  0.5, std::sqrt(3)/2, std::sqrt(2)/2))));
161  EXPECT_TRUE(is_near(QR_decomposition(To2 {1, 2, 3, pi/6, pi/3, pi/4}),
162  QR_decomposition(make_native_matrix<double, 3, 3>(
163  1, 2, 3,
164  std::sqrt(3)/2, 0.5, std::sqrt(2)/2,
165  0.5, std::sqrt(3)/2, std::sqrt(2)/2))));
166
