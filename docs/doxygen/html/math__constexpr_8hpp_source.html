545

552  auto b{detail::convert_to_floating(constexpr_imag(arg))};
553  using R = std::decay_t<decltype(a)>;
554  auto nx = constexpr_sqrt(a * a + b * b);
555  auto sqp = constexpr_sqrt(R{0.5} * (nx + a));
556  auto sqm = constexpr_sqrt(R{0.5} * (nx - a));
557  return make_complex_number<Scalar>(sqp, b >= R{0} ? sqm : -sqm);
558  }
559  else
560  {
561  auto&& x = detail::convert_to_output<Scalar>(std::forward<Arg>(arg));
562 
563  if (x <= Scalar{0})
564  {

574  Scalar previous{0};
575  while (next != previous)

577  previous = next;
578  next = Scalar{0.5} * (previous + std::forward<decltype(x)>(x) / previous);

664  constexpr T pi2 {numbers::pi_v<T> * T{2}};
665  constexpr T max {static_cast<T>(std::numeric_limits<std::intmax_t>::max())};
666  constexpr T lowest {static_cast<T>(std::numeric_limits<std::intmax_t>::lowest())};
667  if (theta > -pi2 and theta < pi2)
668  {
669  return theta;
670  }
671  else if (theta / pi2 >= lowest and theta / pi2 <= max)
672  {
673  return theta - static_cast<std::intmax_t>(theta / pi2) * pi2;
674  }
675  else if (theta > T{0})
676  {
677  T corr {pi2};
678  while ((theta - corr) / pi2 > max) corr *= T{2};
679  return scale_periodic_function(theta - corr);
680  }
681  else
682  {
683  T corr {pi2};
684  while ((theta + corr) / pi2 < lowest) corr *= T{2};
685  return scale_periodic_function(theta + corr);
686  }
