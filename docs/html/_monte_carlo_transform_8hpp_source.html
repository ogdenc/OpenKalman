21  /**
22  * \brief A Monte Carlo transform from one Gaussian distribution to another.
23  * Uses ideas from Chan, Tony F.; Golub, Gene H.; LeVeque, Randall J. (1979),
24  * "Updating Formulae and a Pairwise Algorithm for Computing Sample Variances."
25  * Technical Report STAN-CS-79-773, Department of Computer Science, Stanford University.
26  * http://i.stanford.edu/pub/cstr/reports/cs/tr/79/773/CS-TR-79-773.pdf
27  */

31  template<bool return_cross, typename TransformationType, typename InputDistribution, typename...NoiseDistributions>
32  struct MonteCarloSet
33  {
34  protected:
35  using In_Mean = typename DistributionTraits<InputDistribution>::Mean;
36  using Out_Mean = std::invoke_result_t<TransformationType, In_Mean>;
37  using InputCoefficients = typename MatrixTraits<In_Mean>::RowCoefficients;
38  using OutputCoefficients = typename MatrixTraits<Out_Mean>::RowCoefficients;
39  static_assert((equivalent_to<OutputCoefficients,
40  typename DistributionTraits<NoiseDistributions>::Coefficients> and ...));
41 
42  using InputMeanMatrix = native_matrix_t<
43  typename DistributionTraits<InputDistribution>::Mean, InputCoefficients::size, 1>;
44  using OutputEuclideanMeanMatrix = native_matrix_t<InputMeanMatrix, OutputCoefficients::size, 1>;
45  using OutputCovarianceMatrix = native_matrix_t<InputMeanMatrix, OutputCoefficients::size, OutputCoefficients::size>;
46  using OutputCovarianceSA = typename MatrixTraits<OutputCovarianceMatrix>::template SelfAdjointBaseType<>;
47  using CrossCovarianceMatrix = native_matrix_t<InputMeanMatrix, InputCoefficients::size, OutputCoefficients::size>;
48 
49  using InputMean = Mean<InputCoefficients, InputMeanMatrix>;
50  using OutputEuclideanMean = EuclideanMean<OutputCoefficients, OutputEuclideanMeanMatrix>;
51  using OutputCovariance = Covariance<OutputCoefficients, OutputCovarianceSA>;
52  using CrossCovariance = Matrix<InputCoefficients, OutputCoefficients, CrossCovarianceMatrix>;

74  static constexpr auto
75  zero()
76  {
77  if constexpr (return_cross)
78  return MonteCarloSum {
79  0, MatrixTraits<InputMean>::zero(), MatrixTraits<OutputEuclideanMean>::zero(),
80  MatrixTraits<OutputCovariance>::zero(), MatrixTraits<CrossCovariance>::zero()};
81  else
82  return MonteCarloSum {
83  0, MatrixTraits<InputMean>::zero(), MatrixTraits<OutputEuclideanMean>::zero(),
84  MatrixTraits<OutputCovariance>::zero()};
85  }

87  static auto
88  one(const TransformationType& trans, const InputDistribution& dist, const NoiseDistributions&...noise)
89  {
90  const auto x = dist();
91  const auto y = trans(x, noise()...);
92  if constexpr (return_cross)
93  return MonteCarloSum {1, x, to_Euclidean(y), OutputCovariance::zero(), CrossCovariance::zero()};
94  else
95  return MonteCarloSum {1, x, to_Euclidean(y), OutputCovariance::zero()};
96  }

98  struct MonteCarloBinaryOp
99  {
100  auto operator()(const MonteCarloSum& set1, const MonteCarloSum& set2)
101  {
102  using Scalar = typename MatrixTraits<decltype(set1.y_E)>::Scalar;
103  if (set2.count == 1) // This is most likely. Take advantage of prior knowledge of set2.
104  {
105  const auto count = set1.count + 1;
106  const Scalar s_count = count;
107  const auto x = (set1.count * set1.x + set2.x) / s_count;
108  const auto y_E = (set1.count * set1.y_E + set2.y_E) / s_count;
109  const auto delta = from_Euclidean(set2.y_E) - from_Euclidean(set1.y_E);
110  const auto delta_adj_factor = adjoint(delta) * set1.count / s_count;
111  const auto yy = set1.yy + delta * delta_adj_factor;
112  if constexpr (return_cross)
113  {
114  const auto xy = set1.xy + (set2.x - set1.x) * delta_adj_factor;
115  return MonteCarloSum {count, x, y_E, yy, xy};
116  }
117  else
118  {
119  return MonteCarloSum {count, x, y_E, yy};
120  }
121  }
122  else
123  {
124  const auto count = set1.count + set2.count;
125  const Scalar s_count = count;
126  const auto x = (set1.count * set1.x + set2.count * set2.x) / s_count;
127  const auto y_E = (set1.count * set1.y_E + set2.count * set2.y_E) / s_count;
128  const auto delta = from_Euclidean(set2.y_E) - from_Euclidean(set1.y_E);
129  const auto delta_adj_factor = adjoint(delta) * set1.count * set2.count / s_count;
130  const auto yy = set1.yy + set2.yy + delta * delta_adj_factor;
131  if constexpr (return_cross)
132  {
133  const auto xy = set1.xy + set2.xy + (set2.x - set1.x) * delta_adj_factor;
134  return MonteCarloSum {count, x, y_E, yy, xy};
135  }
136  else
137  {
138  return MonteCarloSum {count, x, y_E, yy};
139  }
140  }
141  }

188  template<typename InputDist, typename Trans, typename ... NoiseDists,
189  std::enable_if_t<(distribution<InputDist> and ... and distribution<NoiseDists>) and
190  std::is_invocable_v<Trans, typename DistributionTraits<InputDist>::Mean, typename DistributionTraits<NoiseDists>::Mean...>, int> = 0>
191  auto operator()(const InputDist& x, const Trans& transformation, const NoiseDists& ...n) const
192  {
193  using MSet = MonteCarloSet<false, Trans, InputDist, NoiseDists...>;
194  auto m_set = MSet(transformation, size, x, n...);
195  auto binary_op = typename MSet::MonteCarloBinaryOp();
196  using MSum = typename MSet::MonteCarloSum;
197 
198  MSum m_sum = std::reduce(std::execution::par_unseq, m_set.begin(), m_set.end(), MSet::zero(), binary_op);
199  auto mean_output = make_self_contained(from_Euclidean(m_sum.y_E));
200  auto out_covariance = make_self_contained(m_sum.yy / (size - 1.));
201  return GaussianDistribution {mean_output, out_covariance};
202  }

204  template<typename InputDist, typename Trans, typename ... NoiseDists,
205  std::enable_if_t<(distribution<InputDist> and ... and distribution<NoiseDists>) and
206  std::is_invocable_v<Trans, typename DistributionTraits<InputDist>::Mean, typename DistributionTraits<NoiseDists>::Mean...>, int> = 0>
207  auto transform_with_cross_covariance(const InputDist& x, const Trans& transformation, const NoiseDists& ...n) const
208  {
209  using MSet = MonteCarloSet<true, Trans, InputDist, NoiseDists...>;
210  auto m_set = MSet(transformation, size, x, n...);
211  auto binary_op = typename MSet::MonteCarloBinaryOp();
212  using MSum = typename MSet::MonteCarloSum;
213 
214  MSum m_sum = std::reduce(std::execution::par_unseq, m_set.begin(), m_set.end(), MSet::zero(), binary_op);
215  auto mean_output = make_self_contained(from_Euclidean(m_sum.y_E));
216  auto out_covariance = make_self_contained(m_sum.yy / (size - 1.));
217  auto cross_covariance = make_self_contained(m_sum.xy / (size - 1.));
218  auto out = GaussianDistribution {mean_output, out_covariance};
219  return std::tuple {std::move(out), std::move(cross_covariance)};
220  }
221 
222  protected:
223  const std::size_t size;
