23  template<typename T>
24  constexpr T sqrt_impl(T x, T lo, T hi)
25  {
26  if (lo == hi) return lo;
27  const T mid = (lo + hi + 1) / 2;
28  if (x / mid < mid) return sqrt_impl<T>(x, lo, mid - 1);
29  else return sqrt_impl(x, mid, hi);
30  }
31  }
32 
33  template<typename T>
34  constexpr T constexpr_sqrt(T x)
35  {
36  return detail::sqrt_impl<T>(x, 0, x / 2 + 1);
37  }
38  }

490  // Mean //
492 
494  template<typename Coefficients, typename BaseMatrix = Eigen::Matrix<double, Coefficients::size, 1>>
495  struct Mean;
496 
498  template<typename ... Args,
499  std::enable_if_t<(sizeof...(Args) > 0) and std::conjunction_v<std::is_arithmetic<Args>...>, int> = 0>
500  Mean(Args ...) -> Mean<Axes<sizeof...(Args)>,
501  Eigen::Matrix<std::decay_t<std::common_type_t<Args...>>, sizeof...(Args), 1>>;
502 
504  template<
505  typename Coefficients, typename ... Args,
506  std::enable_if_t<not std::is_arithmetic_v<Coefficients> and (sizeof...(Args) > 0) and
507  std::conjunction_v<std::is_arithmetic<Args>...>, int> = 0>
508  auto make_Mean(Args ... args)
509  {
510  using Scalar = std::common_type_t<Args...>;
511  constexpr std::size_t dim = Coefficients::size;
512  static_assert(sizeof...(Args) % dim == 0);
513  constexpr auto cols = sizeof...(Args) / dim;
514  using Mat = Eigen::Matrix<Scalar, dim, cols>;
515  return Mean<Coefficients, Mat>(MatrixTraits<Mat>::make(args...));
516  }
517 
519  template<typename ... Args,
520  std::enable_if_t<(sizeof...(Args) > 0) and std::conjunction_v<std::is_arithmetic<Args>...>, int> = 0>
521  auto make_Mean(Args ... args)
522  {
523  return make_Mean<OpenKalman::Axes<sizeof...(Args)>>(args...);
524  }
525 
527  template<
528  typename Scalar, typename Coefficients, std::size_t cols = 1,
529  std::enable_if_t<std::is_arithmetic_v<Scalar>, int> = 0,
530  std::enable_if_t<not std::is_arithmetic_v<Coefficients>, int> = 0>
531  auto make_Mean()
532  {
533  return Mean<Coefficients, Eigen::Matrix<Scalar, Coefficients::size, cols>>();
534  }
