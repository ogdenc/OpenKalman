202  template<typename V, typename ... Vs,
203  std::enable_if_t<
204  std::conjunction_v<is_ToEuclideanExpr<V>, is_ToEuclideanExpr<Vs>...> or
205  std::conjunction_v<is_FromEuclideanExpr<V>, is_FromEuclideanExpr<Vs>...>, int> = 0>
206  constexpr decltype(auto)
207  concatenate_vertical(V&& v, Vs&& ... vs) noexcept
208  {
209  if constexpr(sizeof...(Vs) > 0)
210  {
211  constexpr auto cols = MatrixTraits<V>::columns;
212  static_assert(((cols == MatrixTraits<Vs>::columns) and ...));
213  using C = Concatenate<typename MatrixTraits<V>::Coefficients, typename MatrixTraits<Vs>::Coefficients...>;
214  return MatrixTraits<V>::template make<C>(
215  concatenate_vertical(base_matrix(std::forward<V>(v)), base_matrix(std::forward<Vs>(vs))...));
216  }
217  else
218  {
219  return std::forward<V>(v);
220  }
221  }

225  template<typename V, typename ... Vs,
226  std::enable_if_t<
227  std::conjunction_v<is_ToEuclideanExpr<V>, is_ToEuclideanExpr<Vs>...> or
228  std::conjunction_v<is_FromEuclideanExpr<V>, is_FromEuclideanExpr<Vs>...>, int> = 0>
229  constexpr decltype(auto)
230  concatenate_horizontal(V&& v, Vs&& ... vs) noexcept
231  {
232  if constexpr(sizeof...(Vs) > 0)
233  {
234  using C = typename MatrixTraits<V>::Coefficients;
235  static_assert(std::conjunction_v<std::is_same<C, typename MatrixTraits<Vs>::Coefficients>...>);
236  return MatrixTraits<V>::template make<C>(
237  concatenate_horizontal(base_matrix(std::forward<V>(v)), base_matrix(std::forward<Vs>(vs))...));
238  }
239  else
240  {
241  return std::forward<V>(v);
242  }
243  }
