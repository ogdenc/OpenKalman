25  template<typename InclinationTraits, typename Scalar>
26  static constexpr std::tuple<Scalar, bool>
27  inclination_wrap_impl(const Scalar a)
28  {
29  constexpr Scalar period = InclinationTraits::template wrap_max<Scalar> - InclinationTraits::template wrap_min<Scalar>;
30  constexpr Scalar wrap_mod = 0.5 * period;
31  constexpr Scalar upward_angle = 0.25 * period;
32  if (a >= -upward_angle and a < upward_angle) // A shortcut, for the easy case.
33  {
34  return { a, false };
35  }
36  else
37  {
38  Scalar ar = std::fmod(a + upward_angle, period);
39  if (ar < 0) ar += period;
40  if (ar > wrap_mod)
41  {
42  // Do a mirror reflection about vertical axis.
43  return { 0.75 * period - ar, true };
44  }
45  else
46  {
47  return { ar - upward_angle, false };
48  }
49  }
50  }

52  template<typename CircleTraits, typename Scalar>
53  static inline Scalar azimuth_wrap_impl(const bool reflect_azimuth, const Scalar s)
54  {
55  constexpr Scalar wrap_max = CircleTraits::template wrap_max<Scalar>;
56  constexpr Scalar wrap_min = CircleTraits::template wrap_min<Scalar>;
57  constexpr Scalar period = wrap_max - wrap_min;
58 
59  Scalar a = reflect_azimuth ? s + period * 0.5 : s;
60 
61  if (a >= wrap_min and a < wrap_max) // Check if angle doesn't need wrapping.
62  {
63  return a;
64  }
65  else // Wrap the angle.
66  {
67  Scalar ar = std::fmod(a - wrap_min, period);
68  if (ar < 0)
69  {
70  ar += period;
71  }
72  return ar + wrap_min;
73  }
74  }

112  static constexpr Scalar cf_cir = 2 * M_PI / (CircleTraits::template wrap_max<Scalar> - CircleTraits::template wrap_min<Scalar>);
113 
114  template<std::size_t i, std::size_t d_i, std::size_t x_i, std::size_t y_i, std::size_t z_i>
115  static constexpr std::array<Scalar (*const)(const GetCoeff&), 1>
116  from_Euclidean_array = {[](const GetCoeff& get_coeff) constexpr {
117  const auto x = std::signbit(get_coeff(i + d_i)) ? -get_coeff(i + x_i) : get_coeff(i + x_i);
118  const auto y = std::signbit(get_coeff(i + d_i)) ? -get_coeff(i + y_i) : get_coeff(i + y_i);
119  return std::atan2(y, x) / cf_cir;
120  }};
121 
122  template<std::size_t i, std::size_t d_i, std::size_t a_i, std::size_t i_i>
123  static constexpr std::array<Scalar (*const)(const GetCoeff&), 1>
124  wrap_array_get = {[](const GetCoeff& get_coeff) {
125  const auto [new_i, b] = inclination_wrap_impl<InclinationTraits>(get_coeff(i + i_i));
126  const bool reflect_azimuth = b != std::signbit(get_coeff(i + d_i));
127  return azimuth_wrap_impl<CircleTraits>(reflect_azimuth, get_coeff(i + a_i));
128  }};

141  template<typename CircleTraits, typename InclinationTraits, typename Scalar>
142  struct SphericalImpl<Inclination<InclinationTraits>, CircleTraits, InclinationTraits, Scalar>
143  {
144  using GetCoeff = std::function<Scalar(const std::size_t)>;
145  using SetCoeff = std::function<void(const Scalar, const std::size_t)>;
146 
147  static constexpr Scalar cf_inc = 2 * M_PI / (InclinationTraits::template wrap_max<Scalar> - InclinationTraits::template wrap_min<Scalar>);
148 
149  template<std::size_t i, std::size_t d_i, std::size_t x_i, std::size_t y_i, std::size_t z_i>
150  static constexpr std::array<Scalar (*const)(const GetCoeff&), 1>
151  from_Euclidean_array = {[](const GetCoeff& get_coeff) constexpr {
152  const auto r = std::hypot(get_coeff(i + x_i), get_coeff(i + y_i), get_coeff(i + z_i));
153  const auto ret = std::asin(get_coeff(i + z_i) / r) / cf_inc;
154  if (std::isnan(ret)) return 0.; // Avoid NAN when all coefficients are zero.
155  else return std::signbit(get_coeff(i + d_i)) ? -ret : ret;
156  }};
157 
158  template<std::size_t i, std::size_t d_i, std::size_t a_i, std::size_t i_i>
159  static constexpr std::array<Scalar (*const)(const GetCoeff&), 1>
160  wrap_array_get = {[](const GetCoeff& get_coeff) constexpr {
161  const auto [new_i, b] = inclination_wrap_impl<InclinationTraits>(get_coeff(i + i_i));
162  return std::signbit(get_coeff(i + d_i)) ? -new_i : new_i;
163  }};
