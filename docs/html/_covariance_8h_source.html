129  template<typename Arg, std::enable_if_t<is_covariance_v<Arg> or is_typed_matrix_v<Arg>, int> = 0>
130  auto& operator=(Arg&& other) noexcept
131  {
132  if constexpr(is_covariance_v<Arg>)
133  {
134  static_assert(is_equivalent_v<typename MatrixTraits<Arg>::Coefficients, Coefficients>);
135  }
136  else if constexpr(is_typed_matrix_v<Arg>)
137  {
138  static_assert(is_equivalent_v<typename MatrixTraits<Arg>::RowCoefficients, Coefficients> and
139  is_equivalent_v<typename MatrixTraits<Arg>::RowCoefficients, Coefficients>);

143  {
144  static_assert(is_zero_v<Arg>);
145  }
146  else if constexpr (is_identity_v<BaseMatrix>)
147  {
148  static_assert(is_identity_v<Arg>);
149  }
150  else if constexpr(is_covariance_v<Arg> and
151  is_diagonal_v<Arg> and is_square_root_v<Arg> and is_diagonal_v<BaseMatrix>)
152  {

154  }
155  else
156  {
157  Base::operator=(internal::convert_base_matrix<std::decay_t<BaseMatrix>>(std::forward<Arg>(other)));
158  }

162  template<typename Arg, std::enable_if_t<is_covariance_v<Arg>, int> = 0>
163  auto& operator+=(Arg&& arg) noexcept
164  {
165  static_assert(is_equivalent_v<typename MatrixTraits<Arg>::Coefficients, Coefficients>);
166  static_assert(not is_square_root_v<Arg>);
167  if constexpr(is_self_adjoint_v<BaseMatrix>)
168  {
169  base_matrix() += internal::convert_base_matrix<BaseMatrix>(std::forward<Arg>(arg));
170  }
171  else
172  {
173  decltype(auto) E1 = base_matrix();
174  decltype(auto) E2 = internal::convert_base_matrix<BaseMatrix>(std::forward<Arg>(arg));
175  if constexpr(is_upper_triangular_v<BaseMatrix>)
176  base_matrix() = QR_decomposition(concatenate_vertical(E1, E2));
177  else
178  base_matrix() = LQ_decomposition(concatenate_horizontal(E1, E2));
179  }
180  this->mark_changed();
181  return *this;
182  }

213  template<typename S, std::enable_if_t<std::is_convertible_v<S, Scalar>, int> = 0>
214  auto& operator*=(const S s)
215  {
216  if constexpr(is_self_adjoint_v<BaseMatrix>)
217  {
218  base_matrix() *= s;
219  }
220  else
221  {
222  if (s > S(0))
223  {
224  base_matrix() *= std::sqrt(static_cast<Scalar>(s));
225  }
226  else if (s < S(0))
227  {
228  using TLowerType = typename MatrixTraits<BaseMatrix>::template TriangularBaseType<TriangleType::lower>;
229  const auto U = internal::convert_base_matrix<TLowerType>(*this);
230  base_matrix() = MatrixTraits<BaseMatrix>::zero();
231  rank_update(base_matrix(), U, s);
232  }
233  else
234  {
235  base_matrix() = MatrixTraits<BaseMatrix>::zero();
236  }
237  }
238  this->mark_changed();
239  return *this;
240  }

242  template<typename S, std::enable_if_t<std::is_convertible_v<S, Scalar>, int> = 0>
243  auto& operator/=(const S s)
244  {
245  if constexpr(is_self_adjoint_v<BaseMatrix>)
246  {
247  base_matrix() /= s;
248  }
249  else
250  {
251  if (s > S(0))

255  else if (s < S(0))
256  {
257  using TLowerType = typename MatrixTraits<BaseMatrix>::template TriangularBaseType<TriangleType::lower>;
258  const auto u = internal::convert_base_matrix<TLowerType>(*this);
259  base_matrix() = MatrixTraits<BaseMatrix>::zero();

261  }
262  else
263  {
264  throw (std::runtime_error("Covariance operator/=: divide by zero"));
265  }
266  }
267  this->mark_changed();
268  return *this;
269  }

349  std::enable_if_t<sizeof...(triangle_type) <= 1 and not is_covariance_base_v<Arg> and
350  is_typed_matrix_base_v<Arg>, int> = 0>
351  inline auto
352  make_Covariance(Arg&& arg) noexcept
353  {
354  static_assert(MatrixTraits<Arg>::dimension == MatrixTraits<Arg>::columns);
355  constexpr TriangleType arg_t_type = triangle_type_of_v<typename MatrixTraits<Arg>::template TriangularBaseType<>>;
356  constexpr TriangleType t_type = std::get<0>(std::tuple {triangle_type..., arg_t_type});
357  using T = typename MatrixTraits<Arg>::template TriangularBaseType<t_type>;
358  using SA = typename MatrixTraits<Arg>::template SelfAdjointBaseType<t_type>;
359  using B = std::conditional_t<sizeof...(triangle_type) == 1, T, SA>;
