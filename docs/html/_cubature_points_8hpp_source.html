18  /*************CubaturePoints************
19  * \brief Cubature points, as implemented in
20  * I. Arasaratnam & S. Haykin, Cubature Kalman Filters, IEEE Transactions on Automatic Control,
21  * vol. 54, pp. 1254-1269, 2009.
22  * This class has only static members, and is not to be instantiated.
23  */
24  struct CubaturePoints
25  {
26  protected:
27  CubaturePoints() {} // Disallow instantiation.
28 
29  private:
30  /// Scale and translate normalized sample points based on mean and (square root) covariance.
31  /// The function steps recursively through a tuple of input and noise distributions.
32  template<
33  std::size_t dim, ///< The total number of dimensions for which sigma points are assigned.
34  std::size_t pos = 0, ///< The writing position during this recursive step.
35  typename D, ///< First input or noise distribution.
36  typename...Ds> ///< Other input or noise distributions.
37  static auto
38  sample_points_impl(const D& d, const Ds&...ds)
39  {
40  using Scalar = typename DistributionTraits<D>::Scalar;
41  using Coeffs = typename DistributionTraits<D>::Coefficients;
42  using M = typename DistributionTraits<D>::Mean;
43  constexpr auto points_count = dim * 2;
44  constexpr auto dim_i = DistributionTraits<D>::dimension;
45  constexpr auto frame_size = dim_i * 2;
46  const auto delta = make_Matrix<Coeffs, Axes<dim_i>>(make_native_matrix(square_root(OpenKalman::covariance_of(d) * static_cast<Scalar>(dim))));
47  if constexpr(frame_size == points_count)
48  {
49  auto ret = concatenate_horizontal(delta, -delta);
50  static_assert(MatrixTraits<decltype(ret)>::columns == points_count);
51  return std::tuple {std::move(ret)};
52  }
53  else if constexpr (pos == 0)
54  {
55  constexpr auto width = points_count - frame_size;
56  using MRbase = native_matrix_t<M, dim_i, width>;
57  const auto mright = Matrix<Coeffs, Axes<width>, MRbase>::zero();
58  auto ret = concatenate_horizontal(delta, -delta, mright);
59  static_assert(MatrixTraits<decltype(ret)>::columns == points_count);
60  return std::tuple_cat(std::tuple {std::move(ret)}, sample_points_impl<dim, frame_size>(ds...));
61  }
62  else if constexpr (pos + frame_size < points_count)
63  {
64  using MLbase = native_matrix_t<M, dim_i, pos>;
65  const auto mleft = Matrix<Coeffs, Axes<pos>, MLbase>::zero();
66  constexpr auto width = points_count - (pos + frame_size);
67  using MRbase = native_matrix_t<M, dim_i, width>;
68  const auto mright = Matrix<Coeffs, Axes<width>, MRbase>::zero();
69  auto ret = concatenate_horizontal(mleft, delta, -delta, mright);
70  static_assert(MatrixTraits<decltype(ret)>::columns == points_count);
71  return std::tuple_cat(std::tuple {std::move(ret)}, sample_points_impl<dim, pos + frame_size>(ds...));
72  }
73  else
74  {
75  static_assert(sizeof...(ds) == 0);
76  using MLbase = native_matrix_t<M, dim_i, pos>;
77  const auto mleft = Matrix<Coeffs, Axes<pos>, MLbase>::zero();
78  auto ret = concatenate_horizontal(mleft, delta, -delta);
79  static_assert(MatrixTraits<decltype(ret)>::columns == points_count);
80  return std::tuple {std::move(ret)};
81  }
82  }
83 
84  public:
85  /**
86  * \brief Scale and translate normalized sample points based on mean and (square root) covariance.
87  * \return A matrix of sigma points (each sigma point in a column).
88  */
89  template<typename...Dist>
90  static auto
91  sample_points(const Dist&...ds)
92  {
93  static_assert((gaussian_distribution<Dist> and ...));
94  constexpr auto dim = (DistributionTraits<Dist>::dimension + ...);
95  return sample_points_impl<dim>(ds...);
96  }

98 #ifdef __cpp_concepts
99  template<std::size_t dim, euclidean_mean Arg>
100 #else
101  template<std::size_t dim, typename Arg, std::enable_if_t<euclidean_mean<Arg>, int> = 0>
102 #endif
103  static auto
104  weighted_means(const Arg& y_means)
105  {
106  static_assert(column_vector<Arg>);
107  static_assert(MatrixTraits<Arg>::columns == dim * 2, "Wrong number of cubature points.");
108  return reduce_columns(y_means);
109  };
110 
111  template<std::size_t dim, typename InputDist, bool return_cross = false, typename X, typename Y>
112  static auto
113  covariance(const X& x_deviations, const Y& y_deviations)
114  {
115  static_assert(typed_matrix<X> and typed_matrix<Y>);
116  using Scalar = typename MatrixTraits<X>::Scalar;
117  constexpr auto count = MatrixTraits<X>::columns;
118  static_assert(count == MatrixTraits<Y>::columns);
119  static_assert(count == dim * 2, "Wrong number of cubature points.");
120  constexpr auto inv_weight = 1 / Scalar(count);
121  if constexpr(cholesky_form<InputDist>)
122  {
123  auto out_covariance = Covariance {LQ_decomposition(y_deviations * std::sqrt(inv_weight))};
124  if constexpr (return_cross)
125  {
126  auto cross_covariance = make_self_contained(x_deviations * inv_weight * adjoint(y_deviations));
127  return std::tuple{std::move(out_covariance), std::move(cross_covariance)};
128  }
129  else
130  {
131  return out_covariance;
132  }
133  }
134  else
135  {
136  const auto w_yT = make_self_contained(inv_weight * adjoint(y_deviations));
137  auto out_covariance = Covariance {make_self_contained(y_deviations * w_yT)};
138  if constexpr (return_cross)
139  {
140  auto cross_covariance = make_self_contained(x_deviations * w_yT);
141  return std::tuple{std::move(out_covariance), std::move(cross_covariance)};
142  }
143  else
144  {
145  return out_covariance;
146  }
147  }
148  }
149 
150  };
