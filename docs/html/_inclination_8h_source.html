32  template<typename Scalar>
33  using GetCoeff = std::function<Scalar(const std::size_t)>;

38  template<typename Scalar>
39  static constexpr Scalar cf = 2 * M_PI / (Traits::template wrap_max<Scalar> - Traits::template wrap_min<Scalar>);
40 
41  template<typename Scalar, std::size_t i>
42  static constexpr std::array<Scalar (*const)(const GetCoeff<Scalar>&), dimension>
43  to_Euclidean_array =
44  {
45  [](const GetCoeff<Scalar>& get_coeff) { return std::cos(get_coeff(i)) * cf<Scalar>; },
46  [](const GetCoeff<Scalar>& get_coeff) { return std::sin(get_coeff(i)) * cf<Scalar>; }
47  };
48 
49  template<typename Scalar, std::size_t i>
50  static constexpr std::array<Scalar (*const)(const GetCoeff<Scalar>&), size>
51  from_Euclidean_array =
52  {
53  [](const GetCoeff<Scalar>& get_coeff)
54  {
55  return std::atan2(get_coeff(i + 1), std::abs(get_coeff(i))) / cf<Scalar>;
56  }
57  };
58 
59  protected:
60  template<typename Scalar>
61  static Scalar wrap_impl(const Scalar s)
62  {
63  constexpr Scalar inclination_max = M_PI / 2;
64  constexpr Scalar wrap_mod = inclination_max * 2;
65  Scalar a = std::fmod(s + inclination_max, wrap_mod * 2);
66  if (a < 0)
67  {
68  a += wrap_mod * 2;
69  }
70  if (a > wrap_mod)
71  {
72  a = wrap_mod * 2 - a;
73  }
74  return a - inclination_max;
75  }
76 
77  public:
78  template<typename Scalar, std::size_t i>
79  static constexpr std::array<Scalar (*const)(const GetCoeff<Scalar>&), size>
80  wrap_array_get =
81  {
82  [](const GetCoeff<Scalar>& get_coeff) { return wrap_impl(get_coeff(i)); }
83  };

87  wrap_array_set =
88  {
89  [](const Scalar s, const SetCoeff<Scalar>& set_coeff, const GetCoeff<Scalar>&) { set_coeff(wrap_impl(s), i); }
