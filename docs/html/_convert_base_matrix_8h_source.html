24  template<typename T, typename Arg>
25  constexpr decltype(auto)
26  convert_base_matrix(Arg&& arg) noexcept
27  {
28  static_assert(is_covariance_v<Arg> or is_typed_matrix_v<Arg>);
29  using ArgBase = typename MatrixTraits<Arg>::BaseMatrix;
30 
31  // Typed matrices:
32  if constexpr(is_typed_matrix_v<Arg>)
33  {
34  static_assert(is_equivalent_v<typename MatrixTraits<Arg>::RowCoefficients, typename MatrixTraits<Arg>::ColumnCoefficients>);
35  using SA = typename MatrixTraits<ArgBase>::template SelfAdjointBaseType<>;
36  if constexpr(std::is_void_v<T>)
37  return std::forward<Arg>(arg).base_matrix();
38  else
39  return MatrixTraits<T>::make(base_matrix(MatrixTraits<SA>::make(std::forward<Arg>(arg).base_matrix())));
40  }
41 
42  // Natural conversion:
43  else if constexpr(std::is_void_v<T>)
44  {
45  return std::forward<Arg>(arg).get_apparent_base_matrix();
46  }
47 
48  // strictly triangular or diagonal square root --> self-adjoint
49  else if constexpr(((is_triangular_v<ArgBase> and not is_diagonal_v<ArgBase>) or
50  (is_square_root_v<Arg> and is_diagonal_v<ArgBase>)) and is_self_adjoint_v<T> and not is_diagonal_v<T>)
51  {
52  if constexpr((is_self_adjoint_v<ArgBase> and not is_square_root_v<Arg>) or
53  (is_triangular_v<ArgBase> and is_square_root_v<Arg>))
54  return Cholesky_square(std::forward<Arg>(arg).base_matrix());
55  else
56  return std::forward<Arg>(arg).get_apparent_base_matrix();
57  }
58 
59  // strictly self-adjoint or diagonal non-square root --> strictly triangular
60  else if constexpr(((not is_diagonal_v<ArgBase> and not is_triangular_v<ArgBase>) or
61  (is_diagonal_v<ArgBase> and not is_square_root_v<Arg> )) and is_triangular_v<T> and not is_diagonal_v<T>)
62  {
63  if constexpr(is_diagonal_v<ArgBase>) // diagonal non-square root
64  {
65  return Cholesky_factor(std::forward<Arg>(arg).base_matrix());
66  }
67  else // ArgBase is strictly self-adjoint.
68  {
69  using B = decltype(Cholesky_factor(std::forward<Arg>(arg).base_matrix()));
70  if constexpr(is_upper_triangular_v<T> != is_upper_triangular_v<B>) // Converted triangle types don't match.
71  {
72  return Cholesky_factor(adjoint(std::forward<Arg>(arg).base_matrix()));
73  }
74  else // Converted triangle types match.
75  {
76  if constexpr((is_self_adjoint_v<ArgBase> and not is_square_root_v<Arg>) or
77  (is_triangular_v<ArgBase> and is_square_root_v<Arg>))
78  return Cholesky_factor(std::forward<Arg>(arg).base_matrix());
79  else
80  {
81  return std::forward<Arg>(arg).get_apparent_base_matrix();
82  }
83  }
84  }
85  }
86 
87  // upper triangular <--> lower triangular
88  else if constexpr(is_Cholesky_v<Arg> and is_triangular_v<T>
89  and is_upper_triangular_v<ArgBase> != is_upper_triangular_v<T>)
90  {
91  return adjoint(std::forward<Arg>(arg).base_matrix());
92  }
93 
94  // pass through
95  else
96  {
97  return std::forward<Arg>(arg).base_matrix();
98  }
99  }
