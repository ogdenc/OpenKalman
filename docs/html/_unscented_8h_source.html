51  template<typename Parameters = UnscentedParametersStateEstimation>
52  struct Unscented : internal::ScaledSigmaPointsBase<Unscented<Parameters>, Parameters>
53  {
59  template<std::size_t dim>
60  static constexpr std::size_t
61  sigma_point_count() { return dim * 2 + 1; };
62 
63  private:
67  template<
68  std::size_t dim,
69  std::size_t pos = 0,
70  typename D,
71  typename...Ds>
72  static auto
73  sigma_points_impl(const D& d, const Ds&...ds)
74  {
75  using Scalar = typename DistributionTraits<D>::Scalar;
76  using Coeffs = typename DistributionTraits<D>::Coefficients;
77  using M = typename DistributionTraits<D>::Mean;
78  constexpr auto points_count = sigma_point_count<dim>();
79  constexpr auto dim_i = DistributionTraits<D>::dimension;
80  constexpr auto frame_size = dim_i * 2;
81  constexpr Scalar alpha = Parameters::alpha;
82  constexpr Scalar kappa = Parameters::template kappa<dim>;
83  constexpr Scalar gamma_L = alpha * alpha * (kappa + dim);
84  const auto delta = make_Matrix<Coeffs, Axes<dim_i>>(strict_matrix(square_root(gamma_L * covariance(d))));
85  //
86  using M0base = strict_matrix_t<M, dim_i, 1>;
87  const auto m0 = TypedMatrix<Coeffs, Axis, M0base>::zero();
88  if constexpr(1 + frame_size == points_count)
89  {
90  auto ret = concatenate_horizontal(m0, delta, -delta);
91  static_assert(MatrixTraits<decltype(ret)>::columns == points_count);
92  return std::tuple {std::move(ret)};
93  }
94  else if constexpr (pos == 0)
95  {
96  constexpr auto width = points_count - (1 + frame_size);
97  using MRbase = strict_matrix_t<M, dim_i, width>;
98  const auto mright = TypedMatrix<Coeffs, Axes<width>, MRbase>::zero();
99  auto ret = concatenate_horizontal(m0, delta, -delta, mright);
100  static_assert(MatrixTraits<decltype(ret)>::columns == points_count);
101  return std::tuple_cat(std::tuple {std::move(ret)}, sigma_points_impl<dim, 1 + frame_size>(ds...));
102  }
103  else if constexpr (pos + frame_size < points_count)
104  {
105  using MLbase = strict_matrix_t<M, dim_i, pos>;
106  const auto mleft = TypedMatrix<Coeffs, Axes<pos>, MLbase>::zero();
107  constexpr auto width = points_count - (pos + frame_size);
108  using MRbase = strict_matrix_t<M, dim_i, width>;
109  const auto mright = TypedMatrix<Coeffs, Axes<width>, MRbase>::zero();
110  auto ret = concatenate_horizontal(mleft, delta, -delta, mright);
111  static_assert(MatrixTraits<decltype(ret)>::columns == points_count);
112  return std::tuple_cat(std::tuple {std::move(ret)}, sigma_points_impl<dim, pos + frame_size>(ds...));
113  }
114  else
115  {
116  static_assert(sizeof...(ds) == 0);
117  using MLbase = strict_matrix_t<M, dim_i, pos>;
118  const auto mleft = TypedMatrix<Coeffs, Axes<pos>, MLbase>::zero();
119  auto ret = concatenate_horizontal(mleft, delta, -delta);
120  static_assert(MatrixTraits<decltype(ret)>::columns == points_count);
121  return std::tuple {std::move(ret)};
122  }
123  }
124 
125  public:
132  template<typename...Dist>
133  static constexpr auto
134  sigma_points(const Dist& ...ds)
135  {
136  static_assert(std::conjunction_v<is_Gaussian_distribution<Dist>...>);
137  constexpr auto dim = (DistributionTraits<Dist>::dimension + ...);
138  return sigma_points_impl<dim>(ds...);
139  }
140 
141  protected:
142  Unscented() {}; // Prevent instantiation.
143 
144  friend struct internal::ScaledSigmaPointsBase<Unscented<Parameters>, Parameters>;
145 
146  template<std::size_t dim, typename Scalar = double>
147  static constexpr auto
148  unscaled_W0()
149  {
150  constexpr Scalar kappa = Parameters::template kappa<dim>;
151  return kappa / (dim + kappa);
152  }
153 
154  template<std::size_t dim, typename Scalar = double>
155  static constexpr auto
156  unscaled_W()
157  {
158  constexpr Scalar kappa = Parameters::template kappa<dim>;
159  return 0.5 / (dim + kappa);
160  }
161 
162  };
