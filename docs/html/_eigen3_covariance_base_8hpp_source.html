22  /**
23  * Ultimate base of Covariance and SquareRootCovariance classes, general case.
24  * No conversion is necessary if either
25  * (1) Derived is not a square root and the base is self-adjoint; or
26  * (2) Derived is a square root and the base is triangular.
27  */
28 #ifdef __cpp_concepts
29  template<typename Derived, typename Nested> requires
30  (self_adjoint_matrix<Nested> and not square_root_covariance<Derived>) or
31  (triangular_matrix<Nested> and square_root_covariance<Derived>)
32  struct Eigen3CovarianceBase<Derived, Nested>
33 #else
34  template<typename Derived, typename Nested>
35  struct Eigen3CovarianceBase<Derived, Nested, std::enable_if_t<
36  (self_adjoint_matrix<Nested> and not square_root_covariance<Derived>) or
37  (triangular_matrix<Nested> and square_root_covariance<Derived>)>>
38 #endif
39  : Eigen3MatrixBase<Derived, Nested> {};
40 
41 
42  /**
43  * Ultimate base of Covariance and SquareRootCovariance classes, if Derived is not a square root and
44  * the base is not self-adjoint (i.e., it is triangular but not diagonal).
45  */
46 #ifdef __cpp_concepts
47  template<typename Derived, typename ArgType> requires
48  (not self_adjoint_matrix<ArgType>) and (not square_root_covariance<Derived>)
49  struct Eigen3CovarianceBase<Derived, ArgType>
50 #else
51  template<typename Derived, typename ArgType>
52  struct Eigen3CovarianceBase<Derived, ArgType, std::enable_if_t<
53  not self_adjoint_matrix<ArgType> and not square_root_covariance<Derived>>>
54 #endif
55  : Eigen3MatrixBase<Derived, ArgType>
56  {
57  using Nested = std::decay_t<ArgType>;
58  using Scalar = typename Nested::Scalar;
59  using Base = Eigen3MatrixBase<Derived, Nested>;
60 
61 #ifdef __cpp_concepts
62  template<std::convertible_to<Scalar> S>
63 #else
64  template<typename S, std::enable_if_t<std::is_convertible_v<S, Scalar>, int> = 0>
65 #endif
66  constexpr auto operator<<(const S& s)
67  {
68  auto& xpr = static_cast<Derived&>(*this);
69  return Eigen::CovarianceCommaInitializer(xpr, static_cast<const Scalar&>(s));
70  }
71 
72  template<typename OtherDerived>
73  constexpr auto operator<<(const Eigen::DenseBase<OtherDerived>& other)
74  {
75  auto& xpr = static_cast<Derived&>(*this);
76  return Eigen::CovarianceCommaInitializer(xpr, other);
77  }
78  };

122 namespace Eigen
123 {
124  /**
125  * Alternative version of CommaInitializer for Covariance and SquareRootCovariance.
126  */
127  template<typename CovarianceType>
128  struct CovarianceCommaInitializer
129  {
130  using Scalar = typename CovarianceType::Scalar;
131  using BaseMatrix = OpenKalman::native_matrix_t<typename OpenKalman::MatrixTraits<CovarianceType>::BaseMatrix>;
132  using Nested = CommaInitializer<BaseMatrix>;
133 
134  BaseMatrix matrix;
135  Nested comma_initializer;
136  CovarianceType& cov;
137 
138  template<typename S, std::enable_if_t<std::is_convertible_v<S, Scalar>, int> = 0>
139  CovarianceCommaInitializer(CovarianceType& xpr, const S& s)
140  : matrix(), comma_initializer(matrix, static_cast<const Scalar&>(s)), cov(xpr) {}
141 
142  template<typename OtherDerived>
143  CovarianceCommaInitializer(CovarianceType& xpr, const DenseBase <OtherDerived>& other)
144  : matrix(), comma_initializer(matrix, other), cov(xpr) {}
145 
146  CovarianceCommaInitializer(const CovarianceCommaInitializer& o)
147  : matrix(o.matrix), comma_initializer(o.comma_initializer), cov(o.cov) {}
148 
149  CovarianceCommaInitializer(CovarianceCommaInitializer&& o)
150  : matrix(std::move(o.matrix)),
151  comma_initializer(std::move(o.comma_initializer)), cov(std::move(o.cov)) {}
152 
153  template<typename S, std::enable_if_t<std::is_convertible_v<S, Scalar>, int> = 0>
154  auto& operator,(const S& s)
155  {
156  comma_initializer, static_cast<const Scalar&>(s);
157  return *this;
158  }
159 
160  template<typename OtherDerived>
161  auto& operator,(const DenseBase<OtherDerived>& other)
162  {
163  comma_initializer, other;
164  return *this;
165  }
166 
167  ~CovarianceCommaInitializer()
168  {
169  finished();
170  }
171 
172  auto& finished()
173  {
174  cov = comma_initializer.finished();
175  return cov;
176  }
177  };
