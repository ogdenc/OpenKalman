106  template<typename Arg, std::enable_if_t<is_typed_matrix_v<Arg>, int> = 0>
107  auto& operator=(Arg&& other) noexcept
108  {
109  static_assert(is_equivalent_v<typename MatrixTraits<Arg>::RowCoefficients, Coefficients>);
110  static_assert(MatrixTraits<Arg>::ColumnCoefficients::axes_only);
111  if constexpr (is_zero_v<BaseMatrix>)
112  {
113  static_assert(is_zero_v<Arg>);
114  }
115  else if constexpr (is_identity_v<BaseMatrix>)
116  {
117  static_assert(is_identity_v<Arg>);
118  }
119  else if constexpr(is_mean_v<Arg>)
120  {
121  this->base_matrix() = std::forward<Arg>(other).base_matrix();
122  }
123  else if constexpr(is_Euclidean_transformed_v<Arg>)
124  {
125  this->base_matrix() = OpenKalman::from_Euclidean<Coefficients>(std::forward<Arg>(other).base_matrix());
126  }
127  else
128  {
129  this->base_matrix() = OpenKalman::wrap_angles<Coefficients>(std::forward<Arg>(other).base_matrix());
130  }
131  return *this;
132  }
