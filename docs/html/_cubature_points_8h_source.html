32  template<
33  std::size_t dim,
34  std::size_t pos = 0,
35  typename D,
36  typename...Ds>
37  static auto
38  sample_points_impl(const D& d, const Ds&...ds)
39  {
40  using Scalar = typename DistributionTraits<D>::Scalar;
41  using Coeffs = typename DistributionTraits<D>::Coefficients;
42  using M = typename DistributionTraits<D>::Mean;
43  constexpr auto points_count = dim * 2;
44  constexpr auto dim_i = DistributionTraits<D>::dimension;
45  constexpr auto frame_size = dim_i * 2;
46  const auto delta = make_Matrix<Coeffs, Axes<dim_i>>(strict_matrix(square_root(OpenKalman::covariance(d) * static_cast<Scalar>(dim))));
47  if constexpr(frame_size == points_count)
48  {
49  auto ret = concatenate_horizontal(delta, -delta);
50  static_assert(MatrixTraits<decltype(ret)>::columns == points_count);
51  return std::tuple {std::move(ret)};
52  }
53  else if constexpr (pos == 0)
54  {
55  constexpr auto width = points_count - frame_size;
56  using MRbase = strict_matrix_t<M, dim_i, width>;
57  const auto mright = TypedMatrix<Coeffs, Axes<width>, MRbase>::zero();
58  auto ret = concatenate_horizontal(delta, -delta, mright);
59  static_assert(MatrixTraits<decltype(ret)>::columns == points_count);
60  return std::tuple_cat(std::tuple {std::move(ret)}, sample_points_impl<dim, frame_size>(ds...));
61  }
62  else if constexpr (pos + frame_size < points_count)
63  {
64  using MLbase = strict_matrix_t<M, dim_i, pos>;
65  const auto mleft = TypedMatrix<Coeffs, Axes<pos>, MLbase>::zero();
66  constexpr auto width = points_count - (pos + frame_size);
67  using MRbase = strict_matrix_t<M, dim_i, width>;
68  const auto mright = TypedMatrix<Coeffs, Axes<width>, MRbase>::zero();
69  auto ret = concatenate_horizontal(mleft, delta, -delta, mright);
70  static_assert(MatrixTraits<decltype(ret)>::columns == points_count);
71  return std::tuple_cat(std::tuple {std::move(ret)}, sample_points_impl<dim, pos + frame_size>(ds...));
72  }
73  else
74  {
75  static_assert(sizeof...(ds) == 0);
76  using MLbase = strict_matrix_t<M, dim_i, pos>;
77  const auto mleft = TypedMatrix<Coeffs, Axes<pos>, MLbase>::zero();
78  auto ret = concatenate_horizontal(mleft, delta, -delta);
79  static_assert(MatrixTraits<decltype(ret)>::columns == points_count);
80  return std::tuple {std::move(ret)};
81  }
82  }

98  template<std::size_t dim, typename Arg, std::enable_if_t<is_Euclidean_mean_v<Arg>, int> = 0>
99  static auto
100  weighted_means(const Arg& y_means)

104  return reduce_columns(y_means);

107  template<std::size_t dim, typename InputDist, bool return_cross = false, typename X, typename Y>
108  static auto
109  covariance(const X& x_deviations, const Y& y_deviations)
110  {
111  static_assert(is_typed_matrix_v<X> and is_typed_matrix_v<Y>);
112  using Scalar = typename MatrixTraits<X>::Scalar;
113  constexpr auto count = MatrixTraits<X>::columns;
114  static_assert(count == MatrixTraits<Y>::columns);
115  static_assert(count == dim * 2, "Wrong number of cubature points.");
116  constexpr auto inv_weight = 1 / Scalar(count);
117  if constexpr(is_Cholesky_v<InputDist>)
118  {
119  auto out_covariance = Covariance {LQ_decomposition(y_deviations * std::sqrt(inv_weight))};
120  if constexpr (return_cross)
121  {
122  auto cross_covariance = strict(x_deviations * inv_weight * adjoint(y_deviations));
123  return std::tuple{std::move(out_covariance), std::move(cross_covariance)};
124  }
125  else
126  {
127  return out_covariance;
128  }
129  }
130  else
131  {
132  const auto w_yT = strict(inv_weight * adjoint(y_deviations));
133  auto out_covariance = Covariance {strict(y_deviations * w_yT)};
134  if constexpr (return_cross)
135  {
136  auto cross_covariance = strict(x_deviations * w_yT);
137  return std::tuple{std::move(out_covariance), std::move(cross_covariance)};
138  }
139  else
140  {
141  return out_covariance;
142  }
143  }
144  }
