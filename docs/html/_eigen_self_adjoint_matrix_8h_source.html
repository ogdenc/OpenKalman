274  template<typename B, std::enable_if_t<is_Eigen_matrix_v<B>, int> = 0>
275  auto solve(const B& b) const
276  {
277  using M = Eigen::Matrix<Scalar, dimension, MatrixTraits<B>::columns>;
278  auto llt = this->base_view().llt();
279  M ret = llt.solve(b);
280  if (llt.info() != Eigen::Success) //[[unlikely]] // C++20
281  {
282  // A is semidefinite. Use LDLT decomposition instead.
283  auto ldlt = this->base_view().ldlt();
284  M ret2 = ldlt.solve(b);
285  if ((not ldlt.isPositive() and not ldlt.isNegative()) or ldlt.info() != Eigen::Success)
286  {
287  throw (std::runtime_error("EigenSelfAdjointMatrix solve: A is indefinite"));
288  }
289  return ret2;
290  }
291  return ret;
292  }
