77 #ifdef __cpp_concepts
78  template<coefficients C, coefficients ... Ctail>
79 #else
80  template<typename C, typename ... Ctail>
81 #endif
82  struct Coefficients<C, Ctail ...>
83  {
84 #ifndef __cpp_concepts
85  static_assert((coefficients<C> and ... and coefficients<Ctail>));
86 #endif
87  static constexpr std::size_t size = C::size + Coefficients<Ctail...>::size; ///<Aggregate number of coefficients.
88 
89  /// Aggregate number of coefficients when converted to Euclidian.
90  static constexpr std::size_t dimension = C::dimension + Coefficients<Ctail...>::dimension;
91 
92  /// Whether all the coefficients are of type Axis.
93  static constexpr bool axes_only = C::axes_only and Coefficients<Ctail...>::axes_only;
94 
95  /**
96  * \brief The type of the result when subtracting two Coefficients vectors.
97  * \details Each coefficient is subtracted independently.
98  */
99  using difference_type = Concatenate<typename C::difference_type, typename Ctail::difference_type...>;
100 
101  private:
102  template<typename Scalar>
103  using GetCoeff = std::function<Scalar(const std::size_t)>;
104 
105  template<typename Scalar>
106  using SetCoeff = std::function<void(const Scalar, const std::size_t)>;
107 
108  template<typename Scalar>
109  using GetCoeffFunction = Scalar(*)(const GetCoeff<Scalar>&);
110 
111  template<typename Scalar>
112  using SetCoeffFunction = void(*)(const Scalar, const SetCoeff<Scalar>&, const GetCoeff<Scalar>&);
113 
114  public:
115  /**
116  * \internal
117  * \brief An array of functions that convert the coefficients to coordinates in Euclidean space.
118  * \details The functions in the array take the coefficients and convert them to
119  * Cartesian coordinates in a Euclidean space, depending on the type of each coordinate.
120  * Each array element is a function taking a ''get coefficient'' function and returning a coordinate value.
121  * The ''get coefficient'' function takes the index of a column within a row vector and returns the coefficient.
122  * \tparam Scalar The scalar type (e.g., double).
123  * \tparam i The index of the first spherical coefficient that is being transformed.
124  * This may be non-zero if this set of coefficients is part of a larger set of composite coordinates.
125  */
126  template<typename Scalar, std::size_t i>
127  static constexpr std::array<const GetCoeffFunction<Scalar>, dimension>
128  to_Euclidean_array = internal::join(C::template to_Euclidean_array<Scalar, i>,
129  Coefficients<Ctail...>::template to_Euclidean_array<Scalar, i + C::size>);
130 
131 
132  /**
133  * \internal
134  * \brief An array of functions that convert coordinates in Euclidean space into the typed coordinates.
135  * \details The functions in the array take Cartesian coordinates, and convert them to the typed coordinates.
136  * The array element is a function taking a ''get coefficient'' function and returning the typed coordinates.
137  * The ''get coefficient'' function takes the index of a column within a row vector and returns one of
138  * the Cartesian coordinates.
139  * \tparam Scalar The scalar type (e.g., double).
140  * \tparam i The index of the first of the Cartesian coordinates being transformed back to their respective types.
141  * This may be non-zero if this set of coefficients is part of a larger set of composite coordinates.
142  */
143  template<typename Scalar, std::size_t i>
144  static constexpr std::array<const GetCoeffFunction<Scalar>, size>
145  from_Euclidean_array = internal::join(C::template from_Euclidean_array<Scalar, i>,
146  Coefficients<Ctail...>::template from_Euclidean_array<Scalar, i + C::dimension>);
147 
148 
149  /**
150  * \internal
151  * \brief An array of functions that return a wrapped version of the coefficients.
152  * \details Each function in the array takes a ''get coefficient'' function and returns wrapped coefficients.
153  * The ''get coefficient'' function takes the index of a column within a row vector and returns a coefficient.
154  * \tparam Scalar The scalar type (e.g., double).
155  * \tparam i The index of the first of the coefficients that are being wrapped.
156  * This may be non-zero if this set of coefficients is part of a larger set of composite coordinates.
157  */
158  template<typename Scalar, std::size_t i>
159  static constexpr std::array<const GetCoeffFunction<Scalar>, size>
160  wrap_array_get = internal::join(C::template wrap_array_get<Scalar, i>,
161  Coefficients<Ctail...>::template wrap_array_get<Scalar, i + C::size>);
162 
163 
164  /**
165  * \internal
166  * \brief An array of functions that wraps and sets an existing matrix coefficient.
167  * \details Each void function in the array takes a scalar value and ''set coefficient'' function.
168  * The ''set coefficient'' function takes a scalar value and an index of a column within a row vector and
169  * sets the coefficient at that index to a wrapped version of the scalar input.
170  * \tparam Scalar The scalar type (e.g., double).
171  * \tparam i The index of the first of the coefficients that are being wrapped.
172  * This may be non-zero if this set of coefficients is part of a larger set of composite coordinates.
173  */
174  template<typename Scalar, std::size_t i>
175  static constexpr std::array<const SetCoeffFunction<Scalar>, size>
176  wrap_array_set = internal::join(C::template wrap_array_set<Scalar, i>,
177  Coefficients<Ctail...>::template wrap_array_set<Scalar, i + C::size>);
178 
179  /**
180  * \brief Prepend a set of new coefficients to the existing set.
181  * \tparam Cnew The set of new coordinates to prepend.
182  */
183  template<typename ... Cnew>
184  using Prepend = Coefficients<Cnew..., C, Ctail ...>;
185 
186 
187  /**
188  * \brief Append a set of new coordinates to the existing set.
189  * \tparam Cnew The set of new coordinates to append.
190  */
191  template<typename ... Cnew>
192  using Append = Coefficients<C, Ctail ..., Cnew ...>;
193 
194 
195  /**
196  * \brief Extract a particular coefficient from the set of coefficients.
197  * \tparam i The index of the coefficient.
198  */
199  template<std::size_t i>
200  using Coefficient = std::conditional_t<i == 0, C, typename Coefficients<Ctail...>::template Coefficient<i - 1>>;
201 
202 
203  /**
204  * \brief Take the first <code>count</code> coefficients.
205  * \tparam count The number of coefficients to take.
206  */
207  template<std::size_t count>
208  using Take = std::conditional_t<count == 0,
209  Coefficients<>,
210  typename Coefficients<Ctail...>::template Take<count - 1>::template Prepend<C>>;
211 
212 
213  /**
214  * \brief Discard all remaining coefficients after the first <code>count</code>.
215  * \tparam count The index of the first coefficient to discard.
216  */
217  template<std::size_t count>
218  using Discard = std::conditional_t<count == 0,
219  Coefficients,
220  typename Coefficients<Ctail...>::template Discard<count - 1>>;
221 
222 
223  static_assert(internal::coefficient_class<Coefficients>);
224  };
