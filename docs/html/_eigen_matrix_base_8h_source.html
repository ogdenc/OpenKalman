25  /*
26  * Base class for all OpenKalman classes that are also Eigen3 matrices.
27  */
28  template<typename Derived, typename ArgType>
29  struct EigenMatrixBase : Eigen::MatrixBase<Derived>
30  {
31  using Nested = std::decay_t<ArgType>;
32  using Scalar = typename Nested::Scalar;
33 
34  protected:
35  template<typename Arg>
36  constexpr decltype(auto) get_ultimate_base_matrix_impl(Arg&& arg) noexcept
37  {
38  decltype(auto) b = base_matrix(std::forward<Arg>(arg));
39  using B = decltype(b);
40  if constexpr(
41  is_EigenSelfAdjointMatrix_v<B> or
42  is_EigenTriangularMatrix_v<B> or
43  is_EigenDiagonal_v<B> or
44  is_FromEuclideanExpr_v<B> or
45  is_ToEuclideanExpr_v<B>)
46  {
47  return get_ultimate_base_matrix(b);
48  }
49  else
50  {
51  return b;
52  }
53  }
54 
55  template<typename Arg>
56  constexpr decltype(auto) get_ultimate_base_matrix(Arg&& arg) noexcept
57  {
58  if constexpr(is_EigenSelfAdjointMatrix_v<Arg>)
59  {
60  if constexpr (MatrixTraits<Arg>::storage_type == TriangleType::diagonal) return std::forward<Arg>(arg);
61  else return get_ultimate_base_matrix_impl(std::forward<Arg>(arg));
62  }
63  else if constexpr(is_EigenTriangularMatrix_v<Arg>)
64  {
65  if constexpr(MatrixTraits<Arg>::triangle_type == TriangleType::diagonal) return std::forward<Arg>(arg);
66  else return get_ultimate_base_matrix_impl(std::forward<Arg>(arg));
67  }
68  else
69  {
70  return get_ultimate_base_matrix_impl(std::forward<Arg>(arg));
71  }
72  }
73 
74  public:
75 
76  static constexpr Eigen::Index rows() { return Eigen::internal::traits<Derived>::RowsAtCompileTime; }
77 
78  static constexpr Eigen::Index cols() { return Eigen::internal::traits<Derived>::ColsAtCompileTime; }
79 
80  template<typename S, std::enable_if_t<std::is_convertible_v<S, Scalar>, int> = 0>
81  constexpr auto operator<<(const S& s)
82  {
83  auto& xpr = get_ultimate_base_matrix(static_cast<Derived&>(*this));
84  using Xpr = std::decay_t<decltype(xpr)>;
85  if constexpr(is_mean_v<Derived>)
86  {
87  return Eigen::MeanCommaInitializer<Derived, Xpr>(xpr, static_cast<const Scalar&>(s));
88  }
89  else if constexpr(is_EigenSelfAdjointMatrix_v<Xpr> or is_EigenTriangularMatrix_v<Xpr>)
90  {
91  return Eigen::DiagonalCommaInitializer(xpr, static_cast<const Scalar&>(s));
92  }
93  else
94  {
95  return Eigen::CommaInitializer(xpr, static_cast<const Scalar&>(s));
96  }
97  }
98 
99  template<typename OtherDerived>
100  constexpr auto operator<<(const Eigen::DenseBase<OtherDerived>& other)
101  {
102  auto& xpr = get_ultimate_base_matrix(static_cast<Derived&>(*this));
103  using Xpr = std::decay_t<decltype(xpr)>;
104  if constexpr(is_mean_v<Derived>)
105  {
106  return Eigen::MeanCommaInitializer<Derived, Xpr>(xpr, static_cast<const OtherDerived&>(other));
107  }
108  else if constexpr(is_EigenSelfAdjointMatrix_v<Xpr> or is_EigenTriangularMatrix_v<Xpr>)
109  {
110  std::cout << "a3" << std::endl << std::flush;
111  return Eigen::DiagonalCommaInitializer(xpr, static_cast<const OtherDerived&>(other));
112  }
113  else
114  {
115  std::cout << "a4" << std::endl << std::flush;
116  return Eigen::CommaInitializer(xpr, static_cast<const OtherDerived&>(other));
117  }
118  }
119 
121  static decltype(auto) Zero() { return Derived::zero(); }
122 
124  static decltype(auto) Identity() { return Derived::identity(); }
125  };

135  struct MeanCommaInitializer : CommaInitializer<XprType>
136  {
137  using Base = CommaInitializer<XprType>;
138  using Scalar = typename XprType::Scalar;
139  using Coefficients = typename OpenKalman::MatrixTraits<Derived>::RowCoefficients;
140  using Base::Base;
141 
142  template<typename S, std::enable_if_t<std::is_convertible_v<S, Scalar>, int> = 0>
143  MeanCommaInitializer(XprType& xpr, const S& s) : Base(xpr, static_cast<const Scalar&>(s)) {}
144 
145  template<typename OtherDerived>
146  MeanCommaInitializer(XprType& xpr, const DenseBase <OtherDerived>& other) : Base(xpr, other) {}
147 
148  ~MeanCommaInitializer()

151  }
152

163  template<typename XprType>
164  struct DiagonalCommaInitializer
165  {
166  using Scalar = typename XprType::Scalar;
167  static constexpr auto dim = OpenKalman::MatrixTraits<XprType>::dimension;
168  using BaseMatrix = OpenKalman::strict_matrix_t<typename OpenKalman::MatrixTraits<XprType>::BaseMatrix, dim, 1>;
169  using Nested = CommaInitializer<BaseMatrix>;
170 
171  BaseMatrix matrix;
172  Nested comma_initializer;
173  XprType& diag;
174 
175  template<typename S, std::enable_if_t<std::is_convertible_v<S, Scalar>, int> = 0>
176  DiagonalCommaInitializer(XprType& xpr, const S& s)
177  : matrix(), comma_initializer(matrix, static_cast<const Scalar&>(s)), diag(xpr) {}
178 
179  template<typename OtherDerived>
180  DiagonalCommaInitializer(XprType& xpr, const DenseBase <OtherDerived>& other)
181  : matrix(), comma_initializer(matrix, other), diag(xpr) {}
182 
183  DiagonalCommaInitializer(const DiagonalCommaInitializer& o)
184  : matrix(o.matrix), comma_initializer(o.comma_initializer), diag(o.diag) {}
185 
186  DiagonalCommaInitializer(DiagonalCommaInitializer&& o)
187  : matrix(std::move(o.matrix)),
188  comma_initializer(std::move(o.comma_initializer)), diag(std::move(o.diag)) {}
189 
190  template<typename S, std::enable_if_t<std::is_convertible_v<S, Scalar>, int> = 0>
191  auto& operator,(const S& s)
192  {
193  comma_initializer, static_cast<const Scalar&>(s);
194  return *this;
195  }
196 
197  template<typename OtherDerived>
198  auto& operator,(const DenseBase<OtherDerived>& other)
199  {
200  comma_initializer, other;
201  return *this;
202  }
203 
204  ~DiagonalCommaInitializer()
205  {
206  finished();
207  }
208 
209  auto& finished()
210  {
211  diag = OpenKalman::EigenDiagonal<BaseMatrix>(comma_initializer.finished());
212  return diag;
213  }
