23  /**
24  * \brief A matrix with typed rows and columns.
25  * \details It is a wrapper for a native matrix type from a supported matrix library such as Eigen.
26  * The matrix can be thought of as a transformation from X to Y, where the coefficients for each of X and Y are typed.
27  * Example declarations:
28  * - <code>Matrix<Coefficients<Axis, Axis, angle::Radians>, Coefficients<Axis, Axis>, native_matrix_t<native_matrix_t<double, 3, 2>> x;</code>
29  * - <code>Matrix<double, Coefficients<Axis, Axis, angle::Radians>, Coefficients<Axis, Axis>,
30  * native_matrix_t<double, 3, 2>> x;</code>
31  * \tparam RowCoefficients A set of coefficients (e.g., Axis, Spherical, etc.) corresponding to the rows.
32  * \tparam ColumnCoefficients Another set of coefficients corresponding to the columns.
33  * \tparam NestedMatrix The underlying native matrix or matrix expression.
34  */
35 #ifdef __cpp_concepts
36  template<coefficients RowCoefficients, coefficients ColumnCoefficients, typed_matrix_nestable NestedMatrix> requires
37  (RowCoefficients::size == MatrixTraits<NestedMatrix>::dimension) and
38  (ColumnCoefficients::size == MatrixTraits<NestedMatrix>::columns) and (not std::is_rvalue_reference_v<NestedMatrix>)
39 #else
40  template<typename RowCoefficients, typename ColumnCoefficients, typename NestedMatrix>
41 #endif
42  struct Matrix;

45  /**
46  * \brief A set of one or more column vectors, each representing a statistical mean.
47  * \details Unlike OpenKalman::Matrix, the columns of a Mean are untyped. When a Mean is converted to an
48  * OpenKalman::Matrix, the columns are assigned type Axis.
49  * Example declaration:
50  * <code>Mean<Coefficients<Axis, Axis, angle::Radians>, 1, native_matrix_t<double, 3, 1>> x;</code>
51  * This declares a 3-dimensional vector <var>x</var>, where the coefficients are, respectively, an Axis,
52  * an Axis, and an angle::Radians, all of scalar type <code>double</code>. The underlying representation is an
53  * Eigen3 column vector.
54  * \tparam Coefficients Coefficient types of the mean (e.g., Axis, Polar).
55  * \tparam NestedMatrix The underlying native matrix or matrix expression.
56  */
57 #ifdef __cpp_concepts
58  template<coefficients Coefficients, typed_matrix_nestable NestedMatrix> requires
59  (Coefficients::size == MatrixTraits<NestedMatrix>::dimension) and (not std::is_rvalue_reference_v<NestedMatrix>)
60 #else
61  template<typename Coefficients, typename NestedMatrix>
62 #endif
63  struct Mean;

66  /**
67  * \brief Similar to a Mean, but the coefficients are transformed into Euclidean space, based on their type.
68  * \details Means containing angles should be converted to EuclideanMean before taking an average or weighted average.
69  * Example declaration:
70  * <code>EuclideanMean<Coefficients<Axis, Axis, angle::Radians>, 1, native_matrix_t<double, 4, 1>> x;</code>
71  * This declares a 3-dimensional mean <var>x</var>, where the coefficients are, respectively, an Axis,
72  * an Axis, and an angle::Radians, all of scalar type <code>double</code>. The underlying representation is a
73  * four-dimensional vector in Euclidean space, with the last two of the dimensions representing the angle::Radians coefficient
74  * transformed to x and y locations on a unit circle associated with the angle::Radians-type coefficient.
75  * \tparam Coefficients A set of coefficients (e.g., Axis, angle::Radians, Polar, etc.)
76  * \tparam NestedMatrix The underlying native matrix or matrix expression.
77  */
78 #ifdef __cpp_concepts
79  template<coefficients Coefficients, typed_matrix_nestable NestedMatrix> requires
80  (Coefficients::dimension == MatrixTraits<NestedMatrix>::dimension) and (not std::is_rvalue_reference_v<NestedMatrix>)
81 #else
82  template<typename Coefficients, typename NestedMatrix>

87  /**
88  * \brief A self-adjoint Covariance matrix.
89  * \details The coefficient types for the rows are the same as for the columns.
90  * \tparam Coefficients Coefficient types.
91  * \tparam NestedMatrix The underlying native matrix or matrix expression. It can be either self-adjoint or
92  * (either upper or lower) triangular. If it is triangular, the native matrix will be multiplied by its transpose
93  * when converted to a Matrix or when used in mathematical expressions. The self-adjoint and triangular versions
94  * are functionally identical, but often the triangular version is more efficient.
95  */
96 #ifdef __cpp_concepts

98  (Coefficients::size == MatrixTraits<NestedMatrix>::dimension) and (not std::is_rvalue_reference_v<NestedMatrix>)
99 #else
100  template<typename Coefficients, typename NestedMatrix>
101 #endif
102  struct Covariance;

105  /**
106  * \brief The upper or lower triangle Cholesky factor (square root) of a covariance matrix.
107  * \details If S is a SquareRootCovariance, S*transpose(S) is a Covariance.
108  * If NestedMatrix is triangular, the SquareRootCovariance has the same triangle type (upper or lower). If NestedMatrix
109  * is self-adjoint, the triangle type of SquareRootCovariance is considered either upper ''or'' lower.
110  * \tparam Coefficients Coefficient types.
111  * \tparam NestedMatrix The underlying native matrix or matrix expression. It can be either self-adjoint or
112  * (either upper or lower) triangular. If it is self-adjoint, the native matrix will be Cholesky-factored
113  * when converted to a Matrix or when used in mathematical expressions. The self-adjoint and triangular versions
114  * are functionally identical, but often the triangular version is more efficient.
115  */
116 #ifdef __cpp_concepts
117  template<coefficients Coefficients, covariance_nestable NestedMatrix> requires
118  (Coefficients::size == MatrixTraits<NestedMatrix>::dimension) and (not std::is_rvalue_reference_v<NestedMatrix>)
119 #else
120  template<typename Coefficients, typename NestedMatrix>
