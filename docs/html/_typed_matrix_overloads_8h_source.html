159  template<typename Arg, std::enable_if_t<is_typed_matrix_v<Arg>, int> = 0>
160  constexpr decltype(auto)
161  reduce_columns(Arg&& arg) noexcept
162  {
163  if constexpr(MatrixTraits<Arg>::columns == 1)
164  {
165  return std::forward<Arg>(arg);
166  }
167  else
168  {
169  static_assert(is_column_vector_v<Arg>);
170  if constexpr(is_Euclidean_transformed_v<Arg>)
171  {
172  return MatrixTraits<Arg>::make(reduce_columns(base_matrix(std::forward<Arg>(arg))));
173  }
174  else if constexpr(is_mean_v<Arg>)
175  {
176  using C = typename MatrixTraits<Arg>::RowCoefficients;
177  auto ev = reduce_columns(base_matrix(to_Euclidean(std::forward<Arg>(arg))));
178  return MatrixTraits<Arg>::make(from_Euclidean<C>(std::move(ev)));
179  }
180  else
181  {
182  using C = typename MatrixTraits<Arg>::RowCoefficients;
183  return make_Matrix<C, Axis>(reduce_columns(base_matrix(std::forward<Arg>(arg))));
184  }
