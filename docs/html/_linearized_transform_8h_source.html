32  template<typename MeanType, typename Hessian, typename Cov, std::size_t...ints>
33  static auto make_mean(const Hessian& hessian, const Cov& P, std::index_sequence<ints...>)
34  {
35  return MeanType {0.5 * trace(P * hessian[ints])...};
36  }

39  static constexpr auto make_cov_row(const Hessian& hessian, const Cov& P)
40  {
41  return std::tuple {0.5 * trace(P * hessian[i] * P * hessian[js])...};
42  }

44  template<typename CovType, typename Hessian, typename Cov, std::size_t...is, std::size_t...js>

46  {
47  auto mat = std::tuple_cat(make_cov_row<is, js...>(hessian, P)...);
48  return std::make_from_tuple<CovType>(std::move(mat));

61  template<typename OutputCoeffs, typename Hessian, typename Dist>
62  static auto second_order_term(const Hessian& hessian, const Dist& x)
63  {
64  constexpr auto output_dim = std::tuple_size_v<Hessian>;
65  static_assert(OutputCoeffs::size == output_dim);
66  //
67  // Convert input distribution type to output distribution types, and initialize mean and covariance:
68  using CovIn = typename MatrixTraits<typename DistributionTraits<Dist>::Covariance>::BaseMatrix;
69  using MeanOut = strict_matrix_t<CovIn, output_dim, 1>;
70  using CovOut = typename MatrixTraits<CovIn>::template SelfAdjointBaseType<triangle_type_of_v<CovIn>, output_dim>;
71 
72  const auto P = covariance(x);
73  const std::make_index_sequence<output_dim> ints;
74  auto mean_terms = make_mean<Mean<OutputCoeffs, MeanOut>>(hessian, P, ints);
75  auto cov_terms = make_cov<Covariance<OutputCoeffs, CovOut>>(hessian, P, ints, ints);
76  return GaussianDistribution(mean_terms, cov_terms);
77  }

79  template<typename OutputCoeffs, typename T1, typename T2, std::size_t...I>
80  static auto zip_tuples_impl(const T1& t1, const T2& t2, std::index_sequence<I...>)
81  {
82  return strict((second_order_term<OutputCoeffs>(std::get<I>(t1), std::get<I>(t2)) + ...));
83  }
