146  template<typename OutputCoefficients, typename In>
147  inline auto zero_hessian_impl()
148  {
149  using InputCoefficients = typename MatrixTraits<In>::RowCoefficients;
150  constexpr std::size_t input_size = InputCoefficients::size;
151  constexpr std::size_t output_size = OutputCoefficients::size;
152  using HessianMatrixInBase = strict_matrix_t<In, input_size, input_size>;
153  using HessianMatrixIn = TypedMatrix<InputCoefficients, InputCoefficients, HessianMatrixInBase>;
154  using HessianArrayIn = std::array<HessianMatrixIn, output_size>;
155 
156  HessianArrayIn a;
157  a.fill(HessianMatrixIn::zero());
158  return a;
159  }
160  }
