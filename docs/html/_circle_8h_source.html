40  template<typename Scalar, std::size_t i>
41  static constexpr std::array<Scalar (*const)(const GetCoeff<Scalar>&), dimension>
42  to_Euclidean_array =
43  {
44  [](const GetCoeff<Scalar>& get_coeff) { return std::cos(get_coeff(i) * cf<Scalar>); },
45  [](const GetCoeff<Scalar>& get_coeff) { return std::sin(get_coeff(i) * cf<Scalar>); }
46  };
47 
48  template<typename Scalar, std::size_t i>
49  static constexpr std::array<Scalar (*const)(const GetCoeff<Scalar>&), size>
50  from_Euclidean_array =
51  {
52  [](const GetCoeff<Scalar>& get_coeff) {
53  constexpr Scalar wrap_max = Traits::template wrap_max<Scalar>;
54  constexpr Scalar wrap_min = Traits::template wrap_min<Scalar>;
55  constexpr Scalar period = wrap_max - wrap_min;
56  auto a = std::atan2(get_coeff(i + 1), get_coeff(i)) / cf<Scalar>;
57  if constexpr (wrap_min != - M_PI / cf<Scalar>)
58  {
59  if (a < wrap_min) return a + period; // Generally, this is for positive angle systems where wrap_min is 0.
60  }
61  return a;
62  }
63  };
64 
65  protected:
66  template<typename Scalar>
67  static Scalar wrap_impl(const Scalar a)
68  {
69  constexpr Scalar wrap_max = Traits::template wrap_max<Scalar>;
70  constexpr Scalar wrap_min = Traits::template wrap_min<Scalar>;
71  constexpr Scalar period = wrap_max - wrap_min;
72  if (a >= wrap_min and a < wrap_max)
73  {
74  return a;
75  }
76  else
77  {
78  Scalar ar = std::fmod(a - wrap_min, period);
79  if (ar < 0) ar += period;
80  return ar + wrap_min;
81  }
82  }
