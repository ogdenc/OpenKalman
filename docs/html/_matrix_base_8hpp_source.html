14 namespace OpenKalman::internal
15 {
16  template<typename Derived, typename ArgType>
17  struct MatrixBase : MatrixTraits<ArgType>::template MatrixBaseType<Derived>
18  {
19  using BaseMatrix = ArgType;
20  static constexpr auto columns = MatrixTraits<ArgType>::columns;
21 
22  /// Default constructor.
23  MatrixBase() : m_arg() {}
24 
25  /// Copy constructor.
26  MatrixBase(const MatrixBase& other) : m_arg(other.base_matrix()) {}
27 
28  /// Move constructor.
29  MatrixBase(MatrixBase&& other) noexcept : m_arg(std::move(other).base_matrix()) {}
30 
31  /// Forwarding constructor.
32  template<typename Arg>
33  MatrixBase(Arg&& arg) noexcept : m_arg(std::forward<Arg>(arg)) {}
34 
35  /// Copy assignment operator.
36  auto& operator=(const MatrixBase& other)
37  {
38  if constexpr (not zero_matrix<BaseMatrix> and not identity_matrix<BaseMatrix>) if (this != &other)
39  m_arg = other.base_matrix();
40  return *this;
41  }
42 
43  /// Move assignment operator.
44  auto& operator=(MatrixBase&& other) noexcept
45  {
46  if constexpr (not zero_matrix<BaseMatrix> and not identity_matrix<BaseMatrix>) if (this != &other)
47  m_arg = std::move(other).base_matrix();
48  return *this;
49  }
50 
51  /// Assign from a compatible matrix or covariance base.
52 #ifdef __cpp_concepts
53  template<typename Arg> requires typed_matrix_base<Arg> or covariance_base<Arg>
54 #else
55  template<typename Arg, std::enable_if_t<typed_matrix_base<Arg> or covariance_base<Arg>, int> = 0>
56 #endif
57  auto& operator=(Arg&& arg) noexcept
58  {
59  if constexpr (zero_matrix<BaseMatrix>)
60  {
61  static_assert(zero_matrix<Arg>);
62  }
63  else if constexpr (identity_matrix<BaseMatrix>)
64  {
65  static_assert(identity_matrix<Arg>);
66  }
67  else
68  {
69  m_arg = std::forward<Arg>(arg);
70  }
71  return *this;
72  }
73 
74  /// Get the base matrix.
75  constexpr auto& base_matrix() & { return m_arg; }
76 
77  /// Get the base matrix.
78  constexpr auto&& base_matrix() && { return std::move(m_arg); }
79 
80  /// Get the base matrix.
81  constexpr const auto& base_matrix() const & { return m_arg; }
82 
83  /// Get the base matrix.
84  constexpr const auto&& base_matrix() const && { return std::move(m_arg); }
85 
86  private:
87  BaseMatrix m_arg; ///< Where the base matrix is stored.
88 
89  };
