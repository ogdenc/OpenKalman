210  template<typename Arg, std::enable_if_t<is_typed_matrix_v<Arg>, int> = 0>
211  inline auto make_EuclideanMean(Arg&& arg) noexcept
212  {
213  static_assert(is_column_vector_v<Arg>);
214  using C = typename MatrixTraits<Arg>::RowCoefficients;
215  if constexpr(is_Euclidean_transformed_v<Arg>)
216  return make_EuclideanMean<typename MatrixTraits<Arg>::RowCoefficients>(base_matrix(std::forward<Arg>(arg)));
217  else
218  return make_EuclideanMean<typename MatrixTraits<Arg>::RowCoefficients>(base_matrix(to_Euclidean<C>(std::forward<Arg>(arg))));

252  using Scalar = typename MatrixTraits<BaseMatrix>::Scalar;
253  static_assert(RowCoefficients::dimension == dimension);

256  using StrictMatrix = typename MatrixTraits<BaseMatrix>::template StrictMatrix<rows, cols, S>;
257 
258  using Strict = EuclideanMean<RowCoefficients, typename MatrixTraits<BaseMatrix>::Strict>;

261  template<typename RC = RowCoefficients, typename CC = void, typename Arg,
262  std::enable_if_t<is_typed_matrix_base_v<Arg>, int> = 0>
263  static auto make(Arg&& arg) noexcept
264  {
265  static_assert(MatrixTraits<Arg>::dimension == RC::dimension);
266  if constexpr(not std::is_void_v<CC>) static_assert(is_equivalent_v<CC, Axes<MatrixTraits<Arg>::columns>>);
267  return EuclideanMean<RC, std::decay_t<Arg>>(std::forward<Arg>(arg));
268  }
