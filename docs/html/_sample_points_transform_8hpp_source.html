19  /**
20  * \brief Scaled points transform. Compatible with unscented transform and cubature transform.
21  * As implemented in, e.g.,
22  * E. Wan & R. van der Merwe, "The unscented Kalman filter for nonlinear estimation,"
23  * in Proc. of IEEE Symposium (AS-SPCC), pp. 153-158.
24  * See also R. van der Merwe & E. Wan, "The Square-Root Unscented Kalman Filter for State and
25  * Parameter-Estimation in Proc. Acoustics, Speech, and Signal Processing (ICASSP'01), 2001, pp. 3461-64.
26  */
27  template<
28  /// The type of sample points on which the transform is based.
29  typename SamplePointsType>
30  struct SamplePointsTransform;
31 
32 
33  using CubatureTransform = SamplePointsTransform<CubaturePoints>;
34 
35  using UnscentedTransform = SamplePointsTransform<UnscentedSigmaPoints>;
36  using UnscentedTransformParameterEstimation = SamplePointsTransform<UnscentedSigmaPointsParameterEstimation>;

38 
39  template<typename SamplePointsType>
40  struct SamplePointsTransform : TransformBase<SamplePointsTransform<SamplePointsType>>
41  {
42  private:
43  template<typename Transformation, typename...Points, typename...Dists, std::size_t...ints>
44  static constexpr auto y_means_impl(
45  const Transformation& g,
46  const std::tuple<Points...>& points,
47  const std::tuple<Dists...>& dists,
48  std::index_sequence<ints...>)
49  {
50  constexpr auto count = MatrixTraits<decltype(std::get<0>(points))>::columns;
51  return apply_columnwise<count>([&](size_t i) {
52  return g((column(std::get<ints>(points), i) + mean_of(std::get<ints>(dists)))...);
53  });
54  }
55 
56  template<std::size_t pos = 0, typename FlattenedPs>
57  static constexpr auto construct_ps(const FlattenedPs&)
58  {
59  static_assert(pos == std::tuple_size_v<FlattenedPs>);
60  return std::tuple {};
61  }
62 
63  template<std::size_t pos = 0, typename FlattenedPs, typename D, typename...Ds>
64  static auto construct_ps(const FlattenedPs& flattened_ps, const D&, const Ds&...ds)
65  {
66  constexpr auto group_size = std::tuple_size_v<D>;
67  auto ps_group = internal::tuple_slice<pos, pos + group_size>(flattened_ps);
68  return std::tuple_cat(std::make_tuple(ps_group), construct_ps<pos + group_size>(flattened_ps, ds...));
69  }
70 
71  template<typename D, typename...Ds>
72  static constexpr auto count_dim(const std::tuple<Ds...>&)
73  {
74  return (DistributionTraits<D>::dimension + ... + DistributionTraits<Ds>::dimension);
75  }
76 
77  protected:
78  template<std::size_t dim, typename InputDist, std::size_t i, bool return_cross = false,
79  typename...Gs, typename P, typename...Ps, typename D, typename...Ds>
80  static constexpr auto transform_impl(
81  const std::tuple<Gs...>& gs,
82  const P& xpoints,
83  const std::tuple<Ps...>& ps,
84  const D& x,
85  const std::tuple<Ds...>& ds)
86  {
87  static_assert(sizeof...(Gs) == sizeof...(Ps) and sizeof...(Gs) == sizeof...(Ds));
88  auto g = std::get<i>(gs);
89  auto xpoints_tup = std::tuple_cat(std::tuple {xpoints}, std::get<i>(ps));
90  auto xdists_tup = std::tuple_cat(std::tuple {x}, std::get<i>(ds));
91 
92  constexpr std::size_t N = std::tuple_size_v<decltype(xpoints_tup)>;
93  static_assert(N == std::tuple_size_v<decltype(xdists_tup)>);
94 
95  auto y_means = Mean {y_means_impl(g, xpoints_tup, xdists_tup, std::make_index_sequence<N>())};
96  auto y_mean = from_Euclidean(SamplePointsType::template weighted_means<dim>(to_Euclidean(y_means)));
97  // Each column is a deviation from y mean for each transformed sigma point:
98  auto ypoints = apply_columnwise(y_means, [&](const auto& col) { return make_self_contained(col - y_mean); });
99 
100  if constexpr (i + 1 < sizeof...(Gs))
101  {
102  auto y_covariance = SamplePointsType::template covariance<dim, InputDist, false>(xpoints, ypoints);
103  auto y = GaussianDistribution {make_self_contained(std::move(y_mean)), std::move(y_covariance)};
104  return transform_impl<dim, InputDist, i + 1, return_cross>(gs, ypoints, ps, y, ds);
105  }
106  else
107  {
108  auto y_covariance = SamplePointsType::template covariance<dim, InputDist, return_cross>(xpoints, ypoints);
109  if constexpr (return_cross)
110  {
111  auto [y_cov, cross] = y_covariance;
112  return std::tuple {GaussianDistribution {make_self_contained(std::move(y_mean)), std::move(y_cov)}, std::move(cross)};
113  }
114  else
115  {
116  return GaussianDistribution {std::move(y_mean), std::move(y_covariance)};
117  }
118  }
119  }
120 
121  /**
122  * Perform one or more consecutive sample points transforms.
123  * \tparam InputDist Input distribution.
124  * \tparam Ts A list of tuples containing (1) a transformation and (2) zero or more noise terms for that transformation.
125  **/
126  template<bool return_cross, typename InputDist, typename...Ts>
127  auto transform(const InputDist& x, const Ts&...ts) const
128  {
129  auto gs = std::tuple {std::get<0>(ts)...};
130  auto ds = std::make_tuple(internal::tuple_slice<1, std::tuple_size_v<Ts>>(ts)...);
131 
132  auto flattened_ds = std::apply([](const auto&...args) {return std::tuple_cat(args...); }, ds);
133  auto points_tuple = std::apply([&x](const auto&...args) {
134  return SamplePointsType::sample_points(x, args...);
135  }, flattened_ds);
136  constexpr auto dim = count_dim<InputDist>(flattened_ds);
137 
138  auto xpoints = std::get<0>(points_tuple);
139  auto flattened_ps = internal::tuple_slice<1, std::tuple_size_v<decltype(points_tuple)>>(points_tuple);
140  auto ps = std::apply([&](const auto&...args) { return construct_ps(flattened_ps, args...); }, ds);
141 
142  return transform_impl<dim, InputDist, 0, return_cross>(gs, xpoints, ps, x, ds);
143  }
144 
145  public:
146  /**
147  * Perform one or more consecutive sample points transforms.
148  * \tparam InputDist Input distribution.
149  * \tparam Ts A list of tuples containing (1) a transformation and (2) zero or more noise terms for that transformation.
150  **/
151  template<typename InputDist, typename...T_args, typename...Ts, std::enable_if_t<distribution<InputDist>, int> = 0>
152  auto operator()(const InputDist& x, const std::tuple<T_args...>& t, const Ts&...ts) const
153  {
154  return transform<false>(x, t, ts...);
155  }
156 
157  /**
158  * Perform one sample points transform.
159  * \tparam Trans The transformation on which the transform is based.
160  * \tparam InputDist Input distribution.
161  * \tparam NoiseDist Noise distributions.
162  **/
163  template<typename InputDist, typename Trans, typename ... NoiseDists,
164  std::enable_if_t<(distribution<InputDist> and ... and distribution<NoiseDists>) and
165  std::is_invocable_v<Trans, typename DistributionTraits<InputDist>::Mean, typename DistributionTraits<NoiseDists>::Mean...>, int> = 0>
166  auto operator()(const InputDist& x, const Trans& g, const NoiseDists& ...n) const
167  {
168  return transform<false>(x, std::tuple {g, n...});
169  }
170 
171  /**
172  * Perform one or more consecutive sample points transforms, also returning the cross-covariance.
173  * \tparam InputDist Input distribution.
174  * \tparam Ts A list of tuples containing (1) a transformation and (2) zero or more noise terms for that transformation.
175  **/
176  template<typename InputDist, typename...T_args, typename...Ts, std::enable_if_t<distribution<InputDist>, int> = 0>
177  auto transform_with_cross_covariance(const InputDist& x, const std::tuple<T_args...>& t, const Ts&...ts) const
178  {
179  return transform<true>(x, t, ts...);
180  }
181 
182  /**
183  * Perform one sample points transform, also returning the cross-covariance.
184  * \tparam Trans The transformation on which the transform is based.
185  * \tparam InputDist Input distribution.
186  * \tparam NoiseDist Noise distributions.
187  **/
188  template<typename InputDist, typename Trans, typename ... NoiseDists,
189  std::enable_if_t<(distribution<InputDist> and ... and distribution<NoiseDists>), int> = 0>
190  auto transform_with_cross_covariance(const InputDist& x, const Trans& g, const NoiseDists& ...n) const
191  {
192  return transform<true>(x, std::forward_as_tuple(g, n...));
193  }
194 
195  };
