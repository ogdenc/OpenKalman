220  typename RowCoefficients = void, typename ColumnCoefficients = void, typename M,
221  std::enable_if_t<is_typed_matrix_base_v<M>, int> = 0>
222  inline auto make_Matrix(M&& arg)
223  {
224  using RowCoeffs = std::conditional_t<std::is_void_v<RowCoefficients>,
225  Axes<MatrixTraits<M>::dimension>,
226  RowCoefficients>;
227  using ColCoeffs = std::conditional_t<std::is_void_v<ColumnCoefficients>,
228  Axes<MatrixTraits<M>::columns>,
229  ColumnCoefficients>;

231  static_assert(MatrixTraits<M>::columns == ColCoeffs::size);
232  return TypedMatrix<RowCoeffs, ColCoeffs, lvalue_or_strict_t<M>>(std::forward<M>(arg));

284  struct MatrixTraits<OpenKalman::TypedMatrix<RowCoeffs, ColCoeffs, NestedType>>

291  static constexpr std::size_t columns = MatrixTraits<BaseMatrix>::columns;
292  using Scalar = typename MatrixTraits<BaseMatrix>::Scalar;
293  static_assert(RowCoefficients::size == dimension);
294  static_assert(ColumnCoefficients::size == columns);
295 
296  template<std::size_t rows = dimension, std::size_t cols = columns, typename S = Scalar>
297  using StrictMatrix = typename MatrixTraits<BaseMatrix>::template StrictMatrix<rows, cols, S>;
298 
299  using Strict = TypedMatrix<RowCoefficients, ColumnCoefficients, typename MatrixTraits<BaseMatrix>::Strict>;

301  template<
302  typename RC = RowCoefficients,
303  typename CC = ColumnCoefficients,
304  typename Arg, std::enable_if_t<is_typed_matrix_base_v<Arg>, int> = 0>
305  static auto make(Arg&& arg)
306  {
307  static_assert(MatrixTraits<Arg>::dimension == RC::size);
308  if constexpr(MatrixTraits<Arg>::columns == CC::size)
309  {
310  return TypedMatrix<RC, CC, std::decay_t<Arg>>(std::forward<Arg>(arg));
311  }
312  else
313  {
314  return Mean<RC, std::decay_t<Arg>>(std::forward<Arg>(arg));
315  }
316  }

319 
320  static auto identity() { return make<RowCoefficients, RowCoefficients>(MatrixTraits<BaseMatrix>::identity()); }
