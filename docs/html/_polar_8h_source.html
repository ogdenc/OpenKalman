26  template<typename AngleTraits, typename Scalar>
27  static inline Scalar polar_angle_wrap_impl(const bool distance_is_negative, const Scalar s)
28  {
29  constexpr Scalar wrap_max = AngleTraits::template wrap_max<Scalar>;
30  constexpr Scalar wrap_min = AngleTraits::template wrap_min<Scalar>;
31  constexpr Scalar period = wrap_max - wrap_min;
32 
33  Scalar a = distance_is_negative ? s + period * 0.5 : s;
34 
35  if (a >= wrap_min and a < wrap_max) // Check if the angle doesn't need wrapping.
36  {
37  return a;
38  }
39  else // Wrap the angle.
40  {
41  Scalar ar = std::fmod(a - wrap_min, period);
42  if (ar < 0)
43  {
44  ar += period;
45  }
46  return ar + wrap_min;
47  }
48  }

88  template<typename AngleTraits, typename Scalar>
89  struct PolarImpl<AngleTraits, Circle<AngleTraits>, Scalar>
90  {
91  using GetCoeff = std::function<Scalar(const std::size_t)>;
92  using SetCoeff = std::function<void(const Scalar, const std::size_t)>;
93 
94  static constexpr Scalar cf =
95  2 * M_PI / (AngleTraits::template wrap_max<Scalar> - AngleTraits::template wrap_min<Scalar>);
96 
97  template<std::size_t i, std::size_t d_i, std::size_t a_i>
98  static constexpr std::array<Scalar (*const)(const GetCoeff&), 2>
99  to_Euclidean_array =
100  {
101  [](const GetCoeff& get_coeff) { return std::cos(get_coeff(i + a_i) * cf); },
102  [](const GetCoeff& get_coeff) { return std::sin(get_coeff(i + a_i) * cf); }
103  };
104 
105  template<std::size_t i, std::size_t d_i, std::size_t x_i, std::size_t y_i>
106  static constexpr std::array<Scalar (*const)(const GetCoeff&), 1>
107  from_Euclidean_array = {[](const GetCoeff& get_coeff)
108  {
109  const auto x = std::signbit(get_coeff(i + d_i)) ? -get_coeff(i + x_i) : get_coeff(i + x_i);
110  const auto y = std::signbit(get_coeff(i + d_i)) ? -get_coeff(i + y_i) : get_coeff(i + y_i);
111  return std::atan2(y, x) / cf;
112  }};
