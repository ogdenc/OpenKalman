18  /*************GeneralSigmaPoints*********
19  * @brief This class embodies a set of sample (e.g., sigma) points.
20  *
21  * Given random variable X:Ω->ℝⁿ, sample points S⊂ℝⁿ are a finite set of
22  * samples within ℝⁿ that are specifically arranged so that the
23  * weighted distribution of S encodes approximate statistical information
24  * (e.g., mean, standard deviation, etc.) about the distribution of X.
25  * As implemented in
26  * S. Julier. The scaled unscented transformation. In Proceedings of the American
27  * Control Conference, Evanston, IL, pages 1108–1114, 2002.
28  * This class has only static members, and is not to be instantiated.
29  */
30  template<typename SigmaPointsType>
31  struct SigmaPoints;

57  template<std::size_t dim, typename Weights>
58  static auto mean_weights()
59  {
60  using Scalar = typename MatrixTraits<Weights>::Scalar;
61  constexpr auto count = MatrixTraits<Weights>::dimension;
62  constexpr auto w0 = SigmaPointsType::template W_m0<dim, Scalar>();
63  return cat_weights<dim, Weights>(w0, std::make_index_sequence<count - 1>());
64  };

70  constexpr auto count = MatrixTraits<Weights>::dimension;
71  constexpr auto w0 = SigmaPointsType::template W_c0<dim, Scalar>();
72  return cat_weights<dim, Weights>(w0, std::make_index_sequence<count - 1>());

88  template<std::size_t dim, typename YMeans, std::enable_if_t<is_Euclidean_mean_v<YMeans>, int> = 0>
89  static auto
90  weighted_means(const YMeans& y_means)
91  {
92  static_assert(is_column_vector_v<YMeans>);
93  constexpr auto count = MatrixTraits<YMeans>::columns;
94  static_assert(count == SigmaPointsType::template sigma_point_count<dim>(), "Wrong number of sigma points.");
95  using Weights = TypedMatrix<Axes<count>, Axis, strict_matrix_t<YMeans, count, 1>>;
96  return strict(y_means * mean_weights<dim, Weights>());
97  }

99  template<std::size_t dim, typename InputDist, bool return_cross = false, typename X, typename Y>
100  static auto
101  covariance(const X& x_deviations, const Y& y_deviations)
102  {
103  static_assert(is_typed_matrix_v<X> and is_typed_matrix_v<Y>);
104  static_assert(is_equivalent_v<typename MatrixTraits<X>::RowCoefficients,
105  typename DistributionTraits<InputDist>::Coefficients>);
106  constexpr auto count = MatrixTraits<X>::columns;
107  static_assert(count == MatrixTraits<Y>::columns);
108  static_assert(count == SigmaPointsType::template sigma_point_count<dim>(), "Wrong number of sigma points.");
109  using Weights = TypedMatrix<Axes<count>, Axis, strict_matrix_t<X, count, 1>>;
110  auto weights = covariance_weights<dim, Weights>();
111  if constexpr(is_Cholesky_v<InputDist>)
112  {
113  using Scalar = typename MatrixTraits<X>::Scalar;
114  //
115  constexpr auto W_c0 = SigmaPointsType::template W_c0<dim, Scalar>();
116  if constexpr (W_c0 < 0)
117  {
118  // Discard first weight and first y-deviation column for now, since square root of weight would be negative.
119  const auto [y_deviations_head, y_deviations_tail] = split_horizontal<1, count - 1>(y_deviations);
120  const auto [weights_head, weights_tail] = split_vertical<Axis, Axes<count - 1>>(weights);
121  const auto sqrt_weights_tail = apply_coefficientwise(weights_tail, [](const auto x){ return std::sqrt(x); });
122  auto out_covariance = LQ_decomposition(y_deviations_tail * to_diagonal(sqrt_weights_tail));
123  rank_update(out_covariance, y_deviations_head, W_c0);
124  if constexpr (return_cross)
125  {
126  auto cross_covariance = strict(x_deviations * to_diagonal(weights) * adjoint(y_deviations));
127  return std::tuple{std::move(out_covariance), std::move(cross_covariance)};
128  }
129  else
130  {
131  return out_covariance;
132  }
133  }
134  else
135  {
136  const auto sqrt_weights = apply_coefficientwise(weights, [](const auto x){ return std::sqrt(x); });
137  auto out_covariance = Covariance {LQ_decomposition(y_deviations * to_diagonal(sqrt_weights))};
138  if constexpr (return_cross)
139  {
140  auto cross_covariance = strict(x_deviations * to_diagonal(weights) * adjoint(y_deviations));
141  return std::tuple{std::move(out_covariance), std::move(cross_covariance)};
142  }
143  else
144  {
145  return out_covariance;
146  }
147  }
148  }
149  else
150  {
151  const auto w_yT = to_diagonal(weights) * adjoint(y_deviations);
152  auto out_covariance = strict(make_Covariance(y_deviations * w_yT));
153  if constexpr (return_cross)
154  {
155  auto cross_covariance = strict(x_deviations * w_yT);
156  return std::tuple{std::move(out_covariance), std::move(cross_covariance)};
157  }
158  else
159  {
160  return out_covariance;
161  }
162  }
163  }
164 
165  };
